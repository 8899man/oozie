037    /**
038     * File system action executor.
039     * <p/>
040     * This executes the file system mkdir, move and delete commands
041     */
042    public class FsActionExecutor extends ActionExecutor {
043    
044        public FsActionExecutor() {
045            super("fs");
046        }
047    
048        Path getPath(Element element, String attribute) {
049            String str = element.getAttributeValue(attribute).trim();
050            return new Path(str);
051        }
052    
053        void validatePath(Path path, boolean withScheme) throws ActionExecutorException {
054            String scheme = path.toUri().getScheme();
055            if (withScheme) {
056                if (scheme == null) {
057                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS001",
058                                                      "Missing scheme in path [{0}]", path);
059                }
060                else if (!scheme.equals("hdfs")) {
061                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS002",
062                                                      "Scheme [{0}] not support in path [{1}]", scheme, path);
063                }
064            }
065            else {
066                if (scheme != null) {
067                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS003",
068                                                      "Scheme [{0}] not allowed in path [{1}]", scheme, path);
069                }
070            }
071        }
072    
073        @SuppressWarnings("unchecked")
074        void doOperations(Context context, Element element) throws ActionExecutorException {
075            try {
076                FileSystem fs = context.getAppFileSystem();
077                boolean recovery = fs.exists(getRecoveryPath(context));
078                if (!recovery) {
079                    fs.mkdirs(getRecoveryPath(context));
080                }
081                for (Element commandElement : (List<Element>) element.getChildren()) {
082                    String command = commandElement.getName();
083                    if (command.equals("mkdir")) {
084                        Path path = getPath(commandElement, "path");
085                        mkdir(context, path);
086                    }
087                    else if (command.equals("delete")) {
088                        Path path = getPath(commandElement, "path");
089                        delete(context, path);
090                    }
091                    else if (command.equals("move")) {
092                        Path source = getPath(commandElement, "source");
093                        Path target = getPath(commandElement, "target");
094                        move(context, source, target, recovery);
095                    }
096                    else if (command.equals("chmod")) {
097                        Path path = getPath(commandElement, "path");
098                        String str = commandElement.getAttributeValue("dir-files");
099                        boolean dirFiles = (str == null) || Boolean.parseBoolean(str);
100                        String permissionsMask = commandElement.getAttributeValue("permissions").trim();
101                        chmod(context, path, permissionsMask, dirFiles);
102                    }
103                }
104            }
105            catch (Exception ex) {
106                throw convertException(ex);
107            }
108        }
109    
110        private FileSystem getFileSystemFor(Path path, Context context) throws IOException {
111            String user = context.getWorkflow().getUser();
112            String group = context.getWorkflow().getGroup();
113            return Services.get().get(HadoopAccessorService.class).
114                    createFileSystem(user, group, path.toUri(), new Configuration());
115        }
116    
117        void mkdir(Context context, Path path) throws ActionExecutorException {
118            try {
119                validatePath(path, true);
120                FileSystem fs = getFileSystemFor(path, context);
121    
122                if (!fs.exists(path)) {
123                    if (!fs.mkdirs(path)) {
124                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS004",
125                                                          "mkdir, path [{0}] could not create directory", path);
126                    }
127                }
128            }
129            catch (Exception ex) {
130                throw convertException(ex);
131            }
132        }
133    
134        void delete(Context context, Path path) throws ActionExecutorException {
135            try {
136                validatePath(path, true);
137                FileSystem fs = getFileSystemFor(path, context);
138    
139                if (fs.exists(path)) {
140                    if (!fs.delete(path, true)) {
141                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS005",
142                                                          "delete, path [{0}] could not delete path", path);
143                    }
144                }
145            }
146            catch (Exception ex) {
147                throw convertException(ex);
148            }
149        }
150    
151    
152        void move(Context context, Path source, Path target, boolean recovery) throws ActionExecutorException {
153            try {
154                validatePath(source, true);
155                validatePath(target, false);
156                FileSystem fs = getFileSystemFor(source, context);
157    
158                if (!fs.exists(source) && !recovery) {
159                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS006",
160                                                      "move, source path [{0}] does not exist", source);
161                }
162    
163                Path path = new Path(source, target);
164                if (fs.exists(path) && !recovery) {
165                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS007",
166                                                      "move, target path [{0}] already exists", target);
167                }
168    
169                if (!fs.rename(source, target) && !recovery) {
170                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS008",
171                                                      "move, could not move [{0}] to [{1}]", source, target);
172                }
173            }
174            catch (Exception ex) {
175                throw convertException(ex);
176            }
177        }
178    
179        void chmod(Context context, Path path, String permissions, boolean dirFiles) throws ActionExecutorException {
180            try {
181                validatePath(path, true);
182                FileSystem fs = getFileSystemFor(path, context);
183    
184                if (!fs.exists(path)) {
185                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS009",
186                                                      "chmod, path [{0}] does not exist", path);
187                }
188    
189                FileStatus pathStatus = fs.getFileStatus(path);
190    
191                Path[] paths;
192                if (dirFiles && pathStatus.isDir()) {
193                    FileStatus[] filesStatus = fs.listStatus(path);
194                    paths = new Path[filesStatus.length];
195                    for (int i = 0; i < filesStatus.length; i++) {
196                        paths[i] = filesStatus[i].getPath();
197                    }
198                }
199                else {
200                    paths = new Path[]{path};
201                }
202    
203                FsPermission newFsPermission = createShortPermission(permissions, path);
204                fs.setPermission(path, newFsPermission);
205                for (Path p : paths) {
206                    fs.setPermission(p, newFsPermission);
207                }
208            }
209            catch (Exception ex) {
210                throw convertException(ex);
211            }
212        }
213    
214        FsPermission createShortPermission(String permissions, Path path) throws ActionExecutorException {
215            if (permissions.length() == 3) {
216                char user = permissions.charAt(0);
217                char group = permissions.charAt(1);
218                char other = permissions.charAt(2);
219                int useri = user - '0';
220                int groupi = group - '0';
221                int otheri = other - '0';
222                int mask = useri * 100 + groupi * 10 + otheri;
223                short omask = Short.parseShort(Integer.toString(mask), 8);
224                return new FsPermission(omask);
225            }
226            else if (permissions.length() == 10) {
227                return FsPermission.valueOf(permissions);
228            }
229            else {
230                throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FS010",
231                                                  "chmod, path [{0}] invalid permissions mask [{1}]", path, permissions);
232            }
233        }

235        @Override
236        public void check(Context context, WorkflowAction action) throws ActionExecutorException {
237        }
238    
239        @Override
240        public void kill(Context context, WorkflowAction action) throws ActionExecutorException {
241        }
242    
243        @Override
244        public void start(Context context, WorkflowAction action) throws ActionExecutorException {
245            try {
246                context.setStartData("-", "-", "-");
247                Element actionXml = XmlUtils.parseXml(action.getConf());
248                doOperations(context, actionXml);
249                context.setExecutionData("OK", null);
250            }
251            catch (Exception ex) {
252                throw convertException(ex);
253            }
254        }
255    
256        @Override
257        public void end(Context context, WorkflowAction action) throws ActionExecutorException {
258            String externalStatus = action.getExternalStatus();
259            WorkflowAction.Status status = externalStatus.equals("OK") ? WorkflowAction.Status.OK :
260                                           WorkflowAction.Status.ERROR;
261            context.setEndData(status, getActionSignal(status));
262            if (!context.getProtoActionConf().getBoolean("oozie.action.keep.action.dir", false)) {
263                try {
264                    FileSystem fs = context.getAppFileSystem();
265                    fs.delete(context.getActionDir(), true);
266                }
267                catch (Exception ex) {
268                    throw convertException(ex);
269                }
270            }
271        }
272    
273        public boolean isCompleted(String externalStatus) {
274            return true;
275        }
276    
277        Path getRecoveryPath(Context context) throws IOException, URISyntaxException {
278            return new Path(context.getActionDir(), "fs-" + context.getRecoveryId());
279        }
