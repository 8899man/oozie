109        /**
110         * Check ssh action status.
111         * 
112         * @param context action execution context.
113         * @param action action object.
114         */
115        @Override
116        public void check(Context context, WorkflowAction action) throws ActionExecutorException {
117            Status status = getActionStatus(context, action);
118            boolean captureOutput = false;
119            try {
120                Element eConf = XmlUtils.parseXml(action.getConf());
121                Namespace ns = eConf.getNamespace();
122                captureOutput = eConf.getChild("capture-output", ns) != null;
123            }
124            catch (JDOMException ex) {
125                throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "ERR_XML_PARSE_FAILED",
126                        "unknown error", ex);
127            }
128            XLog log = XLog.getLog(getClass());
129            log.debug("Capture Output: {0}", captureOutput);
130            if (status == Status.OK) {
131                if (captureOutput) {
132                    String outFile = getRemoteFileName(context, action, "stdout", false, true);
133                    String dataCommand = SSH_COMMAND_BASE + action.getTrackerUri() + " cat " + outFile;
134                    log.debug("Ssh command [{0}]", dataCommand);
135                    try {
136                        Process process = Runtime.getRuntime().exec(dataCommand.split("\\s"));
137                        StringBuffer buffer = new StringBuffer();
138                        boolean overflow = false;
139                        drainBuffers(process, buffer, null, maxLen);
140                        if (buffer.length() > maxLen) {
141                            overflow = true;
142                        }
143                        if (overflow) {
144                            throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR,
145                                    "ERR_OUTPUT_EXCEED_MAX_LEN", "unknown error");
146                        }
147                        context.setExecutionData(status.toString(), PropertiesUtils.stringToProperties(buffer.toString()));
148                    }
149                    catch (Exception ex) {
150                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "ERR_UNKNOWN_ERROR",
151                                "unknown error", ex);
152                    }
153                }
154                else {
155                    context.setExecutionData(status.toString(), null);
156                }
157            }
158            else if (status == Status.ERROR) {
159                context.setExecutionData(status.toString(), null);
160            }
161            else {
162                context.setExternalStatus(status.toString());
163            }
164        }

166        /**
167         * Kill ssh action.
168         * 
169         * @param context action execution context.
170         * @param action object.
171         */
172        @Override
173        public void kill(Context context, WorkflowAction action) throws ActionExecutorException {
174            String command = "ssh " + action.getTrackerUri() + " kill  -KILL " + action.getExternalId();
175            int returnValue = getReturnValue(command);
176            if (returnValue != 0) {
177                throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FAILED_TO_KILL", XLog.format(
178                        "Unable to kill process {0} on {1}", action.getExternalId(), action.getTrackerUri()));
179            }
180            context.setEndData(WorkflowAction.Status.KILLED, "ERROR");
181        }

183        /**
184         * Start the ssh action execution.
185         * 
186         * @param context action execution context.
187         * @param action action object.
188         */
189        @SuppressWarnings("unchecked")
190        @Override
191        public void start(final Context context, final WorkflowAction action) throws ActionExecutorException {
192            XLog log = XLog.getLog(getClass());
193            log.info("start() begins");
194            String confStr = action.getConf();
195            Element conf;
196            try {
197                conf = XmlUtils.parseXml(confStr);
198            }
199            catch (Exception ex) {
200                throw convertException(ex);
201            }
202            Namespace nameSpace = conf.getNamespace();
203            Element hostElement = conf.getChild("host", nameSpace);
204            String hostString = hostElement.getValue().trim();
205            hostString = prepareUserHost(hostString, context);
206            final String host = hostString;
207            final String dirLocation = execute(new Callable<String>() {
208                public String call() throws Exception {
209                    return setupRemote(host, context, action);
210                }
211    
212            });
213    
214            String runningPid = execute(new Callable<String>() {
215                public String call() throws Exception {
216                    return checkIfRunning(host, context, action);
217                }
218            });
219            String pid = "";
220    
221            if (runningPid == null) {
222                final Element commandElement = conf.getChild("command", nameSpace);
223                final boolean ignoreOutput = conf.getChild("capture-output", nameSpace) == null;
224    
225                if (commandElement != null) {
226                    List<Element> argsList = conf.getChildren("args", nameSpace);
227                    StringBuilder args = new StringBuilder("");
228                    if ((argsList != null) && (argsList.size() > 0)) {
229                        for (Element argsElement : argsList) {
230                            args = args.append(argsElement.getValue()).append(" ");
231                        }
232                        args.setLength(args.length() - 1);
233                    }
234                    final String argsString = args.toString();
235                    final String recoveryId = context.getRecoveryId();
236                    pid = execute(new Callable<String>() {
237    
238                        @Override
239                        public String call() throws Exception {
240                            return doExecute(host, dirLocation, commandElement.getValue(), argsString, ignoreOutput,
241                                    action, recoveryId);
242                        }
243    
244                    });
245                }
246                context.setStartData(pid, host, host);
247            }
248            else {
249                pid = runningPid;
250                context.setStartData(pid, host, host);
251                check(context, action);
252            }
253            log.info("start() ends");
254        }

256        private String checkIfRunning(String host, final Context context, final WorkflowAction action) {
257            String pid = null;
258            String outFile = getRemoteFileName(context, action, "pid", false, false);
259            String getOutputCmd = SSH_COMMAND_BASE + host + " cat " + outFile;
260            try {
261                Process process = Runtime.getRuntime().exec(getOutputCmd.split("\\s"));
262                StringBuffer buffer = new StringBuffer();
263                drainBuffers(process, buffer, null, maxLen);
264                pid = getFirstLine(buffer);
265    
266                if (Long.valueOf(pid) > 0) {
267                    return pid;
268                }
269                else {
270                    return null;
271                }
272            }
273            catch (Exception e) {
274                return null;
275            }
276        }

278        /**
279         * Get remote host working location.
280         * 
281         * @param context action execution context
282         * @param action Action
283         * @param fileExtension Extension to be added to file name
284         * @param dirOnly Get the Directory only
285         * @param useExtId Flag to use external ID in the path
286         * @return remote host file name/Directory.
287         */
288        public String getRemoteFileName(Context context, WorkflowAction action, String fileExtension, boolean dirOnly,
289                boolean useExtId) {
290            String path = getActionDirPath(context.getWorkflow().getId(), action, ACTION_TYPE, false) + "/";
291            if (dirOnly) {
292                return path;
293            }
294            if (useExtId) {
295                path = path + action.getExternalId() + ".";
296            }
297            path = path + context.getRecoveryId() + "." + fileExtension;
298            return path;
299        }

301        /**
302         * Utility method to execute command.
303         * 
304         * @param command Command to execute as String.
305         * @throws IOException if process exits with status nonzero.
306         * @throws InterruptedException if process does not run properly.
307         * @return exit status of the execution.
308         */
309        public int executeCommand(String command) throws IOException, InterruptedException {
310            Runtime runtime = Runtime.getRuntime();
311            Process p = runtime.exec(command.split("\\s"));
312    
313            StringBuffer errorBuffer = new StringBuffer();
314            int exitValue = drainBuffers(p, null, errorBuffer, maxLen);
315    
316            String error = null;
317            if (exitValue != 0) {
318                error = getTruncatedString(errorBuffer);
319                throw new IOException(XLog.format("Not able to perform operation [{0}]", command) + " | " + "ErrorStream: "
320                        + error);
321            }
322            return exitValue;
323        }

325        /**
326         * Do ssh action execution setup on remote host.
327         * 
328         * @param host host name.
329         * @param context action execution context.
330         * @param action action object.
331         * @return remote host working directory.
332         * @throws IOException thrown if failed to setup.
333         * @throws InterruptedException thrown if any interruption happens.
334         */
335        protected String setupRemote(String host, Context context, WorkflowAction action) throws IOException, InterruptedException {
336            XLog log = XLog.getLog(getClass());
337            log.info("Attempting to copy ssh base scripts to remote host [{0}]", host);
338            String localDirLocation = Services.get().getRuntimeDir() + "/ssh";
339            if (localDirLocation.endsWith("/")) {
340                localDirLocation = localDirLocation.substring(0, localDirLocation.length() - 1);
341            }
342            File file = new File(localDirLocation + "/ssh-base.sh");
343            if (!file.exists()) {
344                throw new IOException("Required Local file " + file.getAbsolutePath() + " not present.");
345            }
346            file = new File(localDirLocation + "/ssh-wrapper.sh");
347            if (!file.exists()) {
348                throw new IOException("Required Local file " + file.getAbsolutePath() + " not present.");
349            }
350            String remoteDirLocation = getRemoteFileName(context, action, null, true, true);
351            String command = XLog.format("{0}{1}  mkdir -p {2} ", SSH_COMMAND_BASE, host, remoteDirLocation).toString();
352            executeCommand(command);
353            command = XLog.format("{0}{1}/ssh-base.sh {2}/ssh-wrapper.sh {3}:{4}", SCP_COMMAND_BASE, localDirLocation,
354                    localDirLocation, host, remoteDirLocation);
355            executeCommand(command);
356            command = XLog.format("{0}{1}  chmod +x {2}ssh-base.sh {3}ssh-wrapper.sh ", SSH_COMMAND_BASE, host,
357                    remoteDirLocation, remoteDirLocation);
358            executeCommand(command);
359            return remoteDirLocation;
360        }

362        /**
363         * Execute the ssh command.
364         * 
365         * @param host hostname.
366         * @param dirLocation location of the base and wrapper scripts.
367         * @param cmnd command to be executed.
368         * @param args command arguments.
369         * @param ignoreOutput ignore output option.
370         * @param action action object.
371         * @param recoveryId action id + run number to enable recovery in rerun
372         * @return process id of the running command.
373         * @throws IOException thrown if failed to run the command.
374         * @throws InterruptedException thrown if any interruption happens.
375         */
376        protected String doExecute(String host, String dirLocation, String cmnd, String args, boolean ignoreOutput,
377                WorkflowAction action, String recoveryId) throws IOException, InterruptedException {
378            XLog log = XLog.getLog(getClass());
379            Runtime runtime = Runtime.getRuntime();
380            String callbackPost = ignoreOutput ? "_" : getOozieConf().get(HTTP_COMMAND_OPTIONS).replace(" ", "%%%");
381            // TODO check
382            String callBackUrl = Services.get().get(CallbackService.class)
383                    .createCallBackUrl(action.getId(), EXT_STATUS_VAR);
384            String command = XLog.format("{0}{1} {2}ssh-base.sh {3} \"{4}\" \"{5}\" {6} {7} {8} ", SSH_COMMAND_BASE, host,
385                    dirLocation, getOozieConf().get(HTTP_COMMAND), callBackUrl, callbackPost, recoveryId, cmnd, args)
386                    .toString();
387            log.trace("Executing ssh command [{0}]", command);
388            Process p = runtime.exec(command.split("\\s"));
389            String pid = "";
390    
391            StringBuffer inputBuffer = new StringBuffer();
392            StringBuffer errorBuffer = new StringBuffer();
393            int exitValue = drainBuffers(p, inputBuffer, errorBuffer, maxLen);
394    
395            pid = getFirstLine(inputBuffer);
396    
397            String error = null;
398            if (exitValue != 0) {
399                error = getTruncatedString(errorBuffer);
400                throw new IOException(XLog.format("Not able to execute ssh-base.sh on {0}", host) + " | " + "ErrorStream: "
401                        + error);
402            }
403            return pid;
404        }

406        /**
407         * End action execution.
408         * 
409         * @param context action execution context.
410         * @param action action object.
411         * @throws ActionExecutorException thrown if action end execution fails.
412         */
413        public void end(final Context context, final WorkflowAction action) throws ActionExecutorException {
414            if (action.getExternalStatus().equals("OK")) {
415                context.setEndData(WorkflowAction.Status.OK, WorkflowAction.Status.OK.toString());
416            }
417            else {
418                context.setEndData(WorkflowAction.Status.ERROR, WorkflowAction.Status.ERROR.toString());
419            }
420            boolean deleteTmpDir = getOozieConf().getBoolean(DELETE_TMP_DIR, true);
421            if (deleteTmpDir) {
422                String tmpDir = getRemoteFileName(context, action, null, true, false);
423                String removeTmpDirCmd = SSH_COMMAND_BASE + action.getTrackerUri() + " rm -rf " + tmpDir;
424                int retVal = getReturnValue(removeTmpDirCmd);
425                if (retVal != 0) {
426                    XLog.getLog(getClass()).warn("Cannot delete temp dir {0}", tmpDir);
427                }
428            }
429        }

431        /**
432         * Get the return value of a process.
433         * 
434         * @param command command to be executed.
435         * @return zero if execution is successful and any non zero value for
436         *         failure.
437         * @throws ActionExecutorException
438         */
439        private int getReturnValue(String command) throws ActionExecutorException {
440            int returnValue;
441            Process ps = null;
442            try {
443                ps = Runtime.getRuntime().exec(command.split("\\s"));
444                returnValue = drainBuffers(ps, null, null, 0);
445            }
446            catch (IOException e) {
447                throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "FAILED_OPERATION", XLog.format(
448                        "Not able to perform operation {0}", command), e);
449            }
450            finally {
451                ps.destroy();
452            }
453            return returnValue;
454        }

456        /**
457         * Copy the ssh base and wrapper scripts to the local directory.
458         */
459        private void initSshScripts() {
460            String dirLocation = Services.get().getRuntimeDir() + "/ssh";
461            File path = new File(dirLocation);
462            if (!path.mkdirs()) {
463                throw new RuntimeException(XLog.format("Not able to create required directory {0}", dirLocation));
464            }
465            try {
466                IOUtils.copyCharStream(IOUtils.getResourceAsReader("ssh-base.sh", -1), new FileWriter(dirLocation
467                        + "/ssh-base.sh"));
468                IOUtils.copyCharStream(IOUtils.getResourceAsReader("ssh-wrapper.sh", -1), new FileWriter(dirLocation
469                        + "/ssh-wrapper.sh"));
470            }
471            catch (IOException ie) {
472                throw new RuntimeException(XLog.format("Not able to copy required scripts file to {0} "
473                        + "for SshActionHandler", dirLocation));
474            }
475        }

477        /**
478         * Get action status.
479         * 
480         * @param action action object.
481         * @return status of the action(RUNNING/OK/ERROR).
482         * @throws ActionExecutorException thrown if there is any error in getting
483         *         status.
484         */
485        protected Status getActionStatus(Context context, WorkflowAction action) throws ActionExecutorException {
486            String command = SSH_COMMAND_BASE + action.getTrackerUri() + " ps -p " + action.getExternalId();
487            Status aStatus;
488            int returnValue = getReturnValue(command);
489            if (returnValue == 0) {
490                aStatus = Status.RUNNING;
491            }
492            else {
493                String outFile = getRemoteFileName(context, action, "error", false, true);
494                String checkErrorCmd = SSH_COMMAND_BASE + action.getTrackerUri() + " ls " + outFile;
495                int retVal = getReturnValue(checkErrorCmd);
496                if (retVal == 0) {
497                    aStatus = Status.ERROR;
498                }
499                else {
500                    aStatus = Status.OK;
501                }
502            }
503            return aStatus;
504        }

506        /**
507         * Execute the callable.
508         * 
509         * @param callable required callable.
510         * @throws ActionExecutorException thrown if there is any error in command
511         *         execution.
512         */
513        private <T> T execute(Callable<T> callable) throws ActionExecutorException {
514            XLog log = XLog.getLog(getClass());
515            try {
516                return callable.call();
517            }
518            catch (IOException ex) {
519                log.warn("Error while executing ssh EXECUTION");
520                String errorMessage = ex.getMessage();
521                if (null == errorMessage) { // Unknown IOException
522                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, ERR_UNKNOWN_ERROR, ex
523                            .getMessage(), ex);
524                } // Host Resolution Issues
525                else if (errorMessage.contains("Could not resolve hostname") ||
526                         errorMessage.contains("service not known")) {
527                    throw new ActionExecutorException(ActionExecutorException.ErrorType.TRANSIENT, ERR_HOST_RESOLUTION, ex
528                            .getMessage(), ex);
529                } // Connection Timeout. Host temporarily down.
530                else if (errorMessage.contains("timed out")) {
531                    throw new ActionExecutorException(ActionExecutorException.ErrorType.TRANSIENT, ERR_COULD_NOT_CONNECT,
532                            ex.getMessage(), ex);
533                }// Local ssh-base or ssh-wrapper missing
534                else if (errorMessage.contains("Required Local file")) {
535                    throw new ActionExecutorException(ActionExecutorException.ErrorType.TRANSIENT, ERR_FNF,
536                            ex.getMessage(), ex); // local_FNF
537                }// Required oozie bash scripts missing, after the copy was
538                // successful
539                else if (errorMessage.contains("No such file or directory")
540                        && (errorMessage.contains("ssh-base") || errorMessage.contains("ssh-wrapper"))) {
541                    throw new ActionExecutorException(ActionExecutorException.ErrorType.TRANSIENT, ERR_FNF,
542                            ex.getMessage(), ex); // remote
543                    // FNF
544                } // Required application execution binary missing (either
545                // caught by ssh-wrapper
546                else if (errorMessage.contains("command not found")) {
547                    throw new ActionExecutorException(ActionExecutorException.ErrorType.NON_TRANSIENT, ERR_FNF, ex
548                            .getMessage(), ex); // remote
549                    // FNF
550                } // Permission denied while connecting
551                else if (errorMessage.contains("Permission denied")) {
552                    throw new ActionExecutorException(ActionExecutorException.ErrorType.NON_TRANSIENT, ERR_AUTH_FAILED, ex
553                            .getMessage(), ex);
554                } // Permission denied while executing
555                else if (errorMessage.contains(": Permission denied")) {
556                    throw new ActionExecutorException(ActionExecutorException.ErrorType.NON_TRANSIENT, ERR_NO_EXEC_PERM, ex
557                            .getMessage(), ex);
558                }
559                else {
560                    throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, ERR_UNKNOWN_ERROR, ex
561                            .getMessage(), ex);
562                }
563            } // Any other type of exception
564            catch (Exception ex) {
565                throw convertException(ex);
566            }
567        }

569        /**
570         * Checks whether the system is configured to always use the oozie user for
571         * ssh, and injects the user if required.
572         * 
573         * @param host the host string.
574         * @param context the execution context.
575         * @return the modified host string with a user parameter added on if
576         *         required.
577         * @throws ActionExecutorException in case the flag to use the oozie user is
578         *         turned on and there is a mismatch between the user specified in
579         *         the host and the oozie user.
580         */
581        private String prepareUserHost(String host, Context context) throws ActionExecutorException {
582            String oozieUser = context.getProtoActionConf().get(OozieClient.USER_NAME);
583            if (allowSshUserAtHost) {
584                if (!host.contains("@")) {
585                    host = oozieUser + "@" + host;
586                }
587            }
588            else{
589                if (host.contains("@")) {
590                    if (!host.toLowerCase().startsWith(oozieUser + "@")) {
591                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, ERR_USER_MISMATCH,
592                                XLog.format("user mismatch between oozie user [{0}] and ssh host [{1}]", oozieUser, host));
593                    }
594                }
595                else {
596                    host = oozieUser + "@" + host;
597                }
598            }
599            return host;
600        }

606        /**
607         * Truncate the string to max length.
608         *
609         * @param strBuffer
610         * @return truncated string string
611         */
612        private String getTruncatedString(StringBuffer strBuffer) {
613    
614            if (strBuffer.length() <= maxLen) {
615                return strBuffer.toString();
616            }
617            else {
618                return strBuffer.substring(0, maxLen);
619            }
620        }

622        /**
623         * Drains the inputStream and errorStream of the Process being executed. The
624         * contents of the streams are stored if a buffer is provided for the
625         * stream.
626         *
627         * @param p The Process instance.
628         * @param inputBuffer The buffer into which STDOUT is to be read. Can be
629         *        null if only draining is required.
630         * @param errorBuffer The buffer into which STDERR is to be read. Can be
631         *        null if only draining is required.
632         * @param maxLength The maximum data length to be stored in these buffers.
633         *        This is an indicative value, and the store content may exceed this
634         *        length.
635         * @return the exit value of the process.
636         * @throws IOException
637         */
638        private int drainBuffers(Process p, StringBuffer inputBuffer, StringBuffer errorBuffer, int maxLength)
639                throws IOException {
640            int exitValue = -1;
641            BufferedReader ir = new BufferedReader(new InputStreamReader(p.getInputStream()));
642            BufferedReader er = new BufferedReader(new InputStreamReader(p.getErrorStream()));
643    
644            int inBytesRead = 0;
645            int errBytesRead = 0;
646    
647            boolean processEnded = false;
648    
649            try {
650                while (!processEnded) {
651                    try {
652                        exitValue = p.exitValue();
653                        processEnded = true;
654                    }
655                    catch (IllegalThreadStateException ex) {
656                        // Continue to drain.
657                    }
658    
659                    inBytesRead += drainBuffer(ir, inputBuffer, maxLength, inBytesRead, processEnded);
660                    errBytesRead += drainBuffer(er, errorBuffer, maxLength, errBytesRead, processEnded);
661                }
662            }
663            finally {
664                ir.close();
665                er.close();
666            }
667            return exitValue;
668        }

670        /**
671         * Reads the contents of a stream and stores them into the provided buffer.
672         *
673         * @param br The stream to be read.
674         * @param storageBuf The buffer into which the contents of the stream are to
675         *        be stored.
676         * @param maxLength The maximum number of bytes to be stored in the buffer.
677         *        An indicative value and may be exceeded.
678         * @param bytesRead The number of bytes read from this stream to date.
679         * @param readAll If true, the stream is drained while their is data
680         *        available in it. Otherwise, only a single chunk of data is read,
681         *        irrespective of how much is available.
682         * @return
683         * @throws IOException
684         */
685        private int drainBuffer(BufferedReader br, StringBuffer storageBuf, int maxLength, int bytesRead, boolean readAll)
686                throws IOException {
687            int bReadSession = 0;
688            if (br.ready()) {
689                char[] buf = new char[1024];
690                do {
691                    int bReadCurrent = br.read(buf, 0, 1024);
692                    if (storageBuf != null && bytesRead < maxLength) {
693                        storageBuf.append(buf, 0, bReadCurrent);
694                    }
695                    bReadSession += bReadCurrent;
696                } while (br.ready() && readAll);
697            }
698            return bReadSession;
699        }

701        /**
702         * Returns the first line from a StringBuffer, recognized by the new line
703         * character \n.
704         *
705         * @param buffer The StringBuffer from which the first line is required.
706         * @return The first line of the buffer.
707         */
708        private String getFirstLine(StringBuffer buffer) {
709            int newLineIndex = 0;
710            newLineIndex = buffer.indexOf("\n");
711            if (newLineIndex == -1) {
712                return buffer.toString();
713            }
714            else {
715                return buffer.substring(0, newLineIndex);
716            }
717        }
