201        /**
202         * Run a CLI programmatically.
203         * <p/>
204         * It does not exit the JVM.
205         * <p/>
206         * A CLI instance can be used only once.
207         *
208         * @param args options and arguments for the Oozie CLI.
209         * @return '0' (success), '-1' (failure).
210         */
211        public synchronized int run(String[] args) {
212            if (used) {
213                throw new IllegalStateException("CLI instance already used");
214            }
215            used = true;
216            CLIParser parser = new CLIParser(OOZIE_OPTION, getCLIHelp());
217            parser.addCommand(HELP_CMD, "", "display usage", new Options(), false);
218            parser.addCommand(VERSION_CMD, "", "show client version", new Options(), false);
219            parser.addCommand(JOB_CMD, "", "job operations", createJobOptions(), false);
220            parser.addCommand(JOBS_CMD, "", "jobs status", createJobsOptions(), false);
221            parser.addCommand(ADMIN_CMD, "", "admin operations", createAdminOptions(), false);
222            parser.addCommand(VALIDATE_CMD, "", "validate a workflow XML file", new Options(), true);
223    
224            try {
225                CLIParser.Command command = parser.parse(args);
226                if (command.getName().equals(HELP_CMD)) {
227                    parser.showHelp();
228                }
229                else if (command.getName().equals(JOB_CMD)) {
230                    jobCommand(command.getCommandLine());
231                }
232                else if (command.getName().equals(JOBS_CMD)) {
233                    jobsCommand(command.getCommandLine());
234                }
235                else if (command.getName().equals(ADMIN_CMD)) {
236                    adminCommand(command.getCommandLine());
237                }
238                else if (command.getName().equals(VERSION_CMD)) {
239                    versionCommand();
240                }
241                else if (command.getName().equals(VALIDATE_CMD)) {
242                    validateCommand(command.getCommandLine());
243                }
244                return 0;
245            }
246            catch (OozieCLIException ex) {
247                System.err.println("Error: " + ex.getMessage());
248                return -1;
249            }
250            catch (ParseException ex) {
251                System.err.println("Invalid sub-command: " + ex.getMessage());
252                System.err.println();
253                System.err.println(parser.shortHelp());
254                return -1;
255            }
256            catch (Exception ex) {
257                System.err.println(ex.getMessage());
258                return -1;
259            }
260        }

262        private String getOozieUrl(CommandLine commandLine) {
263            String url = commandLine.getOptionValue(OOZIE_OPTION);
264            if (url == null) {
265                url = System.getenv(ENV_OOZIE_URL);
266                if (url == null) {
267                    throw new IllegalArgumentException("Oozie URL is no available as option or in the environment");
268                }
269            }
270            return url;
271        }

273        // Canibalized from Hadoop <code>Configuration.loadResource()</code>.
274        private Properties parse(InputStream is, Properties conf) throws IOException {
275            try {
276                DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
277                // ignore all comments inside the xml file
278                docBuilderFactory.setIgnoringComments(true);
279                DocumentBuilder builder = docBuilderFactory.newDocumentBuilder();
280                Document doc = builder.parse(is);
281                return parseDocument(doc, conf);
282            }
283            catch (SAXException e) {
284                throw new IOException(e);
285            }
286            catch (ParserConfigurationException e) {
287                throw new IOException(e);
288            }
289        }

291        // Canibalized from Hadoop <code>Configuration.loadResource()</code>.
292        private Properties parseDocument(Document doc, Properties conf) throws IOException {
293            try {
294                Element root = doc.getDocumentElement();
295                if (!"configuration".equals(root.getTagName())) {
296                    throw new RuntimeException("bad conf file: top-level element not <configuration>");
297                }
298                NodeList props = root.getChildNodes();
299                for (int i = 0; i < props.getLength(); i++) {
300                    Node propNode = props.item(i);
301                    if (!(propNode instanceof Element)) {
302                        continue;
303                    }
304                    Element prop = (Element) propNode;
305                    if (!"property".equals(prop.getTagName())) {
306                        throw new RuntimeException("bad conf file: element not <property>");
307                    }
308                    NodeList fields = prop.getChildNodes();
309                    String attr = null;
310                    String value = null;
311                    for (int j = 0; j < fields.getLength(); j++) {
312                        Node fieldNode = fields.item(j);
313                        if (!(fieldNode instanceof Element)) {
314                            continue;
315                        }
316                        Element field = (Element) fieldNode;
317                        if ("name".equals(field.getTagName()) && field.hasChildNodes()) {
318                            attr = ((Text) field.getFirstChild()).getData().trim();
319                        }
320                        if ("value".equals(field.getTagName()) && field.hasChildNodes()) {
321                            value = ((Text) field.getFirstChild()).getData();
322                        }
323                    }
324    
325                    if (attr != null && value != null) {
326                        conf.setProperty(attr, value);
327                    }
328                }
329                return conf;
330            }
331            catch (DOMException e) {
332                throw new IOException(e);
333            }
334        }

336        private Properties getConfiguration(CommandLine commandLine) throws IOException {
337            Properties conf = new Properties();
338            conf.setProperty("user.name", System.getProperty("user.name"));
339            String configFile = commandLine.getOptionValue(CONFIG_OPTION);
340            if (configFile == null) {
341                throw new IOException("configuration file not specified");
342            }
343            else {
344                File file = new File(configFile);
345                if (!file.exists()) {
346                    throw new IOException("configuration file [" + configFile + "] not found");
347                }
348                if (configFile.endsWith(".properties")) {
349                    conf.load(new FileReader(file));
350                }
351                else if (configFile.endsWith(".xml")) {
352                    parse(new FileInputStream(configFile), conf);
353                }
354                else {
355                    throw new IllegalArgumentException("configuration must be a '.properties' or a '.xml' file");
356                }
357            }
358            return conf;
359        }

361        /**
362         * Create a OozieClient.
363         * <p/>
364         * It injects any '-Dheader:' as header to the the {@link org.apache.oozie.client.OozieClient}.
365         *
366         * @param commandLine the parsed command line options.
367         * @return a pre configured workflow client.
368         * @throws OozieCLIException thrown if the Oozie client could not be configured.
369         */
370        protected OozieClient createOozieClient(CommandLine commandLine) throws OozieCLIException {
371            OozieClient wc = new OozieClient(getOozieUrl(commandLine));
372            for (Map.Entry entry : System.getProperties().entrySet()) {
373                String key = (String) entry.getKey();
374                if (key.startsWith(WS_HEADER_PREFIX)) {
375                    String header = key.substring(WS_HEADER_PREFIX.length());
376                    wc.setHeader(header, (String) entry.getValue());
377                }
378            }
379            return wc;
380        }

382        private static String JOB_ID_PREFIX = "job: ";
383    
384        private void jobCommand(CommandLine commandLine) throws IOException, OozieCLIException {
385            OozieClient wc = createOozieClient(commandLine);
386    
387            List<String> options = new ArrayList<String>();
388            for (Option option : commandLine.getOptions()) {
389                options.add(option.getOpt());
390            }
391    
392            try {
393                if (options.contains(SUBMIT_OPTION)) {
394                    System.out.println(JOB_ID_PREFIX + wc.submit(getConfiguration(commandLine)));
395                }
396                else if (options.contains(START_OPTION)) {
397                    wc.start(commandLine.getOptionValue(START_OPTION));
398                }
399                else if (options.contains(SUSPEND_OPTION)) {
400                    wc.suspend(commandLine.getOptionValue(SUSPEND_OPTION));
401                }
402                else if (options.contains(RESUME_OPTION)) {
403                    wc.resume(commandLine.getOptionValue(RESUME_OPTION));
404                }
405                else if (options.contains(KILL_OPTION)) {
406                    wc.kill(commandLine.getOptionValue(KILL_OPTION));
407                }
408                else if (options.contains(RUN_OPTION)) {
409                    System.out.println(JOB_ID_PREFIX + wc.run(getConfiguration(commandLine)));
410                }
411                else if (options.contains(RERUN_OPTION)) {
412                    wc.reRun(commandLine.getOptionValue(RERUN_OPTION), getConfiguration(commandLine));
413                }
414                else if (options.contains(INFO_OPTION)) {
415                    printJob(wc.getJobInfo(commandLine.getOptionValue(INFO_OPTION)), options.contains(LOCAL_TIME_OPTION));
416                }
417            }
418            catch (OozieClientException ex) {
419                throw new OozieCLIException(ex.toString(), ex);
420            }
421        }

429        private void printJob(WorkflowJob job, boolean localtime) throws IOException {
430            System.out.println("Job Id: " + job.getId());
431            System.out.println(RULER);
432    
433            System.out.println(String.format(JOB_FORMATTER, "Workflow Name", job.getAppName()));
434            System.out.println(String.format(JOB_FORMATTER, "App Path", job.getAppPath()));
435            System.out.println(String.format(JOB_FORMATTER, "Status", job.getStatus()));
436            System.out.println(String.format(JOB_FORMATTER, "Run", job.getRun()));
437            System.out.println(String.format(JOB_FORMATTER, "User", job.getUser()));
438            System.out.println(String.format(JOB_FORMATTER, "Group", job.getGroup()));
439            System.out.println(String.format(JOB_FORMATTER, "Created", maskDate(job.getCreatedTime(), localtime)));
440            System.out.println(String.format(JOB_FORMATTER, "Started", maskDate(job.getStartTime(), localtime)));
441            System.out.println(String.format(JOB_FORMATTER, "Last Modified", maskDate(job.getLastModTime(), localtime)));
442            System.out.println(String.format(JOB_FORMATTER, "Ended", maskDate(job.getEndTime(), localtime)));
443    
444            List<WorkflowAction> actions = job.getActions();
445            if(actions!=null && actions.size()>0){
446                System.out.println();
447                System.out.println("Actions");
448                System.out.println(RULER);
449                System.out.println(String.format(ACTION_FORMATTER,
450                        "Action Name", "Type", "Status", "Transition", 
451                        "Ext. Id", "Ext. Status", "Error Code", 
452                        "Started", "Ended"));
453                System.out.println(RULER);
454                
455                for(WorkflowAction action:job.getActions()){
456                    System.out.println(String.format(ACTION_FORMATTER, action.getName(),
457                            action.getType(), action.getStatus(), maskIfNull(action.getTransition()),
458                            maskIfNull(action.getExternalId()), maskIfNull(action.getExternalStatus()),
459                            maskIfNull(action.getErrorCode()), maskDate(action.getStartTime(), localtime),
460                            maskDate(action.getEndTime(), localtime)));
461                    System.out.println(RULER);
462                }
463            }
464            else {
465                    System.out.println(RULER);
466            }
467            System.out.println();
468        }

470        private void jobsCommand(CommandLine commandLine) throws IOException, OozieCLIException {
471            OozieClient wc = createOozieClient(commandLine);
472    
473            String filter = commandLine.getOptionValue(FILTER_OPTION);
474            String s = commandLine.getOptionValue(OFFSET_OPTION);
475            int start = Integer.parseInt((s != null) ? s : "0");
476            s = commandLine.getOptionValue(LEN_OPTION);
477            int len = Integer.parseInt((s != null) ? s : "0");
478    
479            try {
480                printJobs(wc.getJobsInfo(filter, start, len), commandLine.hasOption(LOCAL_TIME_OPTION));
481            }
482            catch (OozieClientException ex) {
483                throw new OozieCLIException(ex.toString(), ex);
484            }
485        }

487        private void adminCommand(CommandLine commandLine) throws OozieCLIException {
488            OozieClient wc = createOozieClient(commandLine);
489    
490            List<String> options = new ArrayList<String>();
491            for (Option option : commandLine.getOptions()) {
492                options.add(option.getOpt());
493            }
494    
495            try {
496                boolean status = false;
497                if (options.contains(VERSION_OPTION)) {
498                    System.out.println("Oozie server build version: " + wc.getServerBuildVersion());
499                }
500                else {
501                    if (options.contains(SAFEMODE_OPTION)) {
502                        String safeModeOption = commandLine.getOptionValue(SAFEMODE_OPTION);
503                        try {
504                            status = safeModeOption.equalsIgnoreCase("ON");
505                        }
506                        catch (Exception e) {
507                            throw new OozieCLIException("Invalid input provided for option: " + SAFEMODE_OPTION);
508                        }
509                        wc.setSafeMode(status);
510    
511                    }
512                    else if (options.contains(STATUS_OPTION)) {
513                        status = wc.isInSafeMode();
514                    }
515                    System.out.println("Safemode: " + (status ? "ON" : "OFF"));
516                }
517            }
518            catch (OozieClientException ex) {
519                throw new OozieCLIException(ex.toString(), ex);
520            }
521        }

528        private void printJobs(List<WorkflowJob> jobs, boolean localtime) throws IOException {
529            if(jobs!=null && jobs.size() > 0) {
530                System.out.println(String.format(JOBS_FORMATTER, "Job Id", "Workflow Name", "Status", "Run", "User",
531                        "Group", "Created", "Started", "Last Modified", "Ended"));
532                System.out.println(RULER);
533    
534                for (WorkflowJob job : jobs) {
535    
536                    System.out.println(String.format(JOBS_FORMATTER,
537                            job.getId(), job.getAppName(), job.getStatus().toString(),
538                            job.getRun(), job.getUser(), job.getGroup(),
539                            maskDate(job.getCreatedTime(), localtime),
540                            maskDate(job.getStartTime(), localtime),
541                            maskDate(job.getLastModTime(), localtime),
542                            maskDate(job.getEndTime(), localtime)));
543                }
544                System.out.println(RULER);
545            } else {
546                System.out.println("No Jobs match your criteria!");
547            }
548        }

550        private String maskIfNull(String value){
551            if(value!=null && value.length()>0){
552                return value;
553            }
554            return "-";
555        }

557        private String maskDate(Date date, boolean isLocalTimeZone){
558            if(date==null){
559                return "-";
560            }
561            
562            SimpleDateFormat dateFormater = new SimpleDateFormat("yyyy-MM-dd HH:mm Z", 
563                    Locale.US);
564            if(!isLocalTimeZone){
565                    dateFormater.setTimeZone(TimeZone.getTimeZone("GMT"));
566            }
567            return dateFormater.format(date);
568        }

570        private void validateCommand(CommandLine commandLine) throws OozieCLIException {
571            String[] args = commandLine.getArgs();
572            if (args.length != 1) {
573                throw new OozieCLIException("One file must be specified");
574            }
575            File file = new File(args[0]);
576            if (file.exists()) {
577                try {
578                    List<StreamSource> sources = new ArrayList<StreamSource>();
579                    sources.add(new StreamSource(
580                            Thread.currentThread().getContextClassLoader().getResourceAsStream("oozie-workflow-0.1.xsd")));
581                    SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
582                    Schema schema = factory.newSchema(sources.toArray(new StreamSource[sources.size()]));
583                    Validator validator = schema.newValidator();
584                    validator.validate(new StreamSource(new FileReader(file)));
585                    System.out.println("Valid worflow-app");
586                }
587                catch (Exception ex) {
588                    throw new OozieCLIException("Invalid workflow-app, " + ex.toString(), ex);
589                }
590            }
591            else {
592                throw new OozieCLIException("File does not exists");
593            }
594        }
