018    package org.apache.oozie;
019    
020    import org.apache.oozie.workflow.WorkflowInstance;
021    import org.apache.oozie.client.rest.JsonWorkflowJob;
022    import org.apache.oozie.client.WorkflowJob;
023    import org.apache.oozie.util.WritableUtils;
024    import org.apache.hadoop.io.Writable;
025    
026    import java.io.DataInput;
027    import java.io.IOException;
028    import java.io.DataOutput;
029    import java.util.Date;
030    
031    public class WorkflowJobBean extends JsonWorkflowJob implements Writable {
032        private String authToken;
033        private String logToken;
034        private WorkflowInstance workflowInstance;
035        private String protoActionConf;
036    
037        /**
038         * Default constructor.
039         */
040        public WorkflowJobBean() {
041        }
042    
043        /**
044         * Serialize the workflow bean to a data output.
045         *
046         * @param dataOutput data output.
047         * @throws IOException thrown if the workflow bean could not be serialized.
048         */
049        public void write(DataOutput dataOutput) throws IOException {
050            WritableUtils.writeStr(dataOutput, getAppPath());
051            WritableUtils.writeStr(dataOutput, getAppName());
052            WritableUtils.writeStr(dataOutput, getId());
053            WritableUtils.writeStr(dataOutput, getConf());
054            WritableUtils.writeStr(dataOutput, getStatus().toString());
055            dataOutput.writeLong((getCreatedTime() != null) ? getCreatedTime().getTime() : -1);
056            dataOutput.writeLong((getStartTime() != null) ? getStartTime().getTime() : -1);
057            dataOutput.writeLong((getLastModTime() != null) ? getLastModTime().getTime() : -1);
058            dataOutput.writeLong((getEndTime() != null) ? getEndTime().getTime() : -1);
059            WritableUtils.writeStr(dataOutput, getUser());
060            WritableUtils.writeStr(dataOutput, getGroup());
061            dataOutput.writeInt(getRun());
062            WritableUtils.writeStr(dataOutput, authToken);
063            WritableUtils.writeStr(dataOutput, logToken);
064            WritableUtils.writeStr(dataOutput, protoActionConf);
065        }
066    
067        /**
068         * Deserialize a workflow bean from a data input.
069         *
070         * @param dataInput data input.
071         * @throws IOException thrown if the workflow bean could not be deserialized.
072         */
073        public void readFields(DataInput dataInput) throws IOException {
074            setAppPath(WritableUtils.readStr(dataInput));
075            setAppName(WritableUtils.readStr(dataInput));
076            setId(WritableUtils.readStr(dataInput));
077            setConf(WritableUtils.readStr(dataInput));
078            setStatus(WorkflowJob.Status.valueOf(WritableUtils.readStr(dataInput)));
079            long d = dataInput.readLong();
080            if (d != -1) {
081                setCreatedTime(new Date(d));
082            }
083            d = dataInput.readLong();
084            if (d != -1) {
085                setStartTime(new Date(d));
086            }
087            d = dataInput.readLong();
088            if(d != -1) {
089                setLastModTime(new Date(d));
090            }
091            d = dataInput.readLong();
092            if (d != -1) {
093                setEndTime(new Date(d));
094            }
095            setUser(WritableUtils.readStr(dataInput));
096            setGroup(WritableUtils.readStr(dataInput));
097            setRun(dataInput.readInt());
098            authToken = WritableUtils.readStr(dataInput);
099            logToken = WritableUtils.readStr(dataInput);
100            protoActionConf = WritableUtils.readStr(dataInput);
101        }
102    
103        public String getAuthToken() {
104            return authToken;
105        }
106    
107        public void setAuthToken(String authToken) {
108            this.authToken = authToken;
109        }
110    
111        public String getLogToken() {
112            return logToken;
113        }
114    
115        public void setLogToken(String logToken) {
116            this.logToken = logToken;
117        }
118    
119        public WorkflowInstance getWorkflowInstance() {
120            return workflowInstance;
121        }
122    
123        public void setWorkflowInstance(WorkflowInstance workflowInstance) {
124            this.workflowInstance = workflowInstance;
125        }
126    
127        public String getProtoActionConf() {
128            return protoActionConf;
129        }
130    
131        public void setProtoActionConf(String protoActionConf) {
132            this.protoActionConf = protoActionConf;
133        }
134    }
