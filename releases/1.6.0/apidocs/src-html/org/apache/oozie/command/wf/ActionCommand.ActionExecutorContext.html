050    /**
051     * Base class for Action execution commands. Provides common functionality to
052     * handle different types of errors while attempting to start or end an action.
053     */
054    public abstract class ActionCommand<T> extends Command<Void> {
055        private static final String INSTRUMENTATION_GROUP = "action.executors";
056    
057        protected static final String INSTR_FAILED_JOBS_COUNTER = "failed";
058    
059        protected static final String RECOVERY_ID_SEPARATOR = "@";
060    
061        public ActionCommand(String name, String type, int priority) {
062            super(name, type, priority, XLog.STD);
063        }
064    
065        /**
066         * Takes care of Transient failures. Sets the action status to retry and
067         * increments the retry count if not enough attempts have been made.
068         * Otherwise returns false.
069         * 
070         * @param context the execution context.
071         * @param executor the executor instance being used.
072         * @param status the status to be set for the action.
073         * @return true if the action is scheduled for another retry. false if the
074         *         number of retries has exceeded the maximum number of configured
075         *         retries.
076         * @throws StoreException
077         * @throws org.apache.oozie.command.CommandException
078         */
079        protected boolean handleTransient(ActionExecutor.Context context, ActionExecutor executor, WorkflowAction.Status status)
080                throws StoreException, CommandException {
081            XLog.getLog(getClass()).debug("Attempting to retry");
082            ActionExecutorContext aContext = (ActionExecutorContext) context;
083            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
084            incrActionErrorCounter(action.getType(), "transient", 1);
085    
086            int actionRetryCount = action.getRetries();
087    
088            if (actionRetryCount >= executor.getMaxRetries()) {
089                XLog.getLog(getClass()).warn("Exceeded max retry count [{0}]. Suspending Job", executor.getMaxRetries());
090                return false;
091            }
092            else {
093                action.setStatus(status);
094                action.setPending();
095                action.incRetries();
096                long retryDelayMillis = executor.getRetryInterval() * 1000;
097                action.setPendingAge(new Date(System.currentTimeMillis() + retryDelayMillis));
098                XLog.getLog(getClass()).info("Next Retry, Attempt Number [{0}] in [{1}] milliseconds",
099                        actionRetryCount + 1, retryDelayMillis);
100                queueCallable(this, retryDelayMillis);
101                return true;
102            }
103        }
104    
105        /**
106         * Takes care of non transient failures. The job is suspended, and the state
107         * of the action is changed to *MANUAL
108         * 
109         * @param context the execution context.
110         * @param executor the executor instance being used.
111         * @param status the status to be set for the action.
112         * @throws StoreException
113         * @throws org.apache.oozie.command.CommandException
114         */
115        protected void handleNonTransient(ActionExecutor.Context context, ActionExecutor executor, WorkflowAction.Status status)
116                throws StoreException, CommandException {
117            XLog.getLog(getClass()).warn("Suspending Job");
118            ActionExecutorContext aContext = (ActionExecutorContext) context;
119            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
120            incrActionErrorCounter(action.getType(), "nontransient", 1);
121            WorkflowJobBean workflow = (WorkflowJobBean) context.getWorkflow();
122            String id = workflow.getId();
123            action.setStatus(status);
124            try {
125                SuspendCommand.suspendJob(workflow, id);
126            }
127            catch (WorkflowException e) {
128                throw new CommandException(e);
129            }
130        }
131    
132        /**
133         * Takes care of errors.
134         * </p>
135         * For errors while attempting to start the action, the job state is updated
136         * and an {@link ActionEndCommand} is queued.
137         * </p>
138         * For errors while attempting to end the action, the job state is updated.
139         * </p>
140         * 
141         * @param context the execution context.
142         * @param executor the executor instance being used.
143         * @param message
144         * @param isStart whether the error was generated while starting or ending
145         *        an action.
146         * @param status the status to be set for the action.
147         * @throws org.apache.oozie.command.CommandException
148         */
149        protected void handleError(ActionExecutor.Context context, ActionExecutor executor, String message,
150                boolean isStart, WorkflowAction.Status status) throws CommandException {
151            XLog.getLog(getClass()).warn("Setting Action Status to [{0}]", status);
152            ActionExecutorContext aContext = (ActionExecutorContext) context;
153            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
154            incrActionErrorCounter(action.getType(), "error", 1);
155            action.setPending();
156            if (isStart) {
157                action.setExecutionData(message, null);
158                queueCallable(new ActionEndCommand(action.getId(), action.getType()));
159            }
160            else {
161                action.setEndData(status, WorkflowAction.Status.ERROR.toString());
162            }
163        }
164    
165        public void failJob(ActionExecutor.Context context) throws CommandException {
166            ActionExecutorContext aContext = (ActionExecutorContext) context;
167            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
168            incrActionErrorCounter(action.getType(), "failed", 1);
169            WorkflowJobBean workflow = (WorkflowJobBean) context.getWorkflow();
170            XLog.getLog(getClass()).warn("Failing Job due to failed action [{0}]", action.getName());
171            try {
172                workflow.getWorkflowInstance().fail(action.getName());
173                workflow.setStatus(WorkflowJob.Status.FAILED);
174                action.setStatus(WorkflowAction.Status.FAILED);
175                queueCallable(new NotificationCommand(workflow, action));
176                queueCallable(new KillCommand(workflow.getId()));
177                incrJobCounter(INSTR_FAILED_JOBS_COUNTER, 1);
178            } catch (WorkflowException ex) {
179                throw new CommandException(ex);
180            }
181        }
182    
183        private void incrActionErrorCounter(String type, String error, int count) {
184            getInstrumentation().incr(INSTRUMENTATION_GROUP, type + "#ex." + error, count);
185        }
186    
187        protected void incrActionCounter(String type, int count) {
188            getInstrumentation().incr(INSTRUMENTATION_GROUP, type + "#" + getName(), count);
189        }
190    
191        protected void addActionCron(String type, Instrumentation.Cron cron) {
192            getInstrumentation().addCron(INSTRUMENTATION_GROUP, type + "#" + getName(), cron);
193        }
194    
195        public static class ActionExecutorContext implements ActionExecutor.Context {
196            private WorkflowJobBean workflow;
197            private Configuration protoConf;
198            private WorkflowActionBean action;
199            private boolean isRetry;
200            private boolean started;
201            private boolean ended;
202            private boolean executed;
203    
204            public ActionExecutorContext(WorkflowJobBean workflow, WorkflowActionBean action, boolean isRetry) {
205                this.workflow = workflow;
206                this.action = action;
207                this.isRetry = isRetry;
208                try {
209                    protoConf = new XConfiguration(new StringReader(workflow.getProtoActionConf()));
210                }
211                catch (IOException ex) {
212                    throw new RuntimeException("It should not happen", ex);
213                }
214            }
215    
216            public String getCallbackUrl(String externalStatusVar) {
217                return Services.get().get(CallbackService.class).createCallBackUrl(action.getId(), externalStatusVar);
218            }
219    
220            public Configuration getProtoActionConf() {
221                return protoConf;
222            }
223    
224            public WorkflowJob getWorkflow() {
225                return workflow;
226            }
227    
228            public WorkflowAction getAction() {
229                return action;
230            }
231    
232            public ELEvaluator getELEvaluator() {
233                ELEvaluator evaluator = Services.get().get(ELService.class).createEvaluator();
234                DagELFunctions.configureEvaluator(evaluator, workflow, action);
235                return evaluator;
236            }
237    
238            public void setVar(String name, String value) {
239                name = action.getName() + WorkflowInstance.NODE_VAR_SEPARATOR + name;
240                workflow.getWorkflowInstance().setVar(name, value);
241            }
242    
243            public String getVar(String name) {
244                name = action.getName() + WorkflowInstance.NODE_VAR_SEPARATOR + name;
245                return workflow.getWorkflowInstance().getVar(name);
246            }
247    
248            public void setStartData(String externalId, String trackerUri, String consoleUrl) {
249                action.setStartData(externalId, trackerUri, consoleUrl);
250                started = true;
251            }
252    
253            public void setExecutionData(String externalStatus, Properties actionData) {
254                action.setExecutionData(externalStatus, actionData);
255                executed = true;
256            }
257    
258            public void setEndData(WorkflowAction.Status status, String signalValue) {
259                action.setEndData(status, signalValue);
260                ended = true;
261            }
262    
263            public boolean isRetry() {
264                return isRetry;
265            }
266    
267            /**
268             * Returns whether setStartData has been called or not.
269             *
270             * @return true if start completion info has been set.
271             */
272            public boolean isStarted() {
273                return started;
274            }
275    
276            /**
277             * Returns whether setExecutionData has been called or not.
278             *
279             * @return true if execution completion info has been set, otherwise false.
280             */
281            public boolean isExecuted() {
282                return executed;
283            }
284    
285    
286            /**
287             * Returns whether setEndData has been called or not.
288             *
289             * @return true if end completion info has been set.
290             */
291            public boolean isEnded() {
292                return ended;
293            }
294    
295            public void setExternalStatus(String externalStatus) {
296                action.setExternalStatus(externalStatus);
297            }
298    
299            @Override
300            public String getRecoveryId() {
301                return action.getId() + RECOVERY_ID_SEPARATOR + workflow.getRun();
302            }
303    
304            public Path getActionDir() throws URISyntaxException, IOException{
305                String name = getWorkflow().getId() + "/" + action.getName() + "--" + action.getType();
306                FileSystem fs = getAppFileSystem();
307                String actionDirPath = Services.get().getSystemId() + "/" + name;
308                Path fqActionDir = new Path(fs.getHomeDirectory(), actionDirPath);
309                return fqActionDir;
310            }
311    
312            public FileSystem getAppFileSystem() throws IOException, URISyntaxException{
313                WorkflowJob workflow = getWorkflow();
314                XConfiguration jobConf = new XConfiguration(new StringReader(workflow.getConf()));
315                Configuration fsConf = new Configuration();
316                XConfiguration.copy(jobConf, fsConf);
317                return Services.get().get(HadoopAccessorService.class).
318                        createFileSystem(workflow.getUser(), workflow.getGroup(), new URI(getWorkflow().getAppPath()),
319                                         fsConf);
320            }
321        }
