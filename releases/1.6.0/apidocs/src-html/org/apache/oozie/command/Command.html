040    /**
041     * Base class for all synchronous and asynchronous DagEngine commands.
042     */
043    public abstract class Command<T> implements XCallable<T> {
044        /**
045         * The instrumentation group used for Commands.
046         */
047        private static final String INSTRUMENTATION_GROUP = "commands";
048        
049        /**
050         * The instrumentation group used for Jobs.
051         */
052        private static final String INSTRUMENTATION_JOB_GROUP = "jobs";

054        protected Instrumentation instrumentation;
055        private List<XCallable<Void>> callables;
056        private List<XCallable<Void>> delayedCallables;
057        private long delay = 0;
058        private List<XCallable<Void>> exceptionCallables;
059        private String name;
060        private int priority;
061        private int logMask;
062        private boolean withStore;
063        private String type;

066         * This variable is package private for testing purposes only.

070        /**
071         * Create a command that uses a {@link WorkflowStore} instance.
072         * <p/>
073         * The current {@link XLog.Info} values are captured for execution.
074         *
075         * @param name command name.
076         * @param type command type.
077         * @param priority priority of the command, used when queuing for asynchronous execution.
078         * @param logMask log mask for the command logging calls.
079         */
080        public Command(String name, String type, int priority, int logMask) {
081            this(name, type, priority, logMask, true);
082        }
083    
084        /**
085         * Create a command.
086         * <p/>
087         * The current {@link XLog.Info} values are captured for execution.
088         *
089         * @param name command name.
090         * @param type command type.
091         * @param priority priority of the command, used when queuing for asynchronous execution.
092         * @param logMask log mask for the command logging calls.
093         * @param withStore indicates if the command needs a {@link org.apache.oozie.store.WorkflowStore} instance or not.
094         */
095        public Command(String name, String type, int priority, int logMask, boolean withStore) {
096            this.name = ParamChecker.notEmpty(name, "name");
097            this.type = ParamChecker.notEmpty(type, "type");
098            this.priority = priority;
099            this.withStore = withStore;
100            this.logMask = logMask;
101            instrumentation = Services.get().get(InstrumentationService.class).get();
102            logInfo = new XLog.Info(XLog.Info.get());
103        }

106         * Return the name of the command.
107         *
108         * @return the name of the command.
109         */
110        public String getName() {
111            return name;
112        }

120         * @return the callable type.

135        /**
136         * Execute the command {@link #call(WorkflowStore)} setting all the necessary context.
137         * <p/>
138         * The {@link XLog.Info} is set to the values at instance creation time.
139         * <p/>
140         * The command execution is logged and instrumented.
141         * <p/>
142         * If a {@link WorkflowStore} is used, a fresh instance will be passed and it will be commited after the
143         * {@link #call(WorkflowStore)} execution. It will be closed without committing if an exception is thrown.
144         * <p/>
145         * Commands queued via the DagCommand queue methods are queued for execution after the workflow store has been
146         * committed.
147         * <p/>
148         * If an exception happends the queued commands will not be effectively queued for execution. Instead, the
149         * the commands queued for exception will be effectively queued fro execution..
150         *
151         * @throws CommandException thrown if the command could not be executed successfully, the workflow store is
152         * closed without committing, thus doing a rollback.
153         */
154        @SuppressWarnings({"ThrowFromFinallyBlock"})
155        public final T call() throws CommandException {
156            XLog.Info.get().setParameters(logInfo);
157            XLog log = XLog.getLog(getClass());
158            log.trace(logMask, "Start");
159            Instrumentation.Cron cron = new Instrumentation.Cron();
160            cron.start();
161            callables = new ArrayList<XCallable<Void>>();
162            delayedCallables = new ArrayList<XCallable<Void>>();
163            exceptionCallables = new ArrayList<XCallable<Void>>();
164            delay = 0;
165            WorkflowStore store = null;
166            boolean exception = false;
167            try {
168                if (withStore) {
169                    store = Services.get().get(WorkflowStoreService.class).create();
170                }
171                T result = call(store);
172    
173                if (withStore) {
174                    if (store == null) {
175                        throw new IllegalStateException("WorkflowStore should not be null");
176                    }
177                    if (FaultInjection.isActive("org.apache.oozie.command.SkipCommitFaultInjection")) {
178                        throw new RuntimeException("Skipping Commit for Failover Testing");
179                    }
180                    store.commit();
181                }
182    
183                //TODO figure out the reject due to concurrency problems and remove the delayed queuing for callables.
184                Services.get().get(CallableQueueService.class).queueSerial(callables, 10);
185    
186                Services.get().get(CallableQueueService.class).queueSerial(delayedCallables, delay);
187    
188                return result;
189            }
190            catch (XException ex) {
191                exception = true;
192                //TODO figure out the reject due to concurrency problems and remove the delayed queuing for callables.
193                Services.get().get(CallableQueueService.class).queueSerial(exceptionCallables, 10);
194                if (ex instanceof CommandException) {
195                    throw (CommandException) ex;
196                }
197                else {
198                    throw new CommandException(ex);
199                }
200            }
201            catch (RuntimeException ex) {
202                exception = true;
203                throw ex;
204            }
205            finally {
206                FaultInjection.deactivate("org.apache.oozie.command.SkipCommitFaultInjection");
207                cron.stop();
208                instrumentation.addCron(INSTRUMENTATION_GROUP, name, cron);
209                incrCommandCounter(1);
210                log.trace(logMask, "End");
211                if (store != null) {
212                    try {
213                        store.close();
214                    }
215                    catch (StoreException ex) {
216                        if (exception) {
217                            log.warn(logMask | XLog.OPS, "store error, {1}", name, ex.getMessage(), ex);
218                        }
219                        else {
220                            throw new CommandException(ex);
221                        }
222                    }
223                }
224            }
225        }

227        /**
228         * Queue a callable for execution after the current callable call invocation completes and the {@link WorkflowStore}
229         * transaction commits.
230         * <p/>
231         * All queued callables, regardless of the number of queue invocations,   are queued for a single serial execution.
232         * <p/>
233         * If the call invocation throws an exception all queued callables are discarded, they are not queued for execution.
234         *
235         * @param callable callable to queue for execution.
236         */
237        protected void queueCallable(XCallable<Void> callable) {
238            callables.add(callable);
239        }
240    
241        /**
242         * Queue a list of callables for execution after the current callable call invocation completes and the
243         * {@link WorkflowStore} transaction commits.
244         * <p/>
245         * All queued callables, regardless of the number of queue invocations,   are queued for a single serial execution.
246         * <p/>
247         * If the call invocation throws an exception all queued callables are discarded, they are not queued for execution.
248         *
249         * @param callables list of callables to queue for execution.
250         */
251        protected void queueCallable(List<? extends XCallable<Void>> callables) {
252            this.callables.addAll(callables);
253        }
254    
255        /**
256         * Queue a callable for delayed execution after the current callable call invocation completes and the
257         * {@link WorkflowStore} transaction commits.
258         * <p/>
259         * All queued delayed callables, regardless of the number of delay queue invocations,
260         * are queued for a single serial delayed execution with the highest delay of all queued callables.
261         * <p/>
262         * If the call invocation throws an exception all queued callables are discarded, they are not queued for execution.
263         *
264         * @param callable callable to queue for delayed execution.
265         * @param delay the queue delay in milliseconds
266         */
267        protected void queueCallable(XCallable<Void> callable, long delay) {
268            this.delayedCallables.add(callable);
269            this.delay = Math.max(this.delay, delay);
270        }
271    
272        /**
273         * Queue a callable for execution only in the event of an exception being thrown during the call invocation.
274         * <p/>
275         * If an exception does not happen, all the callables queued by this method are discarded, they are not queued for
276         * execution.
277         * <p/>
278         * All queued callables, regardless of the number of queue invocations,   are queued for a single serial execution.
279         *
280         * @param callable callable to queue for execution in the case of an exception.
281         */
282        protected void queueCallableForException(XCallable<Void> callable) {
283            exceptionCallables.add(callable);
284        }
285    
286        /**
287         * DagCallable subclasses must implement this method to perform their task.
288         * <p/>
289         * The workflow store works in transactional mode. The transaction is committed only if this method ends
290         * successfully. Otherwise the transaction is rolledback.
291         *
292         * @param store the workflow store instance for the callable, <code>null</code> if the callable does not use a
293         * store.
294         * @return the return value of the callable.
295         * @throws StoreException thrown if the workflow store could not perform an operation.
296         * @throws CommandException thrown if the command could not perform its operation.
297         */
298        protected abstract T call(WorkflowStore store) throws StoreException, CommandException;
299    
300        /**
301         * Set the log info with the context of the given workflow bean.
302         *
303         * @param workflow workflow bean.
304         */
305        protected void setLogInfo(WorkflowJobBean workflow) {
306            if(logInfo.getParameter(XLogService.GROUP) == null) {
307                logInfo.setParameter(XLogService.GROUP, workflow.getGroup());
308            }
309            if(logInfo.getParameter(XLogService.USER) == null) {
310                logInfo.setParameter(XLogService.USER, workflow.getUser());
311            }
312            logInfo.setParameter(DagXLogInfoService.JOB, workflow.getId());
313            logInfo.setParameter(DagXLogInfoService.TOKEN, workflow.getLogToken());
314            logInfo.setParameter(DagXLogInfoService.APP, workflow.getAppName());
315            XLog.Info.get().setParameters(logInfo);
316        }
317    
318        /**
319         * Set the log info with the context of the given action bean.
320         *
321         * @param action action bean.
322         */
323        protected void setLogInfo(WorkflowActionBean action) {
324            logInfo.setParameter(DagXLogInfoService.JOB, action.getJobId());
325            logInfo.setParameter(DagXLogInfoService.TOKEN, action.getLogToken());
326            logInfo.setParameter(DagXLogInfoService.ACTION, action.getName());
327            XLog.Info.get().setParameters(logInfo);
328        }
329    
330        /**
331         * Reset the action bean information from the log info.
332         */
333        //TODO check if they are used, else delete
334        protected void resetLogInfoAction() {
335            logInfo.clearParameter(DagXLogInfoService.ACTION);
336            XLog.Info.get().clearParameter(DagXLogInfoService.ACTION);
337        }
338    
339        /**
340         * Reset the workflow bean information from the log info.
341         */
342        //TODO check if they are used, else delete
343        protected void resetLogInfoWorkflow() {
344            logInfo.clearParameter(DagXLogInfoService.JOB);
345            logInfo.clearParameter(DagXLogInfoService.APP);
346            logInfo.clearParameter(DagXLogInfoService.TOKEN);
347            XLog.Info.get().clearParameter(DagXLogInfoService.JOB);
348            XLog.Info.get().clearParameter(DagXLogInfoService.APP);
349            XLog.Info.get().clearParameter(DagXLogInfoService.TOKEN);
350        }

352        /**
353         * Convenience method to increment counters.
354         *
355         * @param group the group name.
356         * @param name the counter name.
357         * @param count increment count.
358         */
359        private void incrCounter(String group, String name, int count) {
360            if (instrumentation != null) {
361                instrumentation.incr(group, name, count);
362            }
363        }
364    
365        /**
366         * Used to increment command counters.
367         * 
368         * @param count the increment count.
369         */
370        protected void incrCommandCounter(int count) {
371            incrCounter(INSTRUMENTATION_GROUP, name, count);
372        }
373    
374        /**
375         * Used to increment job counters. The counter name s the same as the
376         * command name.
377         * 
378         * @param count the increment count.
379         */
380        protected void incrJobCounter(int count) {
381            incrJobCounter(name, count);
382        }
383    
384        /**
385         * Used to increment job counters.
386         *
387         * @param name the job name.
388         * @param count the increment count.
389         */
390        protected void incrJobCounter(String name, int count) {
391            incrCounter(INSTRUMENTATION_JOB_GROUP, name, count);
392        }
