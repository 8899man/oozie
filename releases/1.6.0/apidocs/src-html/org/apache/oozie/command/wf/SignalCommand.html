018    package org.apache.oozie.command.wf;
019    
020    import org.apache.oozie.client.WorkflowJob;
021    import org.apache.oozie.WorkflowActionBean;
022    import org.apache.oozie.WorkflowJobBean;
023    import org.apache.oozie.ErrorCode;
024    import org.apache.oozie.command.Command;
025    import org.apache.oozie.command.CommandException;
026    import org.apache.oozie.service.WorkflowStoreService;
027    import org.apache.oozie.store.StoreException;
028    import org.apache.oozie.store.WorkflowStore;
029    import org.apache.oozie.workflow.WorkflowException;
030    import org.apache.oozie.workflow.WorkflowInstance;
031    import org.apache.oozie.util.XLog;
032    import org.apache.oozie.util.ParamChecker;
033    
034    import java.util.Date;
035    
036    public class SignalCommand extends Command<Void> {
037    
038        protected static final String INSTR_SUCCEEDED_JOBS_COUNTER_NAME = "succeeded";
039    
040        private String jobId;
041        private String actionId;
042    
043        protected SignalCommand(String name, int priority, String jobId) {
044            super(name, name, priority, XLog.STD);
045            this.jobId = ParamChecker.notEmpty(jobId, "jobId");
046        }
047    
048        public SignalCommand(String jobId, String actionId) {
049            super("signal", "signal", 0, XLog.STD);
050            this.jobId = ParamChecker.notEmpty(jobId, "jobId");
051            this.actionId = ParamChecker.notEmpty(actionId, "actionId");
052        }
053    
054        @Override
055        protected Void call(WorkflowStore store) throws CommandException, StoreException {
056            WorkflowJobBean workflow = store.getWorkflow(jobId, true);
057            setLogInfo(workflow);
058            WorkflowActionBean action = null;
059            boolean skipAction = false;
060            if (actionId != null) {
061                action = store.getAction(actionId, true);
062                setLogInfo(action);
063            }
064            if ((action == null) || (action.isComplete() && action.isPending())) {
065                try {
066                    if (workflow.getStatus() == WorkflowJob.Status.RUNNING
067                            || workflow.getStatus() == WorkflowJob.Status.PREP) {
068                        WorkflowInstance workflowInstance = workflow.getWorkflowInstance();
069                        workflowInstance.setTransientVar(WorkflowStoreService.WORKFLOW_BEAN, workflow);
070                        boolean completed;
071                        if (action == null) {
072                            if (workflow.getStatus() == WorkflowJob.Status.PREP) {
073                                completed = workflowInstance.start();
074                                workflow.setStatus(WorkflowJob.Status.RUNNING);
075                                workflow.setStartTime(new Date());
076                                queueCallable(new NotificationCommand(workflow));
077                            }
078                            else {
079                                throw new CommandException(ErrorCode.E0801, workflow.getId());
080                            }
081                        }
082                        else {
083                            String skipVar = workflowInstance.getVar(action.getName() + WorkflowInstance.NODE_VAR_SEPARATOR
084                                    + ReRunCommand.TO_SKIP);
085                            if (skipVar != null) {
086                                skipAction = skipVar.equals("true");
087                            }
088                            completed = workflowInstance.signal(action.getExecutionPath(), action.getSignalValue());
089                            action.resetPending();
090                            if (!skipAction) {
091                                action.setTransition(workflowInstance.getTransition(action.getName()));
092                                store.updateAction(action);
093                            }
094                        }
095    
096                        if (completed) {
097                            for (String actionToKillId : WorkflowStoreService.getActionsToKill(workflowInstance)) {
098                                WorkflowActionBean actionToKill = store.getAction(actionToKillId, false);
099                                actionToKill.setPending();
100                                actionToKill.setStatus(WorkflowActionBean.Status.KILLED);
101                                store.updateAction(actionToKill);
102                                queueCallable(new ActionKillCommand(actionToKill.getId(), actionToKill.getType()));
103                            }
104    
105                            for (String actionToFailId : WorkflowStoreService.getActionsToFail(workflowInstance)) {
106                                WorkflowActionBean actionToFail = store.getAction(actionToFailId, false);
107                                actionToFail.resetPending();
108                                actionToFail.setStatus(WorkflowActionBean.Status.FAILED);
109                                store.updateAction(actionToFail);
110                            }
111    
112                            workflow.setStatus(WorkflowJob.Status.valueOf(workflowInstance.getStatus().toString()));
113                            workflow.setEndTime(new Date());
114                            queueCallable(new NotificationCommand(workflow));
115                            if (workflow.getStatus() == WorkflowJob.Status.SUCCEEDED) {
116                                incrJobCounter(INSTR_SUCCEEDED_JOBS_COUNTER_NAME, 1);
117                            }
118                        }
119                        else {
120                            for (WorkflowActionBean newAction : WorkflowStoreService.getStartedActions(workflowInstance)) {
121                                String skipVar = workflowInstance.getVar(newAction.getName()
122                                        + WorkflowInstance.NODE_VAR_SEPARATOR + ReRunCommand.TO_SKIP);
123                                boolean skipNewAction = false;
124                                if (skipVar != null) {
125                                    skipNewAction = skipVar.equals("true");
126                                }
127                                if (skipNewAction) {
128                                    WorkflowActionBean oldAction = store.getAction(newAction.getId(), false);
129                                    oldAction.setPending();
130                                    store.updateAction(oldAction);
131                                    queueCallable(new SignalCommand(jobId, oldAction.getId()));
132                                }
133                                else {
134                                    newAction.setPending();
135                                    store.insertAction(newAction);
136                                    queueCallable(new ActionStartCommand(newAction.getId(), newAction.getType()));
137                                }
138                            }
139                        }
140                        store.updateWorkflow(workflow);
141                    }
142                    else {
143                        XLog.getLog(getClass()).warn("Workflow not RUNNING, current status [{0}]", workflow.getStatus());
144                    }
145                }
146                catch (WorkflowException ex) {
147                    throw new CommandException(ex);
148                }
149            }
150            else {
151                XLog.getLog(getClass()).warn(
152                        "SignalCommand for action id :" + actionId + " is already processed. status=" + action.getStatus()
153                                + ", Pending=" + action.isPending());
154            }
155            return null;
156        }
157    
158    }
