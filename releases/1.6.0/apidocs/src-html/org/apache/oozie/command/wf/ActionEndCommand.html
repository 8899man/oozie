018    package org.apache.oozie.command.wf;
019    
020    import org.apache.hadoop.conf.Configuration;
021    import org.apache.oozie.WorkflowActionBean;
022    import org.apache.oozie.WorkflowJobBean;
023    import org.apache.oozie.DagELFunctions;
024    import org.apache.oozie.ErrorCode;
025    import org.apache.oozie.command.CommandException;
026    import org.apache.oozie.service.ActionService;
027    import org.apache.oozie.action.ActionExecutor;
028    import org.apache.oozie.action.ActionExecutorException;
029    import org.apache.oozie.store.StoreException;
030    import org.apache.oozie.store.WorkflowStore;
031    import org.apache.oozie.client.WorkflowAction;
032    import org.apache.oozie.client.WorkflowJob;
033    import org.apache.oozie.client.OozieClient;
034    import org.apache.oozie.service.Services;
035    import org.apache.oozie.service.UUIDService;
036    import org.apache.oozie.util.XLog;
037    import org.apache.oozie.util.Instrumentation;
038    
039    import java.util.Date;
040    
041    public class ActionEndCommand extends ActionCommand<Void> {
042        public static final String COULD_NOT_END = "COULD_NOT_END";
043        public static final String END_DATA_MISSING = "END_DATA_MISSING";
044    
045        private String id;
046    
047        public ActionEndCommand(String id, String type) {
048            super("action.end", type, 0);
049            this.id = id;
050        }
051    
052        protected Void call(WorkflowStore store) throws StoreException, CommandException {
053            String jobId = Services.get().get(UUIDService.class).getId(id);
054            WorkflowJobBean workflow = store.getWorkflow(jobId, true);
055            setLogInfo(workflow);
056            WorkflowActionBean action = store.getAction(id, true);
057            setLogInfo(action);
058            if (action.isPending()
059                    && (action.getStatus() == WorkflowActionBean.Status.DONE || action.getStatus() == WorkflowActionBean.Status.END_RETRY || action
060                            .getStatus() == WorkflowActionBean.Status.END_MANUAL)) {
061                if (workflow.getStatus() == WorkflowJob.Status.RUNNING) {
062    
063                    ActionExecutor executor = Services.get().get(ActionService.class).getExecutor(action.getType());
064                    Configuration conf = workflow.getWorkflowInstance().getConf();
065                    int maxRetries = conf.getInt(OozieClient.ACTION_MAX_RETRIES, executor.getMaxRetries());
066                    long retryInterval = conf.getLong(OozieClient.ACTION_RETRY_INTERVAL, executor.getRetryInterval());
067                    executor.setMaxRetries(maxRetries);
068                    executor.setRetryInterval(retryInterval);
069    
070                    if (executor != null) {
071                        boolean isRetry = false;
072                        if (action.getStatus() == WorkflowActionBean.Status.END_RETRY
073                                || action.getStatus() == WorkflowActionBean.Status.END_MANUAL) {
074                            isRetry = true;
075                        }
076                        ActionExecutorContext context = new ActionCommand.ActionExecutorContext(workflow, action, isRetry);
077                        try {
078    
079                            XLog.getLog(getClass()).debug(
080                                    "End, name [{0}] type [{1}] status[{2}] external status [{3}] signal value [{4}]",
081                                    action.getName(), action.getType(), action.getStatus(), action.getExternalStatus(),
082                                    action.getSignalValue());
083    
084                            DagELFunctions.setActionInfo(workflow.getWorkflowInstance(), action);
085                            incrActionCounter(action.getType(), 1);
086    
087                            Instrumentation.Cron cron = new Instrumentation.Cron();
088                            cron.start();
089                            executor.end(context, action);
090                            cron.stop();
091                            addActionCron(action.getType(), cron);
092    
093                            if (!context.isEnded()) {
094                                XLog.getLog(getClass()).warn(XLog.OPS,
095                                    "Action Ended, ActionExecutor [{0}] must call setEndData()", executor.getType());
096                                action.setErrorInfo(END_DATA_MISSING, "Execution Ended, but End Data Missing from Action");
097                                failJob(context);
098                                store.updateAction(action);
099                                store.updateWorkflow(workflow);
100                                return null;
101                            }
102                            action.setRetries(0);
103                            action.setEndTime(new Date());
104                            store.updateAction(action);
105                            store.updateWorkflow(workflow);
106                            queueCallable(new NotificationCommand(workflow, action));
107                            queueCallable(new SignalCommand(workflow.getId(), id));
108                        }
109                        catch (ActionExecutorException ex) {
110                            XLog.getLog(getClass()).warn(
111                                    "Error ending action [{0}]. ErrorType [{1}], ErrorCode [{2}], Message [{3}]",
112                                    action.getName(), ex.getErrorType(), ex.getErrorCode(), ex.getMessage());
113                            action.setErrorInfo(ex.getErrorCode(), ex.getMessage());
114                            action.setEndTime(null);
115                            switch (ex.getErrorType()) {
116                                case TRANSIENT:
117                                    if (!handleTransient(context, executor, WorkflowAction.Status.END_RETRY)) {
118                                        handleNonTransient(context, executor, WorkflowAction.Status.END_MANUAL);
119                                        action.setPendingAge(new Date());
120                                        action.setRetries(0);
121                                    }
122                                    action.setEndTime(null);
123                                    break;
124                                case NON_TRANSIENT:
125                                    handleNonTransient(context, executor, WorkflowAction.Status.END_MANUAL);
126                                    action.setEndTime(null);
127                                    break;
128                                case ERROR:
129                                    handleError(context, executor, COULD_NOT_END, false, WorkflowAction.Status.ERROR);
130                                    queueCallable(new SignalCommand(workflow.getId(), id));
131                                    break;
132                                case FAILED:
133                                    failJob(context);
134                                    break;
135                            }
136                            store.updateAction(action);
137                            store.updateWorkflow(workflow);
138                        }
139                    }
140                    else {
141                        throw new CommandException(ErrorCode.E0802, action.getType());
142                    }
143                } else {
144                    XLog.getLog(getClass()).warn("Job state is not {0}. Skipping Action Execution",
145                            WorkflowJob.Status.RUNNING.toString());
146                }
147            }
148            return null;
149        }
150    }
