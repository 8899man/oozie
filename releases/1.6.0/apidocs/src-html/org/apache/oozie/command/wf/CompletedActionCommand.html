018    package org.apache.oozie.command.wf;
019    
020    import org.apache.oozie.WorkflowActionBean;
021    import org.apache.oozie.ErrorCode;
022    import org.apache.oozie.command.Command;
023    import org.apache.oozie.command.CommandException;
024    import org.apache.oozie.service.ActionService;
025    import org.apache.oozie.action.ActionExecutor;
026    import org.apache.oozie.store.StoreException;
027    import org.apache.oozie.store.WorkflowStore;
028    import org.apache.oozie.util.ParamChecker;
029    import org.apache.oozie.util.XLog;
030    import org.apache.oozie.service.Services;
031    
032    import java.util.Properties;
033    
034    public class CompletedActionCommand extends Command<Void> {
035        private String actionId;
036        private String externalStatus;
037        private Properties actionData;
038    
039    
040        public CompletedActionCommand(String actionId, String externalStatus, Properties actionData) {
041            super("callback", "callback", 0, XLog.STD);
042            this.actionId = ParamChecker.notEmpty(actionId, "actionId");
043            this.externalStatus = ParamChecker.notEmpty(externalStatus, "externalStatus");
044            this.actionData = actionData;
045        }
046    
047        @Override
048        protected Void call(WorkflowStore store) throws StoreException, CommandException {
049            WorkflowActionBean action = store.getAction(actionId, false);
050            setLogInfo(action);
051            if (action.getStatus() == WorkflowActionBean.Status.RUNNING) {
052                ActionExecutor executor = Services.get().get(ActionService.class).getExecutor(action.getType());
053                //this is done because oozie notifications (of sub-wfs) is send every status change, not only on completion.
054                if (executor.isCompleted(externalStatus)) {
055                    queueCallable(new ActionCheckCommand(action.getId(), action.getType()));
056                }
057            }
058            else {
059                throw new CommandException(ErrorCode.E0800, actionId, action.getStatus());
060            }
061            return null;
062        }
063    
064    }
