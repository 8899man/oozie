018    package org.apache.oozie.client.rest;

020    import org.apache.oozie.client.WorkflowAction;

024    import java.text.MessageFormat;
025    import java.util.ArrayList;
026    import java.util.Date;

028    
029    /**
030     * Json Bean that represents an Oozie workflow node.
031     */
032    public class JsonWorkflowAction implements WorkflowAction, JsonBean {
033        private String id;
034        private String name;
035        private String type;
036        private String conf;
037        private Status status = WorkflowAction.Status.PREP;
038        private int retries;
039        private Date startTime;
040        private Date endTime;
041        private String transition;
042        private String data;
043        private String externalId;
044        private String externalStatus;
045        private String trackerUri;
046        private String consoleUrl;
047        private String errorCode;
048        private String errorMessage;
049    
050        public JsonWorkflowAction() {
051        }
052    
053        public JsonWorkflowAction(JSONObject jsonObject) {
054            id = (String) jsonObject.get(JsonTags.ACTION_ID);
055            name = (String) jsonObject.get(JsonTags.ACTION_NAME);
056            type = (String) jsonObject.get(JsonTags.ACTION_TYPE);
057            conf = (String) jsonObject.get(JsonTags.ACTION_CONF);
058            status = Status.valueOf((String) jsonObject.get(JsonTags.ACTION_STATUS));
059            retries = (int) JsonUtils.getLongValue(jsonObject, JsonTags.ACTION_RETRIES);
060            startTime = JsonUtils.parseDateRfc822((String) jsonObject.get(JsonTags.ACTION_START_TIME));
061            endTime = JsonUtils.parseDateRfc822((String) jsonObject.get(JsonTags.ACTION_END_TIME));
062            transition = (String) jsonObject.get(JsonTags.ACTION_TRANSITION);
063            data = (String) jsonObject.get(JsonTags.ACTION_DATA);
064            externalId = (String) jsonObject.get(JsonTags.ACTION_EXTERNAL_ID);
065            externalStatus = (String) jsonObject.get(JsonTags.ACTION_EXTERNAL_STATUS);
066            trackerUri = (String) jsonObject.get(JsonTags.ACTION_TRACKER_URI);
067            consoleUrl = (String) jsonObject.get(JsonTags.ACTION_CONSOLE_URL);
068            errorCode = (String) jsonObject.get(JsonTags.ACTION_ERROR_CODE);
069            errorMessage = (String) jsonObject.get(JsonTags.ACTION_ERROR_MESSAGE);
070        }
071    
072        @SuppressWarnings("unchecked")
073        public JSONObject toJSONObject() {
074            JSONObject json = new JSONObject();
075            json.put(JsonTags.ACTION_ID, id);
076            json.put(JsonTags.ACTION_NAME, name);
077            json.put(JsonTags.ACTION_TYPE, type);
078            json.put(JsonTags.ACTION_CONF, conf);
079            json.put(JsonTags.ACTION_START_TIME, JsonUtils.formatDateRfc822(startTime));
080            json.put(JsonTags.ACTION_STATUS, status.toString());
081            json.put(JsonTags.ACTION_RETRIES, (long) retries);
082            json.put(JsonTags.ACTION_START_TIME, JsonUtils.formatDateRfc822(startTime));
083            json.put(JsonTags.ACTION_END_TIME, JsonUtils.formatDateRfc822(endTime));
084            json.put(JsonTags.ACTION_TRANSITION, transition);
085            json.put(JsonTags.ACTION_DATA, data);
086            json.put(JsonTags.ACTION_EXTERNAL_ID, externalId);
087            json.put(JsonTags.ACTION_EXTERNAL_STATUS, externalStatus);
088            json.put(JsonTags.ACTION_TRACKER_URI, trackerUri);
089            json.put(JsonTags.ACTION_CONSOLE_URL, consoleUrl);
090            json.put(JsonTags.ACTION_ERROR_CODE, errorCode);
091            json.put(JsonTags.ACTION_ERROR_MESSAGE, errorMessage);
092            return json;
093        }
094

096            return id;

134    
135        public int getRetries() {
136            return retries;
137        }
138    
139        public void setRetries(int retries) {
140            this.retries = retries;
141        }
142    
143        public Date getStartTime() {
144            return startTime;
145        }
146    
147        public void setStartTime(Date startTime) {
148            this.startTime = startTime;

156            this.endTime = endTime;
157        }
158    
159        public String getTransition() {
160            return transition;
161        }
162

164            this.transition = transition;
165        }
166    
167        public String getData() {
168            return data;
169        }

172            this.data = data;

200            return consoleUrl;
201        }
202

204            this.consoleUrl = consoleUrl;
205        }
206    
207        public String getErrorCode() {
208            return errorCode;
209        }
210    
211        public String getErrorMessage() {
212            return errorMessage;
213        }
214    
215        public void setErrorInfo(String errorCode, String errorMessage) {
216            this.errorCode = errorCode;
217            this.errorMessage = errorMessage;
218        }
219        
220        public String toString() {
221            return MessageFormat.format("Action name[{0}] status[{1}]", getName(), getStatus());
222        }
223    
224        /**
225         * Convert a nodes list into a JSONArray.
226         *
227         * @param nodes nodes list.
228         * @return the corresponding JSON array.
229         */
230        @SuppressWarnings("unchecked")
231        public static JSONArray toJSONArray(List<? extends JsonWorkflowAction> nodes) {
232            JSONArray array = new JSONArray();
233            for (JsonWorkflowAction node : nodes) {
234                array.add(node.toJSONObject());
235            }
236            return array;
237        }
238    
239        /**
240         * Convert a JSONArray into a nodes list.
241         *
242         * @param array JSON array.
243         * @return the corresponding nodes list.
244         */
245        @SuppressWarnings("unchecked")
246        public static List<JsonWorkflowAction> fromJSONArray(JSONArray array) {
247            List<JsonWorkflowAction> list = new ArrayList<JsonWorkflowAction>();
248            for (Object obj : array) {
249                list.add(new JsonWorkflowAction((JSONObject) obj));
250            }
251            return list;
252        }
