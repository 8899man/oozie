049        private static class NoExternalEntityEntityResolver implements EntityResolver {
050    
051            public InputSource resolveEntity(String publicId, String systemId) throws SAXException, IOException {
052                return new InputSource(new ByteArrayInputStream(new byte[0]));
053            }

057        private static SAXBuilder createSAXBuilder() {
058            SAXBuilder saxBuilder = new SAXBuilder();
059    
060            //THIS IS NOT WORKING
061            //saxBuilder.setFeature("http://xml.org/sax/features/external-general-entities", false);
062    
063            //INSTEAD WE ARE JUST SETTING AN EntityResolver that does not resolve entities
064            saxBuilder.setEntityResolver(new NoExternalEntityEntityResolver());
065            return saxBuilder;
066        }

068        /**
069         * Parse a string assuming it is a valid XML document and return an JDOM Element for it.
070         *
071         * @param xmlStr XML string to parse.
072         * @return JDOM element for the parsed XML string.
073         * @throws JDOMException thrown if an error happend while XML parsing.
074         */
075        public static Element parseXml(String xmlStr) throws JDOMException {
076            ParamChecker.notNull(xmlStr, "xmlStr");
077            try {
078                SAXBuilder saxBuilder = createSAXBuilder();
079                Document document = saxBuilder.build(new StringReader(xmlStr));
080                return document.getRootElement();
081            }
082            catch (IOException ex) {
083                throw new RuntimeException("It should not happen, " + ex.getMessage(), ex);
084            }
085        }
086    
087        /**
088         * Parse a inputstream assuming it is a valid XML document and return an JDOM Element for it.
089         *
090         * @param is inputstream to parse.
091         * @return JDOM element for the parsed XML string.
092         * @throws JDOMException thrown if an error happend while XML parsing.
093         * @throws IOException thrown if an IO error occurred.
094         */
095        public static Element parseXml(InputStream is) throws JDOMException, IOException {
096            ParamChecker.notNull(is, "is");
097            SAXBuilder saxBuilder = createSAXBuilder();
098            Document document = saxBuilder.build(is);
099            return document.getRootElement();
100        }
101    
102        /**
103         * //TODO move this to action registry method
104         * Return the value of an attribute from the root element of an XML document.
105         *
106         * @param filePath path of the XML document.
107         * @param attributeName attribute to retrieve value for.
108         * @return value of the specified attribute.
109         */
110        public static String getRootAttribute(String filePath, String attributeName) {
111            ParamChecker.notNull(filePath, "filePath");
112            ParamChecker.notNull(attributeName, "attributeName");
113            SAXBuilder saxBuilder = createSAXBuilder();
114            try {
115                Document doc = saxBuilder.build(Thread.currentThread().getContextClassLoader().getResourceAsStream(filePath));
116                return doc.getRootElement().getAttributeValue(attributeName);
117            }
118            catch (JDOMException e) {
119                throw new RuntimeException();
120            }
121            catch (IOException e) {
122                throw new RuntimeException();
123            }
124        }

126        /**
127         * Pretty print string representation of an XML document that generates the pretty print on lazy mode when the
128         * {@link #toString} method is invoked.
129         */
130        public static class PrettyPrint {
131            private String str;
132            private Element element;
133    
134            private PrettyPrint(String str) {
135                this.str = str;
136            }
137    
138            private PrettyPrint(Element element) {
139                this.element = ParamChecker.notNull(element, "element");
140            }
141    
142            /**
143             * Return the pretty print representation of an XML document.
144             *
145             * @return the pretty print representation of an XML document.
146             */
147            public String toString() {
148                if (str != null) {
149                    return str;
150                }
151                else {
152                    XMLOutputter outputter = new XMLOutputter();
153                    StringWriter stringWriter = new StringWriter();
154                    outputter.setFormat(Format.getPrettyFormat());
155                    try {
156                        outputter.output(element, stringWriter);
157                    }
158                    catch (Exception ex) {
159                        throw new RuntimeException(ex);
160                    }
161                    return stringWriter.toString();
162                }
163            }
164        }
165    
166        /**
167         * Return a pretty print string for a JDOM Element.
168         *
169         * @param element JDOM element.
170         * @return pretty print of the given JDOM Element.
171         */
172        public static PrettyPrint prettyPrint(Element element) {
173            return new PrettyPrint(element);
174    
175        }
176    
177        /**
178         * Return a pretty print string for a XML string. If the given string is not valid XML it returns the original
179         * string.
180         *
181         * @param xmlStr XML string.
182         * @return prettyprint of the given XML string or the original string if the given string is not valid XML.
183         */
184        public static PrettyPrint prettyPrint(String xmlStr) {
185            try {
186                return new PrettyPrint(parseXml(xmlStr));
187            }
188            catch (Exception e) {
189                return new PrettyPrint(xmlStr);
190            }
191        }
192    
193        /**
194         * Return a pretty print string for a Configuration object.
195         *
196         * @param conf Configuration object.
197         * @return prettyprint of the given Configuration object.
198         */
199        public static PrettyPrint prettyPrint(Configuration conf) {
200            Element root = new Element("configuration");
201            for (Map.Entry<String, String> entry : conf) {
202                Element property = new Element("property");
203                Element name = new Element("name");
204                name.setText(entry.getKey());
205                Element value = new Element("value");
206                value.setText(entry.getValue());
207                property.addContent(name);
208                property.addContent(value);
209                root.addContent(property);
210            }
211            return new PrettyPrint(root);
212        }

214        /**
215         * Schema validation for a given xml.
216         * <p/>
217         *
218         *  @param schema for validation
219         *  @param xml to be validated
220         */
221        public static void validateXml(Schema schema, String xml) throws SAXException,IOException{
222    
223            Validator validator = schema.newValidator();
224            validator.validate(new StreamSource(new ByteArrayInputStream(xml.getBytes())));
225        }
226    
227        /** Create schema object for the given xsd
228         *
229         * @param is inputstream to schema.
230         * @return the schema object.
231         */
232        public static Schema createSchema(InputStream is){
233            SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
234            StreamSource src = new StreamSource(is);
235            try {
236                return factory.newSchema(src);
237            }
238            catch (SAXException e) {
239                throw new RuntimeException(e.getMessage(),e);
240            }
241        }
