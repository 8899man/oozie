039        /**
040         * Delete recursively a local directory.
041         *
042         * @param file directory to delete.
043         * @throws IOException thrown if the directory could not be deleted.
044         */
045        public static void delete(File file) throws IOException {
046            ParamChecker.notNull(file, "file");
047            if (file.getAbsolutePath().length() < 5) {
048                throw new RuntimeException(XLog.format("Path[{0}] is too short, not deleting", file.getAbsolutePath()));
049            }
050            if (file.exists()) {
051                if (file.isDirectory()) {
052                    File[] children = file.listFiles();
053                    if (children != null) {
054                        for (File child : children) {
055                            delete(child);
056                        }
057                    }
058                }
059                if (!file.delete()) {
060                    throw new RuntimeException(XLog.format("Could not delete path[{0}]", file.getAbsolutePath()));
061                }
062            }
063        }
064    
065        /**
066         * Return a reader as string.
067         * <p/>
068         *
069         * @param reader reader to read into a string.
070         * @param maxLen max content length allowed, if -1 there is no limit.
071         * @return the reader content.
072         * @throws IOException thrown if the resource could not be read.
073         */
074        public static String getReaderAsString(Reader reader, int maxLen) throws IOException {
075            ParamChecker.notNull(reader, "reader");
076            StringBuffer sb = new StringBuffer();
077            char[] buffer = new char[2048];
078            int read;
079            int count = 0;
080            while ((read = reader.read(buffer)) > -1) {
081                count += read;
082                if (maxLen > -1 && count > maxLen) {
083                    throw new IllegalArgumentException(XLog.format("stream exceeds limit [{0}]", maxLen));
084                }
085                sb.append(buffer, 0, read);
086            }
087            reader.close();
088            return sb.toString();
089        }

092        /**
093         * Return a classpath resource as a stream.
094         * <p/>
095         *
096         * @param path classpath for the resource.
097         * @param maxLen max content length allowed.
098         * @return the stream for the resource.
099         * @throws IOException thrown if the resource could not be read.
100         */
101        public static InputStream getResourceAsStream(String path, int maxLen) throws IOException {
102            ParamChecker.notEmpty(path, "path");
103            InputStream is = Thread.currentThread().getContextClassLoader().getResourceAsStream(path);
104            if (is == null) {
105                throw new IllegalArgumentException(XLog.format("resource [{0}] not found", path));
106            }
107            return is;
108        }

110        /**
111         * Return a classpath resource as a reader.
112         * <p/>
113         * It is assumed that the resource is a text resource.
114         *
115         * @param path classpath for the resource.
116         * @param maxLen max content length allowed.
117         * @return the reader for the resource.
118         * @throws IOException thrown if the resource could not be read.
119         */
120        public static Reader getResourceAsReader(String path, int maxLen) throws IOException {
121            return new InputStreamReader(getResourceAsStream(path, maxLen));
122        }

124        /**
125         * Return a classpath resource as string.
126         * <p/>
127         * It is assumed that the resource is a text resource.
128         *
129         * @param path classpath for the resource.
130         * @param maxLen max content length allowed.
131         * @return the resource content.
132         * @throws IOException thrown if the resource could not be read.
133         */
134        public static String getResourceAsString(String path, int maxLen) throws IOException {
135            ParamChecker.notEmpty(path, "path");
136            InputStream is = Thread.currentThread().getContextClassLoader().getResourceAsStream(path);
137            if (is == null) {
138                throw new IllegalArgumentException(XLog.format("resource [{0}] not found", path));
139            }
140            Reader reader = new InputStreamReader(is);
141            return getReaderAsString(reader, maxLen);
142        }

144        /**
145         * Copies an inputstream into an output stream.
146         *
147         * @param is inputstream to copy from.
148         * @param os outputstream to  copy to.
149         * @throws IOException thrown if the copy failed.
150         */
151        public static void copyStream(InputStream is, OutputStream os) throws IOException {
152            ParamChecker.notNull(is, "is");
153            ParamChecker.notNull(os, "os");
154            byte[] buffer = new byte[4096];
155            int read;
156            while ((read = is.read(buffer)) > -1) {
157                os.write(buffer, 0, read);
158            }
159            os.close();
160            is.close();
161        }

163        /**
164         * Copies an char input stream into an char output stream.
165         *
166         * @param reader reader to copy from.
167         * @param writer writer to  copy to.
168         * @throws IOException thrown if the copy failed.
169         */
170        public static void copyCharStream(Reader reader, Writer writer) throws IOException {
171            ParamChecker.notNull(reader, "reader");
172            ParamChecker.notNull(writer, "writer");
173            char[] buffer = new char[4096];
174            int read;
175            while ((read = reader.read(buffer)) > -1) {
176                writer.write(buffer, 0, read);
177            }
178            writer.close();
179            reader.close();
180        }

182        /**
183         * Zips a local directory, recursively, into a ZIP stream.
184         *
185         * @param dir directory to ZIP.
186         * @param relativePath basePath in the ZIP for the files, normally "/".
187         * @param zos the ZIP output stream to ZIP the directory.
188         * @throws java.io.IOException thrown if the directory could not be zipped.
189         */
190        public static void zipDir(File dir, String relativePath, ZipOutputStream zos) throws IOException {
191            zipDir(dir, relativePath, zos, true);
192            zos.close();
193        }
194    
195        private static void zipDir(File dir, String relativePath, ZipOutputStream zos, boolean start) throws IOException {
196            String[] dirList = dir.list();
197            for (String aDirList : dirList) {
198                File f = new File(dir, aDirList);
199                if (!f.isHidden()) {
200                    if (f.isDirectory()) {
201                        if (!start) {
202                            ZipEntry dirEntry = new ZipEntry(relativePath + f.getName() + "/");
203                            zos.putNextEntry(dirEntry);
204                            zos.closeEntry();
205                        }
206                        String filePath = f.getPath();
207                        File file = new File(filePath);
208                        zipDir(file, relativePath + f.getName() + "/", zos, false);
209                    }
210                    else {
211                        ZipEntry anEntry = new ZipEntry(relativePath + f.getName());
212                        zos.putNextEntry(anEntry);
213                        InputStream is = new FileInputStream(f);
214                        byte[] arr = new byte[4096];
215                        int read = is.read(arr);
216                        while (read > -1) {
217                            zos.write(arr, 0, read);
218                            read = is.read(arr);
219                        }
220                        is.close();
221                        zos.closeEntry();
222                    }
223                }
224            }
225        }

227        /**
228         * Creates a JAR file with the specified classes.
229         *
230         * @param baseDir local directory to create the JAR file, the staging 'classes' directory is created in there.
231         * @param jarName JAR file name, including extesion.
232         * @param classes classes to add to the JAR.
233         * @return an absolute File to the created JAR file.
234         * @throws java.io.IOException thrown if the JAR file could not be created.
235         */
236        public static File createJar(File baseDir, String jarName, Class ... classes) throws IOException {
237            File classesDir = new File(baseDir, "classes");
238            for (Class clazz : classes) {
239                String classPath = clazz.getName().replace(".", "/") + ".class";
240                String classFileName = classPath;
241                if (classPath.lastIndexOf("/") > -1) {
242                    classFileName = classPath.substring(classPath.lastIndexOf("/") + 1);
243                }
244                String packagePath = new File(classPath).getParent();
245                File dir = new File(classesDir, packagePath);
246                if (!dir.exists()) {
247                    if (!dir.mkdirs()) {
248                        throw new IOException(XLog.format("could not create dir [{0}]", dir));
249                    }
250                }
251                InputStream is = getResourceAsStream(classPath, -1);
252                OutputStream os = new FileOutputStream(new File(dir, classFileName));
253                copyStream(is, os);
254            }
255            File jar = new File(baseDir, jarName);
256            File jarDir = jar.getParentFile();
257            if (!jarDir.exists()) {
258                if (!jarDir.mkdirs()) {
259                    throw new IOException(XLog.format("could not create dir [{0}]", jarDir));
260                }
261            }
262            JarOutputStream zos = new JarOutputStream(new FileOutputStream(jar), new Manifest());
263            zipDir(classesDir, "", zos);
264            return jar;
265        }
