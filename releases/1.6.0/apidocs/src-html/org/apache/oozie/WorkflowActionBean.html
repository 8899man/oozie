020    import org.apache.oozie.client.WorkflowAction;
021    import org.apache.oozie.client.rest.JsonWorkflowAction;
022    import org.apache.oozie.util.ParamChecker;
023    import org.apache.oozie.util.PropertiesUtils;
024    import org.apache.oozie.util.WritableUtils;
025    import org.apache.hadoop.io.Writable;
026    
027    import java.util.Date;
028    import java.util.Properties;
029    import java.io.DataOutput;
030    import java.io.IOException;
031    import java.io.DataInput;
032    
033    /**
034     * Bean that contains all the information to start an action for a workflow node.
035     */
036    public class WorkflowActionBean extends JsonWorkflowAction implements Writable {
037        private String jobId;
038        private String executionPath;
039        private boolean pending;
040        private Date pendingAge;
041        private Date lastCheckTime;
042        private String signalValue;
043        private String logToken;
044    
045        /**
046         * Default constructor.
047         */
048        public WorkflowActionBean() {
049        }
050    
051        /**
052         * Serialize the action bean to a data output.
053         *
054         * @param dataOutput data output.
055         * @throws IOException thrown if the action bean could not be serialized.
056         */
057        public void write(DataOutput dataOutput) throws IOException {
058            WritableUtils.writeStr(dataOutput, getId());
059            WritableUtils.writeStr(dataOutput, getName());
060            WritableUtils.writeStr(dataOutput, getType());
061            WritableUtils.writeStr(dataOutput, getConf());
062            WritableUtils.writeStr(dataOutput, getStatus().toString());
063            dataOutput.writeInt(getRetries());
064            dataOutput.writeLong((getStartTime() != null) ? getStartTime().getTime() : -1);
065            dataOutput.writeLong((getEndTime() != null) ? getEndTime().getTime() : -1);
066            dataOutput.writeLong((getLastCheckTime() != null) ? getLastCheckTime().getTime() : -1);
067            WritableUtils.writeStr(dataOutput, getTransition());
068            WritableUtils.writeStr(dataOutput, getData());
069            WritableUtils.writeStr(dataOutput, getExternalId());
070            WritableUtils.writeStr(dataOutput, getExternalStatus());
071            WritableUtils.writeStr(dataOutput, getTrackerUri());
072            WritableUtils.writeStr(dataOutput, getConsoleUrl());
073            WritableUtils.writeStr(dataOutput, getErrorCode());
074            WritableUtils.writeStr(dataOutput, getErrorMessage());
075            WritableUtils.writeStr(dataOutput, jobId);
076            WritableUtils.writeStr(dataOutput, executionPath);
077            dataOutput.writeBoolean(pending);
078            dataOutput.writeLong((pendingAge != null) ? pendingAge.getTime() : -1);
079            WritableUtils.writeStr(dataOutput, signalValue);
080            WritableUtils.writeStr(dataOutput, logToken);
081        }
082    
083        /**
084         * Deserialize an action bean from a data input.
085         *
086         * @param dataInput data input.
087         * @throws IOException thrown if the action bean could not be deserialized.
088         */
089        public void readFields(DataInput dataInput) throws IOException {
090            setId(WritableUtils.readStr(dataInput));
091            setName(WritableUtils.readStr(dataInput));
092            setType(WritableUtils.readStr(dataInput));
093            setConf(WritableUtils.readStr(dataInput));
094            setStatus(WorkflowAction.Status.valueOf(WritableUtils.readStr(dataInput)));
095            setRetries(dataInput.readInt());
096            long d = dataInput.readLong();
097            if (d != -1) {
098                setStartTime(new Date(d));
099            }
100            d = dataInput.readLong();
101            if (d != -1) {
102                setEndTime(new Date(d));
103            }
104            d = dataInput.readLong();
105            if (d != -1) {
106                setLastCheckTime(new Date(d));
107            }
108            setTransition(WritableUtils.readStr(dataInput));
109            setData(WritableUtils.readStr(dataInput));
110            setExternalId(WritableUtils.readStr(dataInput));
111            setExternalStatus(WritableUtils.readStr(dataInput));
112            setTrackerUri(WritableUtils.readStr(dataInput));
113            setConsoleUrl(WritableUtils.readStr(dataInput));
114            setErrorInfo(WritableUtils.readStr(dataInput), WritableUtils.readStr(dataInput));
115            jobId = WritableUtils.readStr(dataInput);
116            executionPath = WritableUtils.readStr(dataInput);
117            pending = dataInput.readBoolean();
118            d = dataInput.readLong();
119            if (d != -1) {
120                pendingAge = new Date(d);
121            }
122            signalValue = WritableUtils.readStr(dataInput);
123            logToken = WritableUtils.readStr(dataInput);
124        }
125    
126        /**
127         * Return if the action execution is complete.
128         *
129         * @return if the action start is complete.
130         */
131        public boolean isExecutionComplete() {
132            return getStatus() == WorkflowAction.Status.DONE;
133        }
134    
135        /**
136         * Return if the action is complete.
137         *
138         * @return if the action is complete.
139         */
140        public boolean isComplete() {
141            return getStatus() == WorkflowAction.Status.OK || getStatus() == WorkflowAction.Status.KILLED ||
142                   getStatus() == WorkflowAction.Status.ERROR;
143        }
144    
145        /**
146         * Set the action as pending and the current time as pending.
147         */
148        public void setPending() {
149            pending = true;
150            pendingAge = new Date();
151        }
152    
153        /**
154         * Set a time when the action will be pending, normally a time in the future.
155         *
156         * @param pendingAge the time when the action will be pending.
157         */
158        public void setPendingAge(Date pendingAge) {
159            this.pendingAge = pendingAge;
160        }
161    
162        /**
163         * Return the pending age of the action.
164         *
165         * @return the pending age of the action, <code>null</code> if the action is not pending.
166         */
167        public Date getPendingAge() {
168            return pendingAge;
169        }
170    
171        /**
172         * Return if the action is pending.
173         *
174         * @return if the action is pending.
175         */
176        public boolean isPending() {
177            return pending;
178        }
179    
180        /**
181         * Removes the pending flag from the action.
182         */
183        public void resetPending() {
184            pending = false;
185            pendingAge = null;
186        }
187    
188    
189        /**
190         * Increments the number of retries for the action.
191         */
192        public void incRetries() {
193            setRetries(getRetries() + 1);
194        }
195    
196        /**
197         * Set a tracking information for an action, and set the action status to {@link org.apache.oozie.client.WorkflowAction.Status#DONE}
198         *
199         * @param externalId external ID for the action.
200         * @param trackerUri tracker URI for the action.
201         * @param consoleUrl console URL for the action.
202         */
203        public void setStartData(String externalId, String trackerUri, String consoleUrl) {
204            setExternalId(ParamChecker.notEmpty(externalId, "externalId"));
205            setTrackerUri(ParamChecker.notEmpty(trackerUri, "trackerUri"));
206            setConsoleUrl(ParamChecker.notEmpty(consoleUrl, "consoleUrl"));
207            Date now = new Date();
208            setStartTime(now);
209            setLastCheckTime(now);
210            setStatus(Status.RUNNING);
211        }
212    
213        /**
214         * Set the completion information for an action start. Sets the Action status to {@link org.apache.oozie.client.WorkflowAction.Status#DONE}
215         *
216         * @param externalStatus action external end status.
217         * @param actionData     action output data, <code>null</code> if there is no action output data.
218         */
219        public void setExecutionData(String externalStatus, Properties actionData) {
220            setStatus(Status.DONE);
221            setExternalStatus(ParamChecker.notEmpty(externalStatus, "externalStatus"));
222            if (actionData != null) {
223                setData(PropertiesUtils.propertiesToString(actionData));
224            }
225        }
226    
227        /**
228         * Set the completion information for an action end.
229         *
230         * @param status action status, {@link org.apache.oozie.client.WorkflowAction.Status#OK} or
231         *        {@link org.apache.oozie.client.WorkflowAction.Status#ERROR} or {@link org.apache.oozie.client.WorkflowAction.Status#KILLED}
232         * @param signalValue the signal value. In most cases, the value should be
233         *        OK or ERROR.
234         */
235        public void setEndData(Status status, String signalValue) {
236            if (status == null || (status != Status.OK && status != Status.ERROR && status != Status.KILLED)) {
237                throw new IllegalArgumentException("Action status must be OK, ERROR or KILLED. Received ["
238                        + status.toString() + "]");
239            }
240            if (status == Status.OK) {
241                setErrorInfo(null, null);
242            }
243            setStatus(status);
244            setSignalValue(ParamChecker.notEmpty(signalValue, "signalValue"));
245        }
246

250         * @return the job Id.
251         */
252        public String getJobId() {
253            return jobId;
254        }
255

257         * Set the job id.
258         *
259         * @param id jobId;
260         */
261        public void setJobId(String id) {
262            this.jobId = id;
263        }
264    
265        /**
266         * Return the node execution path.
267         *
268         * @return the node execution path.
269         */
270        public String getExecutionPath() {
271            return executionPath;
272        }
273    
274        /**
275         * Set the node execution path.
276         *
277         * @param executionPath the node execution path.
278         */
279        public void setExecutionPath(String executionPath) {
280            this.executionPath = executionPath;
281        }
282    
283        /**
284         * Return the signal value for the action.
285         * <p/>
286         * For decision nodes it is the choosen transition, for actions it is OK or ERROR.
287         *
288         * @return  the action signal value.
289         */
290        public String getSignalValue() {
291            return signalValue;
292        }
293    
294        /**
295         * Set the signal value for the action.
296         * <p/>
297         * For decision nodes it is the choosen transition, for actions it is OK or ERROR.
298         *
299         * @param signalValue the action signal value.
300         */
301        public void setSignalValue(String signalValue) {
302            this.signalValue = signalValue;
303        }
304    
305        /**
306         * Return the job log token.
307         *
308         * @return the job log token.
309         */
310        public String getLogToken() {
311            return logToken;
312        }
313    
314        /**
315         * Set the job log token.
316         *
317         * @param logToken the job log token.
318         */
319        public void setLogToken(String logToken) {
320            this.logToken = logToken;
321        }
322        
323        /**
324         * Return the action last check time
325         *
326         * @return the last check time
327         */
328        public Date getLastCheckTime() {
329            return lastCheckTime;
330        }
331    
332        /**
333         * Sets the action last check time
334         *
335         * @param lastCheckTime the last check time to set.
336         */
337        public void setLastCheckTime(Date lastCheckTime) {
338            this.lastCheckTime = lastCheckTime;
339        }
