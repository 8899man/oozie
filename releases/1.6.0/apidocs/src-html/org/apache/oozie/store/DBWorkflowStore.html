344        /**
345         * Load the action data and returns a bean.
346         * 
347         * @param id Action Id
348         * @param locking true if the action is to be locked
349         * @return Action Bean
350         * @throws StoreException If action doesn't exist
351         */
352        public WorkflowActionBean getAction(final String id, final boolean locking) throws StoreException {
353            ParamChecker.notEmpty(id, "ActionID");
354            WorkflowActionBean action = doOperation("getAction", new Callable<WorkflowActionBean>() {
355                public WorkflowActionBean call() throws SQLException, StoreException, WorkflowException, InterruptedException {
356                    WorkflowActionBean action = new WorkflowActionBean();
357                    ResultSet rs = null;
358                    Map<Object, Object> values = new HashMap<Object, Object>();
359                    values.put("1", id);
360                    if (selectForUpdate && locking) {
361                        if (prepGetActionForUpdate == null) {
362                            prepGetActionForUpdate = GET_ACTION_FOR_UPDATE.prepare(conn);
363                        }
364                        rs = GET_ACTION_FOR_UPDATE.getNewStatementWithValues(values).prepare(prepGetActionForUpdate)
365                                .executeQuery();
366                    }
367                    else {
368                        if (locking) {
369                            LockToken token = Services.get().get(MemoryLocksService.class).getWriteLock(id, LOCK_TIMEOUT);
370                            if(token == null) {
371                                throw new StoreException(ErrorCode.E0606, id);
372                            }
373                            locks.add(token);
374                        }
375                        if (prepGetAction == null) {
376                            prepGetAction = GET_ACTION.prepare(conn);
377                        }
378                        rs = GET_ACTION.getNewStatementWithValues(values).prepare(prepGetAction).executeQuery();
379                    }
380                    ResultSetReader rsReader;
381                    if (rs.next()) {
382                        rsReader = parse(rs);
383                        action = getBeanForRunningAction(rsReader);
384                        rsReader.close();
385                    }
386                    else {
387                        rs.close();
388                        throw new StoreException(ErrorCode.E0605, id);
389                    }
390                    return action;
391                }
392            });
393            return action;
394        }

542    
543                    SqlStatement s = GET_WORKFLOWS;
544                    SqlStatement countQuery = GET_WORKFLOWS_COUNT;
545                    List<Condition> andArray = new ArrayList<Condition>();
546                    for (Map.Entry<String, List<String>> entry : filter.entrySet()) {
547                        OozieColumn col = null;
548                        if (entry.getKey().equals(OozieClient.FILTER_GROUP)) {
549                            col = OozieColumn.WF_groupName;
550                        }
551                        else if (entry.getKey().equals(OozieClient.FILTER_NAME)) {
552                            col = OozieColumn.WF_appName;
553                        }
554                        else if (entry.getKey().equals(OozieClient.FILTER_STATUS)) {
555                            col = OozieColumn.WF_status;
556                        }
557                        else if (entry.getKey().equals(OozieClient.FILTER_USER)) {
558                            col = OozieColumn.WF_user;
559                        }
560                        if (col != null) {
561                            List<Condition> orArray = new ArrayList<Condition>();
562                            for (String val : entry.getValue()) {
563                                if (col.equals(OozieColumn.WF_status)) {
564                                    val = val.toUpperCase();
565                                }
566                                orArray.add(isEqual(col, val));
567                            }
568                            if (orArray.size() > 1) {
569                                andArray.add(or(orArray.toArray(new Condition[orArray.size()])));
570                            }
571                            else if (orArray.size() == 1) {
572                                andArray.add(orArray.get(0));
573                            }
574                        }
575                    }
576                    if (andArray.size() > 0) {
577                        s = ((Select) s).where(and(andArray.toArray(new Condition[andArray.size()])));
578                        countQuery = ((Select) countQuery).where(and(andArray.toArray(new Condition[andArray.size()])));
579                    }
580                    s = ((Select) s).orderBy(WF_startTime, false, WF_endTime, false);
581                    if ((start > 0) && (len > 0)) {
582                        s = ((Select) s).limit(start-1, len);
583                    }
584    
585                    int count = 0;
586                    ResultSet rsCount = countQuery.prepareAndSetValues(conn).executeQuery();
587                    if (rsCount.next()) {
588                        count = rsCount.getInt(1);
589                    }
590    
591                    int realLen = 0;
592                    ResultSetReader rsReader = parse(s.prepareAndSetValues(conn).executeQuery());
593                    while (rsReader.next()) {
594                        WorkflowJobBean wf = getBeanForRunningWorkflow(rsReader);
595                        wfBeans.add(wf);
596                        realLen++;
597                    }
598                    return new WorkflowsInfo(wfBeans, start, realLen, count);
599                }
600            });
601            return workFlowsInfo;
602        }

663        /**
664         * Purge the Workflows Completed older than given days.
665         * 
666         * @param olderThanDays number of days for which to preserve the workflows
667         * @throws StoreException
668         */
669        public void purge(final long olderThanDays) throws StoreException {
670            doOperation("purge", new Callable<Void>() {
671                public Void call() throws SQLException, StoreException, WorkflowException {
672                    Timestamp maxEndTime = new Timestamp(System.currentTimeMillis() - (olderThanDays * DAY_IN_MS));
673                    Map<Object, Object> values = new HashMap<Object, Object>();
674                    values.put("1", maxEndTime);
675                    if (prepGetCompletedWorkflowsOlderThan == null) {
676                        prepGetCompletedWorkflowsOlderThan = GET_COMPLETED_WORKFLOWS_OLDER_THAN.prepare(conn);
677                    }
678                    if (prepDeleteWorkflow == null) {
679                        prepDeleteWorkflow = DELETE_WORKFLOW.prepare(conn);
680                    }
681                    if (prepDeleteActionsForWorkflow == null) {
682                        prepDeleteActionsForWorkflow = DELETE_ACTIONS_FOR_WORKFLOW.prepare(conn);
683                    }
684                    ResultSetReader rsReader = parse(GET_COMPLETED_WORKFLOWS_OLDER_THAN.getNewStatementWithValues(values)
685                            .prepare(prepGetCompletedWorkflowsOlderThan).executeQuery());
686                    while (rsReader.next()) {
687                        Map<Object, Object> wfIdMap = new HashMap<Object, Object>();
688                        wfIdMap.put("1", rsReader.getString(WF_id));
689                        DELETE_WORKFLOW.getNewStatementWithValues(wfIdMap).prepare(prepDeleteWorkflow).executeUpdate();
690                        DELETE_ACTIONS_FOR_WORKFLOW.getNewStatementWithValues(wfIdMap)
691                                .prepare(prepDeleteActionsForWorkflow).executeUpdate();
692                    }
693                    rsReader.close();
694                    return null;
695                }
696            });
697        }

752        private <V> V doOperation(String name, Callable<V> command) throws StoreException {
753            try {
754                Instrumentation.Cron cron = new Instrumentation.Cron();
755                cron.start();
756                V retVal;
757                try {
758                    retVal = command.call();
759                }
760                finally {
761                    cron.stop();
762                }
763                Services.get().get(InstrumentationService.class).get().addCron(INSTR_GROUP, name, cron);
764                return retVal;
765            }
766            catch (StoreException ex) {
767                throw ex;
768            }
769            catch (SQLException ex) {
770                throw new StoreException(ErrorCode.E0603, name, ex.getMessage(), ex);
771            }
772            catch (Exception e) {
773                throw new StoreException(ErrorCode.E0607, name, e.getMessage(), e);
774            }
775        }

777        private WorkflowJobBean getWorkflowOnly(final String id, boolean locking) throws SQLException, InterruptedException,
778                StoreException {
779            ResultSet rs;
780            Map<Object, Object> values = new HashMap<Object, Object>();
781            values.put("1", id);
782            if (selectForUpdate && locking) {
783                if (prepGetWorkflowForUpdate == null) {
784                    prepGetWorkflowForUpdate = GET_WORKFLOW_FOR_UPDATE.prepare(conn);
785                }
786                rs = GET_WORKFLOW_FOR_UPDATE.getNewStatementWithValues(values).prepare(prepGetWorkflowForUpdate)
787                        .executeQuery();
788            }
789            else {
790                if (locking) {
791                    LockToken token = Services.get().get(MemoryLocksService.class).getWriteLock(id, LOCK_TIMEOUT);
792                    if(token == null) {
793                        throw new StoreException(ErrorCode.E0606, id);
794                    }
795                    locks.add(token);
796                }
797                if (prepGetWorkflow == null) {
798                    prepGetWorkflow = GET_WORKFLOW.prepare(conn);
799                }
800                rs = GET_WORKFLOW.getNewStatementWithValues(values).prepare(prepGetWorkflow).executeQuery();
801            }
802            if (!rs.next()) {
803                rs.close();
804                return null;
805            }
806            ResultSetReader rsReader = parse(rs);
807            WorkflowJobBean wfBean = getBeanForRunningWorkflow(rsReader);
808            rsReader.close();
809            return wfBean;
810        }

833        private List<WorkflowActionBean> getActionsOnlyForWorkflow(String wfId, boolean locking) throws SQLException,
834                InterruptedException, StoreException {
835            List<WorkflowActionBean> actions = new ArrayList<WorkflowActionBean>();
836            Map<Object, Object> values = new HashMap<Object, Object>();
837            values.put("1", wfId);
838            ResultSet rs = null;
839            if (selectForUpdate && locking) {
840                if (prepGetActionsForWorkflowForUpdate == null) {
841                    prepGetActionsForWorkflowForUpdate = GET_ACTIONS_OF_WORKFLOW_FOR_UPDATE.prepare(conn);
842                }
843                rs = GET_ACTIONS_OF_WORKFLOW_FOR_UPDATE.getNewStatementWithValues(values).prepare(
844                        prepGetActionsForWorkflowForUpdate).executeQuery();
845            }
846            else {
847                if (prepGetActionsForWorkflow == null) {
848                    prepGetActionsForWorkflow = GET_ACTIONS_FOR_WORKFLOW.prepare(conn);
849                }
850                rs = GET_ACTIONS_FOR_WORKFLOW.getNewStatementWithValues(values).prepare(prepGetActionsForWorkflow)
851                        .executeQuery();
852            }
853            ResultSetReader rsReader = parse(rs);
854            while (rsReader.next()) {
855                WorkflowActionBean action = getBeanForRunningAction(rsReader);
856                if (locking && !selectForUpdate) {
857                    LockToken token = Services.get().get(MemoryLocksService.class).getWriteLock(action.getId(),
858                            LOCK_TIMEOUT);
859                    if(token == null) {
860                        throw new StoreException(ErrorCode.E0606, action.getId());
861                    }
862                    locks.add(token);
863                }
864                actions.add(action);
865            }
866            rsReader.close();
867            return actions;
868        }
