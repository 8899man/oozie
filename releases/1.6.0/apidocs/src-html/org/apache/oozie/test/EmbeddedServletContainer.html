018    package org.apache.oozie.test;
019    
020    import org.mortbay.jetty.Server;
021    import org.mortbay.jetty.servlet.ServletHolder;
022    import org.mortbay.jetty.servlet.Context;
023    
024    import java.net.InetAddress;
025    import java.net.ServerSocket;
026    
027    /**
028     * An embedded servlet container for testing purposes.
029     * <p/>
030     * It provides reduced functionality, it supports only Servlets.
031     * <p/>
032     * The servlet container is started in a free port.
033     */
034    public class EmbeddedServletContainer {
035        private Server server;
036        private String host = null;
037        private int port = -1;
038        private String contextPath;
039        Context context;
040    
041        /**
042         * Create a servlet container.
043         *
044         * @param contextPath  context path for the servlet, it must not be prefixed or append with
045         *                     "/", for the default context use ""
046         */
047        public EmbeddedServletContainer(String contextPath) {
048            this.contextPath = contextPath;
049            server = new Server(0);
050            context = new Context();
051            context.setContextPath("/" + contextPath);
052            server.setHandler(context);
053        }
054    
055        /**
056         * Add a servlet to the container.
057         *
058         * @param servletPath  servlet path for the servlet, it should be prefixed with '/", it may
059         *                     contain a wild card at the end.
060         * @param servletClass servlet class
061         */
062        public void addServletEndpoint(String servletPath, Class servletClass) {
063            context.addServlet(new ServletHolder(servletClass), servletPath);
064        }
065    
066        /**
067         * Start the servlet container.
068         * <p/>
069         * The container starts on a free port.
070         *
071         * @throws Exception thrown if the container could not start.
072         */
073        public void start()  throws Exception {
074            host = InetAddress.getLocalHost().getHostName();
075            ServerSocket ss = new ServerSocket(0);
076            port = ss.getLocalPort();
077            ss.close();
078            server.getConnectors()[0].setHost(host);
079            server.getConnectors()[0].setPort(port);
080            server.start();
081            System.out.println("Running embedded servlet container at: http://" + host + ":" + port);
082        }
083    
084        /**
085         * Return the hostname the servlet container is bound to.
086         *
087         * @return the hostname.
088         */
089        public String getHost() {
090            return host;
091        }
092    
093        /**
094         * Return the port number the servlet container is bound to.
095         *
096         * @return the port number.
097         */
098        public int getPort() {
099            return port;
100        }
101    
102        /**
103         * Return the full URL (including protocol, host, port, context path, servlet path) for the context path.
104         * 
105         * @return URL to the context path.
106         */
107        public String getContextURL() {
108            return "http://" + host + ":" + port + "/" + contextPath;
109        }
110    
111        /**
112         * Return the full URL (including protocol, host, port, context path, servlet path) for a servlet path.
113         *
114         * @param servletPath the path which will be expanded to a full URL.
115         * @return URL to the servlet.
116         */
117        public String getServletURL(String servletPath) {
118            String path = servletPath;
119            if (path.endsWith("*")) {
120                path = path.substring(0, path.length() - 1);
121            }
122            return getContextURL() + path;
123        }
124    
125        /**
126         * Stop the servlet container.
127         */
128        public void stop() {
129            try {
130            server.stop();
131            }
132            catch (Exception e) {
133                // ignore exception
134            }
135    
136            try {
137                server.destroy();
138            }
139            catch (Exception e) {
140                // ignore exception
141            }
142    
143            host = null;
144            port = -1;
145        }
146    
147    }
