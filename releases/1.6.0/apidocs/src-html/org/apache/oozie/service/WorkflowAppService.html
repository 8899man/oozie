041    /**
042     * Service that provides application workflow definition reading from the path
043     * and creation of the proto configuration.
044     */
045    public abstract class WorkflowAppService implements Service {
046    
047        public static final String APP_LIB_JAR_PATH_LIST = "oozie.wf.application.jar.paths";
048    
049        public static final String APP_LIB_SO_PATH_LIST = "oozie.wf.applcation.so.paths";
050    
051        public static final String HADOOP_UGI = "hadoop.job.ugi";
052    
053        public static final String HADOOP_USER = "user.name";
054    
055        public static final String HADOOP_JT_KERBEROS_NAME = "mapreduce.jobtracker.kerberos.principal";
056    
057        public static final String HADOOP_NN_KERBEROS_NAME = "dfs.namenode.kerberos.principal";
058    
059        /**
060         * Initialize the workflow application service.
061         *
062         * @param services services instance.
063         */
064        public void init(Services services) {
065        }
066    
067        /**
068         * Destroy the workflow application service.
069         */
070        public void destroy() {
071        }
072    
073        /**
074         * Return the public interface for workflow application service.
075         *
076         * @return {@link WorkflowAppService}.
077         */
078        public Class<? extends Service> getInterface() {
079            return WorkflowAppService.class;
080        }
081    
082        /**
083         * Read workflow definition.
084         *
085         * @param appPath  application path.
086         * @param user     user name.
087         * @param group    group name.
088         * @param autToken authentication token.
089         * @return workflow definition.
090         * @throws WorkflowException thrown if the definition could not be read.
091         */
092        protected String readDefinition(String appPath, String user, String group, String autToken)
093                throws WorkflowException {
094            try {
095                URI uri = new URI(appPath);
096                FileSystem fs = Services.get().get(HadoopAccessorService.class).
097                        createFileSystem(user, group, uri, new Configuration());
098                Reader reader = new InputStreamReader(fs.open(new Path(uri.getPath(), "workflow.xml")));
099                StringWriter writer = new StringWriter();
100                IOUtils.copyCharStream(reader, writer);
101                return writer.toString();
102    
103            }
104            catch (IOException ex) {
105                throw new WorkflowException(ErrorCode.E0710, ex.getMessage(), ex);
106            }
107            catch (URISyntaxException ex) {
108                throw new WorkflowException(ErrorCode.E0711, appPath, ex.getMessage(), ex);
109            }
110            catch (Exception ex) {
111                throw new WorkflowException(ErrorCode.E0710, ex.getMessage(), ex);
112            }
113        }
114    
115        /**
116         * Create proto configuration.
117         * <p/>
118         * The proto configuration includes the user,group and the paths which need
119         * to be added to distributed cache. These paths include .jar,.so and the
120         * resource file paths.
121         *
122         * @param jobConf   job configuration.
123         * @param authToken authentication token.
124         * @return proto configuration.
125         * @throws WorkflowException thrown if the proto action configuration could not be created.
126         */
127        public XConfiguration createProtoActionConf(Configuration jobConf, String authToken) throws WorkflowException {
128            XConfiguration conf = new XConfiguration();
129            try {
130                String user = jobConf.get(OozieClient.USER_NAME);
131                String group = jobConf.get(OozieClient.GROUP_NAME);
132                String hadoopUgi = user + "," + group;
133    
134                conf.set(OozieClient.USER_NAME, user);
135                conf.set(OozieClient.GROUP_NAME, group);
136                conf.set(HADOOP_UGI, hadoopUgi);
137                conf.set(HADOOP_JT_KERBEROS_NAME, jobConf.get(HADOOP_JT_KERBEROS_NAME));
138                conf.set(HADOOP_NN_KERBEROS_NAME, jobConf.get(HADOOP_NN_KERBEROS_NAME));
139    
140                URI uri = new URI(jobConf.get(OozieClient.APP_PATH));
141    
142                FileSystem fs = Services.get().get(HadoopAccessorService.class).createFileSystem(user, group, uri, conf);
143    
144                Path appPath = new Path(uri.getPath());
145                List<String> jarFilePaths = getLibPaths(fs, appPath, ".jar");
146                List<String> soFilepaths = getLibPaths(fs, appPath, ".so");
147                conf.setStrings(APP_LIB_JAR_PATH_LIST, jarFilePaths.toArray(new String[jarFilePaths.size()]));
148                conf.setStrings(APP_LIB_SO_PATH_LIST, soFilepaths.toArray(new String[soFilepaths.size()]));
149                return conf;
150    
151            }
152            catch (IOException ex) {
153                throw new WorkflowException(ErrorCode.E0712, jobConf.get(OozieClient.APP_PATH),
154                                            ex.getMessage(), ex);
155            }
156            catch (URISyntaxException ex) {
157                throw new WorkflowException(ErrorCode.E0711, jobConf.get(OozieClient.APP_PATH),
158                                            ex.getMessage(), ex);
159            }
160            catch (Exception ex) {
161                throw new WorkflowException(ErrorCode.E0712, jobConf.get(OozieClient.APP_PATH),
162                                            ex.getMessage(), ex);
163            }
164        }
165    
166        /**
167         * Parse workflow definition.
168         *
169         * @param jobConf   job configuration.
170         * @param authToken authentication token.
171         * @return workflow application.
172         * @throws WorkflowException thrown if the workflow application could not be parsed.
173         */
174        public abstract WorkflowApp parseDef(Configuration jobConf, String authToken) throws WorkflowException;
175    
176        /**
177         * Get library paths for a given extension.
178         *
179         * @param fs        file system object.
180         * @param appPath   hdfs application path.
181         * @param extension to be listed.
182         * @return list of paths.
183         * @throws IOException thrown if the lib paths could not be obtained.
184         */
185        private List<String> getLibPaths(FileSystem fs, Path appPath, String extension) throws IOException {
186            List<String> libPaths = new ArrayList<String>();
187            FileStatus[] files =
188                    fs.listStatus(new Path(appPath + "/lib"), new LibPathFilter(appPath.toUri().getPath(), extension));
189            for (FileStatus file : files) {
190                libPaths.add((String) file.getPath().toUri().getPath().trim());
191            }
192            return libPaths;
193        }
194    
195        /**
196         * Filter for listing library paths.
197         */
198        private class LibPathFilter implements PathFilter {
199            private String extension;
200            private String libPath;
201    
202            /**
203             * Creates library paths filter.
204             *
205             * @param appPath   workflow application path.
206             * @param extension file extension to be listed.
207             */
208            public LibPathFilter(String appPath, String extension) {
209                this.extension = extension;
210                this.libPath = appPath + "/lib";
211            }
212    
213            /**
214             * Check the library path.
215             *
216             * @param path path to be checked
217             * @return true if path has the extension and start with application
218             *         /lib directory else false
219             */
220            @Override
221            public boolean accept(Path path) {
222                return path.getName().trim().endsWith(extension) && path.toUri().getPath().startsWith(libPath);
223            }
224        }
