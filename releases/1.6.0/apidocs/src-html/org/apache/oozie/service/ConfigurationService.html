038    /**
039     * Built in service that initializes the services configuration.
040     * <p/>
041     * The configuration loading sequence is identical to Hadoop configuration loading sequence.
042     * <p/>
043     * First the default values are loaded from the {@link #DEFAULT_CONFIG_FILE}, then the site configured values are
044     * loaded from the site configuration file.
045     * <p/>
046     * The {@link #DEFAULT_CONFIG_FILE} is always loaded from the classpath root.
047     * <p/>
048     * The site configuration file and loading location is determined as follow:
049     * <p/>
050     * The site configuration file name is set by the system property {@link #CONFIG_FILE}. The default value is
051     * {@link #SITE_CONFIG_FILE}.
052     * <p/>
053     * The site configuration file is loaded from the directory specified by the system property {@link #CONFIG_PATH}. If
054     * not set, no site configuration is loaded.
055     * <p/>
056     * Configuration properties, prefixed with 'oozie.', passed as system properties overrides default and site values.
057     * <p/>
058     * The configuration service logs details on how the configuration was loaded as well as what properties were overriden
059     * via system properties settings.
060     */
061    public class ConfigurationService implements Service, Instrumentable {
062        private static final String INSTRUMENTATION_GROUP = "configuration";
063    
064        public static final String CONF_PREFIX = Service.CONF_PREFIX + "ConfigurationService.";
065    
066        public static final String CONF_IGNORE_SYS_PROPS = CONF_PREFIX + "ignore.system.properties";
067    
068        /**
069         * System property that indicates the path where the site configuration should be loaded from.
070         */
071        public static final String CONFIG_PATH = "oozie.config.dir";
072    
073        /**
074         * System property that indicates the name of the site configuration file to load.
075         */
076        public static final String CONFIG_FILE = "oozie.config.file";
077    
078        private static final Set<String> IGNORE_SYS_PROPS = new HashSet<String>();
079        private static final String IGNORE_SYS_PROPS_PREFIX = "oozie.test.";
080    
081        private static final String MASK_PROPS_VALUES_POSTIX = ".password";
082    
083        static {
084            IGNORE_SYS_PROPS.add(CONFIG_FILE);
085            IGNORE_SYS_PROPS.add(CONFIG_PATH);
086            IGNORE_SYS_PROPS.add(XLogService.LOG4J_FILE);
087            IGNORE_SYS_PROPS.add(XLogService.RELOAD_INTERVAL);
088            IGNORE_SYS_PROPS.add(CONF_IGNORE_SYS_PROPS);
089        }
090    
091        //for testing purposes
092        static boolean testingDefaultFile;
093    
094        public static final String DEFAULT_CONFIG_FILE = "oozie-default.xml";
095        public static final String SITE_CONFIG_FILE = "oozie-site.xml";
096    
097        private final XLog log = XLog.getLog(getClass());
098    
099        private String configDir;
100        private String configFile;
101    
102        private LogChangesConfiguration configuration;
103    
104        public ConfigurationService() {
105            testingDefaultFile = false;
106        }
107    
108        /**
109         * Initialize the log service.
110         *
111         * @param services services instance.
112         * @throws ServiceException thrown if the log service could not be initialized.
113         */
114        public void init(Services services) throws ServiceException {
115            configDir = System.getProperty(CONFIG_PATH);
116            configFile = System.getProperty(CONFIG_FILE, SITE_CONFIG_FILE);
117            configuration = loadConf();
118        }
119    
120        /**
121         * Destroy the configuration service.
122         */
123        public void destroy() {
124            configuration = null;
125        }
126    
127        /**
128         * Return the public interface for configuration service.
129         *
130         * @return {@link ConfigurationService}.
131         */
132        public Class<? extends Service> getInterface() {
133            return ConfigurationService.class;
134        }
135    
136        /**
137         * Return the services configuration.
138         *
139         * @return the services configuration.
140         */
141        public Configuration getConf() {
142            if (configuration == null) {
143                throw new IllegalStateException("Not initialized");
144            }
145            return configuration;
146        }
147    
148        private InputStream getConfFileFromClassPath(String configFile) throws IOException {
149            ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
150            InputStream inputStream = classLoader.getResourceAsStream(configFile);
151            if (inputStream != null) {
152                URL fileLocation = classLoader.getResource(configFile);
153                log.info("Loading configuration file [{0}] from classpath [{1}]", configFile, fileLocation);
154            }
155            else {
156                log.warn("Configuration file [{0}] not available in classpath", configFile);
157            }
158            return inputStream;
159        }
160    
161        private InputStream getConfFile(String configFile) throws ServiceException, IOException {
162            if (configDir == null) {
163                return getConfFileFromClassPath(configFile);
164            }
165            else {
166                if (!configDir.startsWith("/")) {
167                    throw new ServiceException(ErrorCode.E0020, configDir);
168                }
169                if (configFile.contains("/")) {
170                    throw new ServiceException(ErrorCode.E0022, configFile);
171                }
172                File file = new File(configDir, configFile);
173                if (file.exists()) {
174                    log.info("Loading configuration file from path [{0}]", file.getAbsolutePath());
175                    return new FileInputStream(file);
176                }
177                else {
178                    log.info("Configuration file [{0}] not available at [{1}]", configFile, configDir);
179                    return null;
180                }
181            }
182        }
183    
184        private LogChangesConfiguration loadConf() throws ServiceException {
185            XConfiguration configuration;
186            try {
187                String defaultConfig = (testingDefaultFile) ? "test-" + DEFAULT_CONFIG_FILE : DEFAULT_CONFIG_FILE;
188                InputStream inputStream = getConfFileFromClassPath(defaultConfig);
189                if (inputStream != null) {
190                    configuration = new XConfiguration(inputStream);
191                }
192                else {
193                    throw new ServiceException(ErrorCode.E0023, defaultConfig);
194                }
195                if (configFile != null) {
196                    inputStream = getConfFile(configFile);
197                    if (inputStream != null) {
198                        XConfiguration siteConfiguration = new XConfiguration(inputStream);
199                        XConfiguration.injectDefaults(configuration, siteConfiguration);
200                        configuration = siteConfiguration;
201                    }
202                } 
203            }
204            catch (IOException ex) {
205                throw new ServiceException(ErrorCode.E0024, configFile, ex.getMessage(), ex);
206            }
207    
208            if (log.isTraceEnabled()) {
209                try {
210                    StringWriter writer = new StringWriter();
211                    for (Map.Entry<String, String> entry : configuration) {
212                        boolean maskValue = entry.getKey().endsWith(MASK_PROPS_VALUES_POSTIX);
213                        String value = (maskValue) ? "**MASKED**" : entry.getValue();
214                        writer.write(" " + entry.getKey() + " = " + value + "\n");
215                    }
216                    writer.close();
217                    log.trace("Configuration:\n{0}---", writer.toString());
218                }
219                catch (IOException ex) {
220                    throw new ServiceException(ErrorCode.E0025, ex.getMessage(), ex);
221                }
222            }
223    
224            String[] ignoreSysProps = configuration.getStrings(CONF_IGNORE_SYS_PROPS);
225            if (ignoreSysProps != null) {
226                IGNORE_SYS_PROPS.addAll(Arrays.asList(ignoreSysProps));
227            }
228    
229            for (Map.Entry<String, String> entry : configuration) {
230                String sysValue = System.getProperty(entry.getKey());
231                if (sysValue != null && !IGNORE_SYS_PROPS.contains(entry.getKey())) {
232                    log.info("SysProps configuration change, property[{0}]=[{1}]", entry.getKey(), sysValue);
233                    configuration.set(entry.getKey(), sysValue);
234                }
235            }
236            for (Map.Entry<Object, Object> entry : System.getProperties().entrySet()) {
237                String name = (String) entry.getKey();
238                if (IGNORE_SYS_PROPS.contains(name) && !name.startsWith(IGNORE_SYS_PROPS_PREFIX)) {
239                    log.warn("System property [{0}] in ignore list, ignored", name);
240                }
241                else
242                if (name.startsWith("oozie.")) {
243                    if (configuration.get(name) == null) {
244                        log.warn("System property [{0}] no defined in Oozie configuration, ignored", name);
245                    }
246                }
247            }
248    
249            return new LogChangesConfiguration(configuration);
250        }
251    
252        private class LogChangesConfiguration extends XConfiguration {
253    
254            public LogChangesConfiguration(Configuration conf) {
255                for (Map.Entry<String, String> entry : conf) {
256                    if (get(entry.getKey()) == null) {
257                        setValue(entry.getKey(), entry.getValue());
258                    }
259                }
260            }
261    
262            public String[] getStrings(String name) {
263                String s = get(name);
264                return (s != null && s.trim().length() > 0) ? super.getStrings(name) : new String[0];
265            }
266    
267            public String get(String name, String defaultValue) {
268                String value = get(name);
269                if (value == null) {
270                    boolean maskValue = name.endsWith(MASK_PROPS_VALUES_POSTIX);
271                    value = (maskValue) ? "**MASKED**" : defaultValue;
272                    log.warn(XLog.OPS, "Configuration property [{0}] not found, using default [{1}]", name, value);
273                }
274                return value;
275            }
276            
277            public void set(String name, String value) {
278                setValue(name, value);
279                boolean maskValue = name.endsWith(MASK_PROPS_VALUES_POSTIX);
280                value = (maskValue) ? "**MASKED**" : value;
281                log.info(XLog.OPS, "Programmatic configuration change, property[{0}]=[{1}]", name, value);
282            }
283    
284            private void setValue(String name, String value) {
285                super.set(name, value);
286            }
287    
288        }
289    
290        /**
291         * Instruments the configuration service.
292         * <p/>
293         * It sets instrumentation variables indicating the config dir and config file used.
294         *
295         * @param instr instrumentation to use.
296         */
297        public void instrument(Instrumentation instr) {
298            instr.addVariable(INSTRUMENTATION_GROUP, "config.dir", new Instrumentation.Variable<String>() {
299                public String getValue() {
300                    return configDir;
301                }
302            });
303            instr.addVariable(INSTRUMENTATION_GROUP, "config.file", new Instrumentation.Variable<String>() {
304                public String getValue() {
305                    return configFile;
306                }
307            });
308            instr.setConfiguration(configuration);
309        }
310    
311        /**
312         * Return a configuration with all sensitive values masked.
313         *
314         * @param conf configuration to mask.
315         * @return masked configuration.
316         */
317        public static Configuration maskSecretValues(Configuration conf) {
318            XConfiguration maskedConf = new XConfiguration();
319            for (Map.Entry<String, String> entry : conf) {
320                String name = entry.getKey();
321                boolean maskValue = name.endsWith(MASK_PROPS_VALUES_POSTIX);
322                String value = (maskValue) ? "**MASKED**" : entry.getValue();
323                maskedConf.set(name, value);
324            }
325            return maskedConf;
326        }
