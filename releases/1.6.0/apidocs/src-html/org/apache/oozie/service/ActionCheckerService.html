018    package org.apache.oozie.service;
019    
020    import java.util.ArrayList;
021    import java.util.List;
022    import org.apache.hadoop.conf.Configuration;
023    import org.apache.oozie.WorkflowActionBean;
024    import org.apache.oozie.command.wf.ActionCheckCommand;
025    import org.apache.oozie.store.StoreException;
026    import org.apache.oozie.store.WorkflowStore;
027    import org.apache.oozie.service.CallableQueueService;
028    import org.apache.oozie.service.InstrumentationService;
029    import org.apache.oozie.service.SchedulerService;
030    import org.apache.oozie.service.Service;
031    import org.apache.oozie.service.Services;
032    import org.apache.oozie.util.XCallable;
033    import org.apache.oozie.util.XLog;
034    
035    /**
036     * The Action Checker Service queue ActionCheckCommands to check the status of
037     * running actions. The delay between checks on the same action can be
038     * configured.
039     */
040    public class ActionCheckerService implements Service {
041    
042        public static final String CONF_PREFIX = Service.CONF_PREFIX + "ActionCheckerService.";
043        /**
044         * The frequency at which the ActionCheckService will run.
045         */
046        public static final String CONF_ACTION_CHECK_INTERVAL = CONF_PREFIX + "action.check.interval";
047        /**
048         * The time, in seconds, between an ActionCheck for the same action.
049         */
050        public static final String CONF_ACTION_CHECK_DELAY = CONF_PREFIX + "action.check.delay";
051    
052        /**
053         * The number of callables to be queued in a batch.
054         */
055        public static final String CONF_CALLABLE_BATCH_SIZE = CONF_PREFIX + "callable.batch.size";
056    
057        protected static final String INSTRUMENTATION_GROUP = "actionchecker";
058        protected static final String INSTR_CHECK_ACTIONS_COUNTER = "checks";
059    
060        /**
061         * {@link ActionCheckRunnable} is the runnable which is scheduled to run and
062         * queue Action checks.
063         */
064        static class ActionCheckRunnable implements Runnable {
065            private int actionCheckDelay;
066            private List<XCallable<Void>> callables;
067    
068            public ActionCheckRunnable(int actionCheckDelay) {
069                this.actionCheckDelay = actionCheckDelay;
070            }
071    
072            public void run() {
073                XLog.Info.get().clear();
074                XLog log = XLog.getLog(getClass());
075    
076                WorkflowStore store = null;
077                try {
078                    store = Services.get().get(WorkflowStoreService.class).create();
079                    List<WorkflowActionBean> actions =  store.getRunningActions(actionCheckDelay);
080                    for (WorkflowActionBean action : actions) {
081                        if (action.isPending() && action.getStatus() == WorkflowActionBean.Status.RUNNING) {
082                            Services.get().get(InstrumentationService.class).get().incr(INSTRUMENTATION_GROUP,
083                                    INSTR_CHECK_ACTIONS_COUNTER, 1);
084                            queueCallable(new ActionCheckCommand(action.getId(), action.getType()));
085                        }
086                    }
087                    if (null != callables) {
088                        Services.get().get(CallableQueueService.class).queueSerial(callables);
089                        callables = null;
090                    }
091                    log.info("Queuing [{0}] running actions for external status check", actions.size());
092                }
093                catch (StoreException ex) {
094                    log.warn(XLog.OPS, "Exception while accessing the store", ex);
095                }
096                finally {
097                    try {
098                        if (store != null) {
099                            store.close();
100                        }
101                    }
102                    catch (StoreException ex) {
103                        log.warn("Exception while attempting to close store", ex);
104                    }
105                }
106            }
107    
108            /**
109             * Adds callables to a list. If the number of callables in the list
110             * reaches {@link ActionCheckerService#CONF_CALLABLE_BATCH_SIZE}, the
111             * entire batch is queued and the callables list is reset.
112             *
113             * @param callable the callable to queue.
114             */
115            private void queueCallable(XCallable<Void> callable) {
116                if (callables == null) {
117                    callables = new ArrayList<XCallable<Void>>();
118                }
119                callables.add(callable);
120                if (callables.size() == Services.get().getConf().getInt(CONF_CALLABLE_BATCH_SIZE, 10)) {
121                    Services.get().get(CallableQueueService.class).queueSerial(callables);
122                    callables = new ArrayList<XCallable<Void>>();
123                }
124            }
125        }
126    
127        /**
128         * Initializes the Action Check service.
129         * 
130         * @param services services instance.
131         */
132        @Override
133        public void init(Services services) {
134            Configuration conf = services.getConf();
135            Runnable actionCheckRunnable = new ActionCheckRunnable(conf.getInt(CONF_ACTION_CHECK_DELAY, 600));
136            services.get(SchedulerService.class).schedule(actionCheckRunnable, 10,
137                                                          conf.getInt(CONF_ACTION_CHECK_INTERVAL, 60),
138                                                          SchedulerService.Unit.SEC);
139        }
140    
141        /**
142         * Destroy the Action Checker Services.
143         */
144        @Override
145        public void destroy() {
146        }
147    
148        /**
149         * Return the public interface for the action checker service.
150         *
151         * @return {@link ActionCheckerService}.
152         */
153        @Override
154        public Class<? extends Service> getInterface() {
155            return ActionCheckerService.class;
156        }
157    }
