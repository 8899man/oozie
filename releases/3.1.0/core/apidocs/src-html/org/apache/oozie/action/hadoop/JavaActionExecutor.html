153        public Configuration createBaseHadoopConf(Context context, Element actionXml) {
154            Configuration conf = new XConfiguration();
155            conf.set(HADOOP_USER, context.getProtoActionConf().get(WorkflowAppService.HADOOP_USER));
156            conf.set(HADOOP_UGI, context.getProtoActionConf().get(WorkflowAppService.HADOOP_UGI));
157            if (context.getProtoActionConf().get(WorkflowAppService.HADOOP_JT_KERBEROS_NAME) != null) {
158                conf.set(WorkflowAppService.HADOOP_JT_KERBEROS_NAME, context.getProtoActionConf().get(
159                        WorkflowAppService.HADOOP_JT_KERBEROS_NAME));
160            }
161            if (context.getProtoActionConf().get(WorkflowAppService.HADOOP_NN_KERBEROS_NAME) != null) {
162                conf.set(WorkflowAppService.HADOOP_NN_KERBEROS_NAME, context.getProtoActionConf().get(
163                        WorkflowAppService.HADOOP_NN_KERBEROS_NAME));
164            }
165            conf.set(OozieClient.GROUP_NAME, context.getProtoActionConf().get(OozieClient.GROUP_NAME));
166            Namespace ns = actionXml.getNamespace();
167            String jobTracker = actionXml.getChild("job-tracker", ns).getTextTrim();
168            String nameNode = actionXml.getChild("name-node", ns).getTextTrim();
169            conf.set(HADOOP_JOB_TRACKER, jobTracker);
170            conf.set(HADOOP_NAME_NODE, nameNode);
171            conf.set("mapreduce.fileoutputcommitter.marksuccessfuljobs", "true");
172            return conf;
173        }

175        Configuration setupLauncherConf(Configuration conf, Element actionXml, Path appPath, Context context)
176                throws ActionExecutorException {
177            try {
178                Namespace ns = actionXml.getNamespace();
179                Element e = actionXml.getChild("configuration", ns);
180                if (e != null) {
181                    String strConf = XmlUtils.prettyPrint(e).toString();
182                    XConfiguration inlineConf = new XConfiguration(new StringReader(strConf));
183    
184                    XConfiguration launcherConf = new XConfiguration();
185                    for (Map.Entry<String, String> entry : inlineConf) {
186                        if (entry.getKey().startsWith("oozie.launcher.")) {
187                            String name = entry.getKey().substring("oozie.launcher.".length());
188                            String value = entry.getValue();
189                            // setting original KEY
190                            launcherConf.set(entry.getKey(), value);
191                            // setting un-prefixed key (to allow Hadoop job config
192                            // for the launcher job
193                            launcherConf.set(name, value);
194                        }
195                    }
196                    checkForDisallowedProps(launcherConf, "inline launcher configuration");
197                    XConfiguration.copy(launcherConf, conf);
198                }
199                return conf;
200            }
201            catch (IOException ex) {
202                throw convertException(ex);
203            }
204        }

229                Element e = actionXml.getChild("job-xml", ns);

243                    XConfiguration.copy(inlineConf, actionConf);
244                }
245                return actionConf;
246            }
247            catch (IOException ex) {
248                throw convertException(ex);
249            }

252        Configuration addToCache(Configuration conf, Path appPath, String filePath, boolean archive)
253                throws ActionExecutorException {
254            Path path = null;
255            try {
256                if (filePath.startsWith("/")) {
257                    path = new Path(filePath);
258                }
259                else {
260                    path = new Path(appPath, filePath);
261                }
262                URI uri = new URI(path.toUri().getPath());
263                if (archive) {
264                    DistributedCache.addCacheArchive(uri, conf);
265                }
266                else {
267                    String fileName = filePath.substring(filePath.lastIndexOf("/") + 1);
268                    if (fileName.endsWith(".so") || fileName.contains(".so.")) {  // .so files
269                        uri = new Path(path.toString() + "#" + fileName).toUri();
270                        uri = new URI(uri.getPath());
271                        DistributedCache.addCacheFile(uri, conf);
272                    }
273                    else if (fileName.endsWith(".jar")) { // .jar files
274                        if (!fileName.contains("#")) {
275                            path = new Path(uri.toString());
276    
277                            String user = conf.get("user.name");
278                            String group = conf.get("group.name");
279                            Services.get().get(HadoopAccessorService.class).addFileToClassPath(user, group, path, conf);
280                        }
281                        else {
282                            DistributedCache.addCacheFile(uri, conf);
283                        }
284                    }
285                    else { // regular files
286                        if (!fileName.contains("#")) {
287                            uri = new Path(path.toString() + "#" + fileName).toUri();
288                            uri = new URI(uri.getPath());
289                        }
290                        DistributedCache.addCacheFile(uri, conf);
291                    }
292                }
293                DistributedCache.createSymlink(conf);
294                return conf;
295            }
296            catch (Exception ex) {
297                XLog.getLog(getClass()).debug(
298                        "Errors when add to DistributedCache. Path=" + path + ", archive=" + archive + ", conf="
299                                + XmlUtils.prettyPrint(conf).toString());
300                throw convertException(ex);
301            }
302        }

313        public void prepareActionDir(FileSystem actionFs, Context context) throws ActionExecutorException {
314            try {
315                Path actionDir = context.getActionDir();
316                Path tempActionDir = new Path(actionDir.getParent(), actionDir.getName() + ".tmp");
317                if (!actionFs.exists(actionDir)) {
318                    try {
319                        actionFs.copyFromLocalFile(new Path(getOozieRuntimeDir(), getLauncherJarName()), new Path(
320                                tempActionDir, getLauncherJarName()));
321                        actionFs.rename(tempActionDir, actionDir);
322                    }
323                    catch (IOException ex) {
324                        actionFs.delete(tempActionDir, true);
325                        actionFs.delete(actionDir, true);
326                        throw ex;
327                    }
328                }
329            }
330            catch (Exception ex) {
331                throw convertException(ex);
332            }
333        }

348        @SuppressWarnings("unchecked")
349        void setLibFilesArchives(Context context, Element actionXml, Path appPath, Configuration conf)
350                throws ActionExecutorException {
351            Configuration proto = context.getProtoActionConf();
352    
353            addToCache(conf, appPath, getOozieLauncherJar(context), false);
354    
355            String[] paths = proto.getStrings(WorkflowAppService.APP_LIB_PATH_LIST);
356            if (paths != null) {
357                for (String path : paths) {
358                    addToCache(conf, appPath, path, false);
359                }
360            }
361    
362            for (Element eProp : (List<Element>) actionXml.getChildren()) {
363                if (eProp.getName().equals("file")) {
364                    String path = eProp.getTextTrim();
365                    addToCache(conf, appPath, path, false);
366                }
367                else {
368                    if (eProp.getName().equals("archive")) {
369                        String path = eProp.getTextTrim();
370                        addToCache(conf, appPath, path, true);
371                    }
372                }
373            }
374        }

393        @SuppressWarnings("unchecked")
394        JobConf createLauncherConf(FileSystem actionFs, Context context, WorkflowAction action, Element actionXml, Configuration actionConf)
395                throws ActionExecutorException {
396            try {
397    
398                // app path could be a file
399                Path appPathRoot = new Path(context.getWorkflow().getAppPath());
400                if (actionFs.isFile(appPathRoot)) {
401                    appPathRoot = appPathRoot.getParent();
402                }
403    
404                // launcher job configuration
405                Configuration launcherConf = createBaseHadoopConf(context, actionXml);
406                setupLauncherConf(launcherConf, actionXml, appPathRoot, context);
407    
408                // we are doing init+copy because if not we are getting 'hdfs'
409                // scheme not known
410                // its seems that new JobConf(Conf) does not load defaults, it
411                // assumes parameter Conf does.
412                JobConf launcherJobConf = new JobConf();
413                XConfiguration.copy(launcherConf, launcherJobConf);
414                setLibFilesArchives(context, actionXml, appPathRoot, launcherJobConf);
415                String jobName = XLog.format("oozie:launcher:T={0}:W={1}:A={2}:ID={3}", getType(), context.getWorkflow()
416                        .getAppName(), action.getName(), context.getWorkflow().getId());
417                launcherJobConf.setJobName(jobName);
418    
419                String jobId = context.getWorkflow().getId();
420                String actionId = action.getId();
421                Path actionDir = context.getActionDir();
422                String recoveryId = context.getRecoveryId();
423    
424                LauncherMapper.setupLauncherInfo(launcherJobConf, jobId, actionId, actionDir, recoveryId, actionConf);
425    
426                LauncherMapper.setupMainClass(launcherJobConf, getLauncherMain(launcherConf, actionXml));
427    
428                LauncherMapper.setupMaxOutputData(launcherJobConf, maxActionOutputLen);
429    
430                Namespace ns = actionXml.getNamespace();
431                List<Element> list = actionXml.getChildren("arg", ns);
432                String[] args = new String[list.size()];
433                for (int i = 0; i < list.size(); i++) {
434                    args[i] = list.get(i).getTextTrim();
435                }
436                LauncherMapper.setupMainArguments(launcherJobConf, args);
437    
438                Element opt = actionXml.getChild("java-opts", ns);
439                if (opt != null) {
440                    String opts = launcherConf.get("mapred.child.java.opts", "");
441                    opts = opts + " " + opt.getTextTrim();
442                    opts = opts.trim();
443                    launcherJobConf.set("mapred.child.java.opts", opts);
444                }
445    
446                // properties from action that are needed by the launcher (QUEUE NAME)
447                // maybe we should add queue to the WF schema, below job-tracker
448                for (String name : SPECIAL_PROPERTIES) {
449                    String value = actionConf.get(name);
450                    if (value != null) {
451                        if (!name.equals(QUEUE_NAME) ||
452                            (name.equals(QUEUE_NAME) && launcherJobConf.get(OOZIE_LAUNCHER_QUEUE_NAME) == null)) {
453                            launcherJobConf.set(name, value);
454                        }
455                    }
456                }
457    
458                // to disable cancelation of delegation token on launcher job end
459                launcherJobConf.setBoolean("mapreduce.job.complete.cancel.delegation.tokens", false);
460    
461                // setting the group owning the Oozie job to allow anybody in that
462                // group to kill the jobs.
463                launcherJobConf.set("mapreduce.job.acl-modify-job", context.getWorkflow().getGroup());
464    
465                return launcherJobConf;
466            }
467            catch (Exception ex) {
468                throw convertException(ex);
469            }
470        }

488        public void submitLauncher(FileSystem actionFs, Context context, WorkflowAction action) throws ActionExecutorException {
489            JobClient jobClient = null;
490            boolean exception = false;
491            try {
492                Path appPathRoot = new Path(context.getWorkflow().getAppPath());
493    
494                // app path could be a file
495                if (actionFs.isFile(appPathRoot)) {
496                    appPathRoot = appPathRoot.getParent();
497                }
498    
499                Element actionXml = XmlUtils.parseXml(action.getConf());
500    
501                // action job configuration
502                Configuration actionConf = createBaseHadoopConf(context, actionXml);
503                setupActionConf(actionConf, context, actionXml, appPathRoot);
504                XLog.getLog(getClass()).debug("Setting LibFilesArchives ");
505                setLibFilesArchives(context, actionXml, appPathRoot, actionConf);
506                String jobName = XLog.format("oozie:action:T={0}:W={1}:A={2}:ID={3}", getType(), context.getWorkflow()
507                        .getAppName(), action.getName(), context.getWorkflow().getId());
508                actionConf.set("mapred.job.name", jobName);
509                injectActionCallback(context, actionConf);
510    
511                // setting the group owning the Oozie job to allow anybody in that
512                // group to kill the jobs.
513                actionConf.set("mapreduce.job.acl-modify-job", context.getWorkflow().getGroup());
514    
515                // Setting the credential properties in launcher conf
516                HashMap<String, CredentialsProperties> credentialsProperties = setCredentialPropertyToActionConf(context,
517                        action, actionConf);
518    
519                // Adding if action need to set more credential tokens
520                JobConf credentialsConf = new JobConf(false);
521                XConfiguration.copy(actionConf, credentialsConf);
522                setCredentialTokens(credentialsConf, context, action, credentialsProperties);
523    
524                // insert conf to action conf from credentialsConf
525                for (Entry<String, String> entry : credentialsConf) {
526                    if (actionConf.get(entry.getKey()) == null) {
527                        actionConf.set(entry.getKey(), entry.getValue());
528                    }
529                }
530    
531                JobConf launcherJobConf = createLauncherConf(actionFs, context, action, actionXml, actionConf);
532                injectLauncherCallback(context, launcherJobConf);
533                XLog.getLog(getClass()).debug("Creating Job Client for action " + action.getId());
534                jobClient = createJobClient(context, launcherJobConf);
535                String launcherId = LauncherMapper.getRecoveryId(launcherJobConf, context.getActionDir(), context
536                        .getRecoveryId());
537                boolean alreadyRunning = launcherId != null;
538                RunningJob runningJob;
539    
540                // if user-retry is on, always submit new launcher
541                boolean isUserRetry = ((WorkflowActionBean)action).isUserRetry();
542    
543                if (alreadyRunning && !isUserRetry) {
544                    runningJob = jobClient.getJob(JobID.forName(launcherId));
545                    if (runningJob == null) {
546                        String jobTracker = launcherJobConf.get("mapred.job.tracker");
547                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "JA017",
548                                "unknown job [{0}@{1}], cannot recover", launcherId, jobTracker);
549                    }
550                }
551                else {
552                    prepare(context, actionXml);
553                    XLog.getLog(getClass()).debug("Submitting the job through Job Client for action " + action.getId());
554    
555                    // setting up propagation of the delegation token.
556                    AuthHelper.get().set(jobClient, launcherJobConf);
557                    log.debug(WorkflowAppService.HADOOP_JT_KERBEROS_NAME + " = "
558                            + launcherJobConf.get(WorkflowAppService.HADOOP_JT_KERBEROS_NAME));
559                    log.debug(WorkflowAppService.HADOOP_NN_KERBEROS_NAME + " = "
560                            + launcherJobConf.get(WorkflowAppService.HADOOP_NN_KERBEROS_NAME));
561    
562                    // insert credentials tokens to launcher job conf if needed
563                    if (needInjectCredentials()) {
564                        for (Token<? extends TokenIdentifier> tk : credentialsConf.getCredentials().getAllTokens()) {
565                            log.debug("ADDING TOKEN: " + tk.getKind().toString());
566                            launcherJobConf.getCredentials().addToken(tk.getKind(), tk);
567                        }
568                    }
569                    else {
570                        log.info("No need to inject credentials.");
571                    }
572                    runningJob = jobClient.submitJob(launcherJobConf);
573                    if (runningJob == null) {
574                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "JA017",
575                                "Error submitting launcher for action [{0}]", action.getId());
576                    }
577                    launcherId = runningJob.getID().toString();
578                    XLog.getLog(getClass()).debug("After submission get the launcherId " + launcherId);
579                }
580    
581                String jobTracker = launcherJobConf.get(HADOOP_JOB_TRACKER);
582                String consoleUrl = runningJob.getTrackingURL();
583                context.setStartData(launcherId, jobTracker, consoleUrl);
584            }
585            catch (Exception ex) {
586                exception = true;

596                            log.error("JobClient error: ", e);
597                        }

599                            throw convertException(e);
600                        }
601                    }
602                }
603            }
604        }

624        protected HashMap<String, CredentialsProperties> setCredentialPropertyToActionConf(Context context,
625                WorkflowAction action, Configuration actionConf) throws Exception {
626            HashMap<String, CredentialsProperties> credPropertiesMap = null;
627            if (context != null && action != null) {
628                credPropertiesMap = getActionCredentialsProperties(context, action);
629                if (credPropertiesMap != null) {
630                    for (String key : credPropertiesMap.keySet()) {
631                        CredentialsProperties prop = credPropertiesMap.get(key);
632                        if (prop != null) {
633                            log.debug("Credential Properties set for action : " + action.getId());
634                            for (String property : prop.getProperties().keySet()) {
635                                actionConf.set(property, prop.getProperties().get(property));
636                                log.debug("property : '" + property + "', value : '" + prop.getProperties().get(property) + "'");
637                            }
638                        }
639                    }
640                }
641                else {
642                    log.warn("No credential properties found for action : " + action.getId() + ", cred : " + action.getCred());
643                }
644            }
645            else {
646                log.warn("context or action is null");
647            }
648            return credPropertiesMap;
649        }

651        protected void setCredentialTokens(JobConf jobconf, Context context, WorkflowAction action,
652                HashMap<String, CredentialsProperties> credPropertiesMap) throws Exception {
653    
654            if (context != null && action != null && credPropertiesMap != null) {
655                for (Entry<String, CredentialsProperties> entry : credPropertiesMap.entrySet()) {
656                    String credName = entry.getKey();
657                    CredentialsProperties credProps = entry.getValue();
658                    if (credProps != null) {
659                        CredentialsProvider credProvider = new CredentialsProvider(credProps.getType());
660                        Credentials credentialObject = credProvider.createCredentialObject();
661                        if (credentialObject != null) {
662                            credentialObject.addtoJobConf(jobconf, credProps, context);
663                            log.debug("Retrieved Credential '" + credName + "' for action " + action.getId());
664                        }
665                        else {
666                            log.debug("Credentials object is null for name= " + credName + ", type=" + credProps.getType());
667                        }
668                    }
669                    else {
670                        log.warn("Could not find credentials properties for: " + credName);
671                    }
672                }
673            }

677        protected HashMap<String, CredentialsProperties> getActionCredentialsProperties(Context context,
678                WorkflowAction action) throws Exception {
679            HashMap<String, CredentialsProperties> props = new HashMap<String, CredentialsProperties>();
680            if (context != null && action != null) {
681                String credsInAction = action.getCred();
682                log.debug("Get credential '" + credsInAction + "' properties for action : " + action.getId());
683                String[] credNames = credsInAction.split(",");
684                for (String credName : credNames) {
685                    CredentialsProperties credProps = getCredProperties(context, credName);
686                    props.put(credName, credProps);
687                }
688            }
689            else {
690                log.warn("context or action is null");
691            }
692            return props;
693        }

695        @SuppressWarnings("unchecked")
696        protected CredentialsProperties getCredProperties(Context context, String credName)
697                throws Exception {
698            CredentialsProperties credProp = null;
699            String workflowXml = ((WorkflowJobBean) context.getWorkflow()).getWorkflowInstance().getApp().getDefinition();
700            Element elementJob = XmlUtils.parseXml(workflowXml);
701            Element credentials = elementJob.getChild("credentials", elementJob.getNamespace());
702            if (credentials != null) {
703                for (Element credential : (List<Element>) credentials.getChildren("credential", credentials
704                        .getNamespace())) {
705                    String name = credential.getAttributeValue("name");
706                    String type = credential.getAttributeValue("type");
707                    log.debug("getCredProperties: Name: " + name + ", Type: " + type);
708                    if (name.equalsIgnoreCase(credName)) {
709                        credProp = new CredentialsProperties(name, type);
710                        for (Element property : (List<Element>) credential.getChildren("property", credential
711                                .getNamespace())) {
712                            credProp.getProperties().put(property.getChildText("name", property.getNamespace()),
713                                    property.getChildText("value", property.getNamespace()));
714                            log.debug("getCredProperties: Properties name :'"
715                                    + property.getChildText("name", property.getNamespace()) + "', Value : '"
716                                    + property.getChildText("value", property.getNamespace()) + "'");
717                        }
718                    }
719                }
720            }
721            else {
722                log.warn("credentials is null for the action");
723            }
724            return credProp;
725        }

739        public void start(Context context, WorkflowAction action) throws ActionExecutorException {
740            try {
741                XLog.getLog(getClass()).debug("Starting action " + action.getId() + " getting Action File System");
742                FileSystem actionFs = getActionFileSystem(context, action);
743                XLog.getLog(getClass()).debug("Preparing action Dir through copying " + context.getActionDir());
744                prepareActionDir(actionFs, context);
745                XLog.getLog(getClass()).debug("Action Dir is ready. Submitting the action ");
746                submitLauncher(actionFs, context, action);
747                XLog.getLog(getClass()).debug("Action submit completed. Performing check ");
748                check(context, action);
749                XLog.getLog(getClass()).debug("Action check is done after submission");
750            }
751            catch (Exception ex) {
752                throw convertException(ex);
753            }

758            try {
759                String externalStatus = action.getExternalStatus();
760                WorkflowAction.Status status = externalStatus.equals(SUCCEEDED) ? WorkflowAction.Status.OK

770                    cleanUpActionDir(actionFs, context);

772                catch (Exception ex) {
773                    throw convertException(ex);
774                }
775            }

792        @Override
793        public void check(Context context, WorkflowAction action) throws ActionExecutorException {
794            JobClient jobClient = null;
795            boolean exception = false;
796            try {
797                Element actionXml = XmlUtils.parseXml(action.getConf());
798                FileSystem actionFs = getActionFileSystem(context, actionXml);
799                Configuration conf = createBaseHadoopConf(context, actionXml);
800                JobConf jobConf = new JobConf();
801                XConfiguration.copy(conf, jobConf);
802                jobClient = createJobClient(context, jobConf);
803                RunningJob runningJob = jobClient.getJob(JobID.forName(action.getExternalId()));
804                if (runningJob == null) {
805                    context.setExternalStatus(FAILED);
806                    context.setExecutionData(FAILED, null);
807                    throw new ActionExecutorException(ActionExecutorException.ErrorType.FAILED, "JA017",
808                            "Unknown hadoop job [{0}] associated with action [{1}].  Failing this action!", action
809                                    .getExternalId(), action.getId());
810                }
811                if (runningJob.isComplete()) {
812                    Path actionDir = context.getActionDir();
813    
814                    String user = context.getWorkflow().getUser();
815                    String group = context.getWorkflow().getGroup();
816                    if (LauncherMapper.hasIdSwap(runningJob, user, group, actionDir)) {
817                        String launcherId = action.getExternalId();
818                        Path idSwapPath = LauncherMapper.getIdSwapPath(context.getActionDir());
819                        InputStream is = actionFs.open(idSwapPath);
820                        BufferedReader reader = new BufferedReader(new InputStreamReader(is));
821                        Properties props = PropertiesUtils.readProperties(reader, maxActionOutputLen);
822                        reader.close();
823                        String newId = props.getProperty("id");
824                        runningJob = jobClient.getJob(JobID.forName(newId));
825                        if (runningJob == null) {
826                            context.setExternalStatus(FAILED);
827                            throw new ActionExecutorException(ActionExecutorException.ErrorType.FAILED, "JA017",
828                                    "Unknown hadoop job [{0}] associated with action [{1}].  Failing this action!", newId,
829                                    action.getId());
830                        }
831    
832                        context.setStartData(newId, action.getTrackerUri(), runningJob.getTrackingURL());
833                        XLog.getLog(getClass()).info(XLog.STD, "External ID swap, old ID [{0}] new ID [{1}]", launcherId,
834                                newId);
835                    }
836                    if (runningJob.isComplete()) {
837                        XLog.getLog(getClass()).info(XLog.STD, "action completed, external ID [{0}]",
838                                action.getExternalId());
839                        if (runningJob.isSuccessful() && LauncherMapper.isMainSuccessful(runningJob)) {
840                            Properties props = null;
841                            if (getCaptureOutput(action)) {
842                                props = new Properties();
843                                if (LauncherMapper.hasOutputData(runningJob)) {
844                                    Path actionOutput = LauncherMapper.getOutputDataPath(context.getActionDir());
845                                    InputStream is = actionFs.open(actionOutput);
846                                    BufferedReader reader = new BufferedReader(new InputStreamReader(is));
847                                    props = PropertiesUtils.readProperties(reader, maxActionOutputLen);
848                                    reader.close();
849                                }
850                            }
851                            context.setExecutionData(SUCCEEDED, props);
852                            XLog.getLog(getClass()).info(XLog.STD, "action produced output");
853                        }
854                        else {
855                            XLog log = XLog.getLog(getClass());
856                            String errorReason;
857                            Path actionError = LauncherMapper.getErrorPath(context.getActionDir());
858                            if (actionFs.exists(actionError)) {
859                                InputStream is = actionFs.open(actionError);
860                                BufferedReader reader = new BufferedReader(new InputStreamReader(is));
861                                Properties props = PropertiesUtils.readProperties(reader, -1);
862                                reader.close();
863                                String errorCode = props.getProperty("error.code");
864                                if (errorCode.equals("0")) {
865                                    errorCode = "JA018";
866                                }
867                                errorReason = props.getProperty("error.reason");
868                                log.warn("Launcher ERROR, reason: {0}", errorReason);
869                                String exMsg = props.getProperty("exception.message");
870                                String errorInfo = (exMsg != null) ? exMsg : errorReason;
871                                context.setErrorInfo(errorCode, errorInfo);
872                                String exStackTrace = props.getProperty("exception.stacktrace");
873                                if (exMsg != null) {
874                                    log.warn("Launcher exception: {0}{E}{1}", exMsg, exStackTrace);
875                                }
876                            }
877                            else {
878                                errorReason = XLog.format("LauncherMapper died, check Hadoop log for job [{0}:{1}]", action
879                                        .getTrackerUri(), action.getExternalId());
880                                log.warn(errorReason);
881                            }
882                            context.setExecutionData(FAILED_KILLED, null);
883                        }
884                    }
885                    else {
886                        context.setExternalStatus(RUNNING);
887                        XLog.getLog(getClass()).info(XLog.STD, "checking action, external ID [{0}] status [{1}]",
888                                action.getExternalId(), action.getExternalStatus());
889                    }
890                }
891                else {
892                    context.setExternalStatus(RUNNING);
893                    XLog.getLog(getClass()).info(XLog.STD, "checking action, external ID [{0}] status [{1}]",
894                            action.getExternalId(), action.getExternalStatus());
895                }
896            }
897            catch (Exception ex) {
898                XLog.getLog(getClass()).warn("Exception in check(). Message[{0}]", ex.getMessage(), ex);
899                exception = true;
900                throw convertException(ex);
901            }
902            finally {
903                if (jobClient != null) {
904                    try {
905                        jobClient.close();
906                    }
907                    catch (Exception e) {
908                        if (exception) {
909                            log.error("JobClient error: ", e);
910                        }
911                        else {
912                            throw convertException(e);
913                        }
914                    }
915                }
916            }
917        }

927        public void kill(Context context, WorkflowAction action) throws ActionExecutorException {
928            JobClient jobClient = null;
929            boolean exception = false;
930            try {
931                Element actionXml = XmlUtils.parseXml(action.getConf());
932                Configuration conf = createBaseHadoopConf(context, actionXml);
933                JobConf jobConf = new JobConf();
934                XConfiguration.copy(conf, jobConf);
935                jobClient = createJobClient(context, jobConf);
936                RunningJob runningJob = jobClient.getJob(JobID.forName(action.getExternalId()));
937                if (runningJob != null) {
938                    runningJob.killJob();
939                }
940                context.setExternalStatus(KILLED);
941                context.setExecutionData(KILLED, null);
942            }
943            catch (Exception ex) {
944                exception = true;
945                throw convertException(ex);
946            }
947            finally {
948                try {
949                    FileSystem actionFs = getActionFileSystem(context, action);
950                    cleanUpActionDir(actionFs, context);
951                    if (jobClient != null) {
952                        jobClient.close();
953                    }
954                }
955                catch (Exception ex) {
956                    if (exception) {
957                        log.error("Error: ", ex);
958                    }
959                    else {
960                        throw convertException(ex);
961                    }
962                }
963            }
