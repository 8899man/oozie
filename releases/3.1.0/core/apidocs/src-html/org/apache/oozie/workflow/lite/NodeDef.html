015    package org.apache.oozie.workflow.lite;
016    
017    import org.apache.hadoop.io.Writable;
018    import org.apache.oozie.service.LiteWorkflowStoreService;
019    import org.apache.oozie.util.ParamChecker;
020    import org.apache.oozie.workflow.WorkflowException;
021    
022    import java.io.DataInput;
023    import java.io.DataOutput;
024    import java.io.IOException;
025    import java.util.ArrayList;
026    import java.util.Collections;
027    import java.util.List;
028    
029    /**
030     * This node definition is serialized object and should provide readFields() and write() for read and write of fields in
031     * this class.
032     */
033    public class NodeDef implements Writable {
034        private String nodeDefVersion = null;
035        private String name = null;
036        private Class<? extends NodeHandler> handlerClass;
037        private String conf = null;
038        private List<String> transitions = new ArrayList<String>();
039        private String cred = "null";
040        private String userRetryMax = "null";
041        private String userRetryInterval = "null";
042    
043        NodeDef() {
044        }
045    
046        NodeDef(String name, String conf, Class<? extends NodeHandler> handlerClass, List<String> transitions) {
047            this.name = ParamChecker.notEmpty(name, "name");
048            this.conf = conf;
049            this.handlerClass = ParamChecker.notNull(handlerClass, "handlerClass");
050            this.transitions = Collections.unmodifiableList(ParamChecker.notEmptyElements(transitions, "transitions"));
051        }
052    
053        NodeDef(String name, String conf, Class<? extends NodeHandler> handlerClass, List<String> transitions, String cred) {
054            this(name, conf, handlerClass, transitions);
055            if (cred != null) {
056                this.cred = cred;
057            }
058        }
059    
060        NodeDef(String name, String conf, Class<? extends NodeHandler> handlerClass, List<String> transitions, String cred,
061                String userRetryMax, String userRetryInterval) {
062            this(name, conf, handlerClass, transitions, cred);
063            if (userRetryMax != null) {
064                this.userRetryMax = userRetryMax;
065            }
066            if (userRetryInterval != null) {
067                this.userRetryInterval = userRetryInterval;
068            }
069        }
070    
071        public boolean equals(NodeDef other) {
072            return !(other == null || getClass() != other.getClass() || !getName().equals(other.getName()));
073        }
074    
075        @Override
076        public int hashCode() {
077            return name.hashCode();
078        }
079    
080        public String getName() {
081            return name;
082        }
083    
084        public String getCred() {
085            return cred;
086        }
087    
088        public Class<? extends NodeHandler> getHandlerClass() {
089            return handlerClass;
090        }
091    
092        public List<String> getTransitions() {
093            return transitions;
094        }
095    
096        public String getConf() {
097            return conf;
098        }
099    
100        public String getUserRetryMax() {
101            return userRetryMax;
102        }
103    
104        public String getUserRetryInterval() {
105            return userRetryInterval;
106        }
107    
108        public String getNodeDefVersion() {
109            if (nodeDefVersion == null) {
110                try {
111                    nodeDefVersion = LiteWorkflowStoreService.getNodeDefDefaultVersion();
112                }
113                catch (WorkflowException e) {
114                    nodeDefVersion = LiteWorkflowStoreService.NODE_DEF_VERSION_1;
115                }
116            }
117            return nodeDefVersion;
118        }
119    
120        @SuppressWarnings("unchecked")
121        private void readVersionZero(DataInput dataInput, String firstField) throws IOException {
122            if (firstField.equals(LiteWorkflowStoreService.NODE_DEF_VERSION_0)) {
123                name = dataInput.readUTF();
124            } else {
125                name = firstField;
126            }
127            nodeDefVersion = LiteWorkflowStoreService.NODE_DEF_VERSION_0;
128            cred = dataInput.readUTF();
129            String handlerClassName = dataInput.readUTF();
130            if ((handlerClassName != null) && (handlerClassName.length() > 0)) {
131                try {
132                    handlerClass = (Class<? extends NodeHandler>) Class.forName(handlerClassName);
133                }
134                catch (ClassNotFoundException ex) {
135                    throw new IOException(ex);
136                }
137            }
138            conf = dataInput.readUTF();
139            if (conf.equals("null")) {
140                conf = null;
141            }
142            int numTrans = dataInput.readInt();
143            transitions = new ArrayList<String>(numTrans);
144            for (int i = 0; i < numTrans; i++) {
145                transitions.add(dataInput.readUTF());
146            }
147        }
148        @SuppressWarnings("unchecked")
149        private void readVersionOne(DataInput dataInput, String firstField) throws IOException {
150            nodeDefVersion = LiteWorkflowStoreService.NODE_DEF_VERSION_1;
151            name = dataInput.readUTF();
152            cred = dataInput.readUTF();
153            String handlerClassName = dataInput.readUTF();
154            if ((handlerClassName != null) && (handlerClassName.length() > 0)) {
155                try {
156                    handlerClass = (Class<? extends NodeHandler>) Class.forName(handlerClassName);
157                }
158                catch (ClassNotFoundException ex) {
159                    throw new IOException(ex);
160                }
161            }
162            conf = dataInput.readUTF();
163            if (conf.equals("null")) {
164                conf = null;
165            }
166            int numTrans = dataInput.readInt();
167            transitions = new ArrayList<String>(numTrans);
168            for (int i = 0; i < numTrans; i++) {
169                transitions.add(dataInput.readUTF());
170            }
171            userRetryMax = dataInput.readUTF();
172            userRetryInterval = dataInput.readUTF();
173        }
174    
175        /* (non-Javadoc)
176         * @see org.apache.hadoop.io.Writable#readFields(java.io.DataInput)
177         */
178        @Override
179        public void readFields(DataInput dataInput) throws IOException {
180            String firstField = dataInput.readUTF();
181            if (!firstField.equals(LiteWorkflowStoreService.NODE_DEF_VERSION_1)) {
182                readVersionZero(dataInput, firstField);
183            } else {
184                //since oozie version 3.1
185                readVersionOne(dataInput, firstField);
186            }
187        }
188    
189        private void writeVersionZero(DataOutput dataOutput) throws IOException {
190            dataOutput.writeUTF(nodeDefVersion);
191            dataOutput.writeUTF(name);
192            if (cred != null) {
193                dataOutput.writeUTF(cred);
194            }
195            else {
196                dataOutput.writeUTF("null");
197            }
198            dataOutput.writeUTF(handlerClass.getName());
199            if (conf != null) {
200                dataOutput.writeUTF(conf);
201            }
202            else {
203                dataOutput.writeUTF("null");
204            }
205            dataOutput.writeInt(transitions.size());
206            for (String transition : transitions) {
207                dataOutput.writeUTF(transition);
208            }
209        }
210    
211        /**
212         * Write as version one format, this version was since 3.1.
213         *
214         * @param dataOutput data output to serialize node def
215         * @throws IOException thrown if fail to write
216         */
217        private void writeVersionOne(DataOutput dataOutput) throws IOException {
218            dataOutput.writeUTF(nodeDefVersion);
219            dataOutput.writeUTF(name);
220            if (cred != null) {
221                dataOutput.writeUTF(cred);
222            }
223            else {
224                dataOutput.writeUTF("null");
225            }
226            dataOutput.writeUTF(handlerClass.getName());
227            if (conf != null) {
228                dataOutput.writeUTF(conf);
229            }
230            else {
231                dataOutput.writeUTF("null");
232            }
233            dataOutput.writeInt(transitions.size());
234            for (String transition : transitions) {
235                dataOutput.writeUTF(transition);
236            }
237            if (userRetryMax != null) {
238                dataOutput.writeUTF(userRetryMax);
239            }
240            else {
241                dataOutput.writeUTF("null");
242            }
243            if (userRetryInterval != null) {
244                dataOutput.writeUTF(userRetryInterval);
245            }
246            else {
247                dataOutput.writeUTF("null");
248            }
249        }
250    
251        /* (non-Javadoc)
252         * @see org.apache.hadoop.io.Writable#write(java.io.DataOutput)
253         */
254        @Override
255        public void write(DataOutput dataOutput) throws IOException {
256            if (!getNodeDefVersion().equals(LiteWorkflowStoreService.NODE_DEF_VERSION_1)) {
257                writeVersionZero(dataOutput);
258            } else {
259                //since oozie version 3.1
260                writeVersionOne(dataOutput);
261            }
262        }
