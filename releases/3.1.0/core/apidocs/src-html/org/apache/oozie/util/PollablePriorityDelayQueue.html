015    package org.apache.oozie.util;
016    
017    import java.util.Iterator;
018    import java.util.concurrent.TimeUnit;
019    
020    public class PollablePriorityDelayQueue<E> extends PriorityDelayQueue<E> {
021    
022        public PollablePriorityDelayQueue(int priorities, long maxWait, TimeUnit unit, int maxSize) {
023            super(priorities, maxWait, unit, maxSize);
024        }
025    
026        /**
027         * Retrieve and remove the head of this queue if it is eligible to poll. If not, iterate next element until find
028         * the one which is eligible to poll from queue.
029         *
030         * Return <tt>null</tt> if this queue has no elements eligible to run.
031         *
032         * <p/>
033         * Invocations to this method run the anti-starvation (once every interval check).
034         *
035         * @return the element of this queue, for which eligibleToPoll is true.
036         */
037        @Override
038        public QueueElement<E> poll() {
039            try {
040                lock.lock();
041                antiStarvation();
042                QueueElement<E> e = null;
043                int i = priorities;
044                for (; e == null && i > 0; i--) {
045                    e = queues[i - 1].peek();
046                    if (eligibleToPoll(e)) {
047                        e = queues[i - 1].poll();
048                    }
049                    else {
050                        if (e != null) {
051                            debug("poll(): the peek element [{0}], from P[{1}] is not eligible to poll", e.getElement().toString(), i);
052                        }
053                        e = null;
054                        Iterator<QueueElement<E>> iter = queues[i - 1].iterator();
055                        while(e == null && iter.hasNext()) {
056                            e = iter.next();
057                            if (eligibleToPoll(e)) {
058                                queues[i - 1].remove(e);
059                            }
060                            else {
061                                debug("poll(): the iterator element [{0}], from P[{1}] is not eligible to poll", e.getElement().toString(), i);
062                                e = null;
063                            }
064                        }
065                    }
066                }
067                if (e != null) {
068                    if (currentSize != null) {
069                        currentSize.decrementAndGet();
070                    }
071                    e.inQueue = false;
072                    debug("poll(): [{0}], from P[{1}]", e.getElement().toString(), i);
073                }
074                return e;
075            }
076            finally {
077                lock.unlock();
078            }
079        }
080    
081        /**
082         * Method for checking the QueueElement eligible to poll before remove it from queue.
083         * <p/>
084         * This method should be overriden for checking purposes.
085         *
086         * @param element the element to check
087         */
088        protected boolean eligibleToPoll(QueueElement<?> element) {
089            return true;
090        }
091    }
