052    /**
053     * Base class for Action execution commands. Provides common functionality to handle different types of errors while
054     * attempting to start or end an action.
055     */
056    public abstract class ActionXCommand<T> extends WorkflowXCommand<Void> {
057        private static final String INSTRUMENTATION_GROUP = "action.executors";
058    
059        protected static final String INSTR_FAILED_JOBS_COUNTER = "failed";
060    
061        protected static final String RECOVERY_ID_SEPARATOR = "@";
062    
063        public ActionXCommand(String name, String type, int priority) {
064            super(name, type, priority);
065        }
066    
067        /**
068         * Takes care of Transient failures. Sets the action status to retry and increments the retry count if not enough
069         * attempts have been made. Otherwise returns false.
070         *
071         * @param context the execution context.
072         * @param executor the executor instance being used.
073         * @param status the status to be set for the action.
074         * @return true if the action is scheduled for another retry. false if the number of retries has exceeded the
075         *         maximum number of configured retries.
076         * @throws CommandException thrown if unable to handle transient
077         */
078        protected boolean handleTransient(ActionExecutor.Context context, ActionExecutor executor,
079                WorkflowAction.Status status) throws CommandException {
080            LOG.debug("Attempting to retry");
081            ActionExecutorContext aContext = (ActionExecutorContext) context;
082            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
083            incrActionErrorCounter(action.getType(), "transient", 1);
084    
085            int actionRetryCount = action.getRetries();
086            if (actionRetryCount >= executor.getMaxRetries()) {
087                LOG.warn("Exceeded max retry count [{0}]. Suspending Job", executor.getMaxRetries());
088                return false;
089            }
090            else {
091                action.setStatus(status);
092                action.setPending();
093                action.incRetries();
094                long retryDelayMillis = executor.getRetryInterval() * 1000;
095                action.setPendingAge(new Date(System.currentTimeMillis() + retryDelayMillis));
096                LOG.info("Next Retry, Attempt Number [{0}] in [{1}] milliseconds", actionRetryCount + 1, retryDelayMillis);
097                this.resetUsed();
098                queue(this, retryDelayMillis);
099                return true;
100            }
101        }
102    
103        /**
104         * Takes care of non transient failures. The job is suspended, and the state of the action is changed to *MANUAL and
105         * set pending flag of action to false
106         *
107         * @param context the execution context.
108         * @param executor the executor instance being used.
109         * @param status the status to be set for the action.
110         * @throws CommandException thrown if unable to suspend job
111         */
112        protected void handleNonTransient(ActionExecutor.Context context, ActionExecutor executor,
113                WorkflowAction.Status status) throws CommandException {
114            ActionExecutorContext aContext = (ActionExecutorContext) context;
115            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
116            incrActionErrorCounter(action.getType(), "nontransient", 1);
117            WorkflowJobBean workflow = (WorkflowJobBean) context.getWorkflow();
118            String id = workflow.getId();
119            action.setStatus(status);
120            action.resetPendingOnly();
121            LOG.warn("Suspending Workflow Job id=" + id);
122            try {
123                SuspendXCommand.suspendJob(Services.get().get(JPAService.class), workflow, id, action.getId());
124            }
125            catch (Exception e) {
126                throw new CommandException(ErrorCode.E0727, e.getMessage());
127            }
128            finally {
129                // update coordinator action
130                new CoordActionUpdateXCommand(workflow, 3).call();
131            }
132        }
133    
134        /**
135         * Takes care of errors. </p> For errors while attempting to start the action, the job state is updated and an
136         * {@link ActionEndCommand} is queued. </p> For errors while attempting to end the action, the job state is updated.
137         * </p>
138         *
139         * @param context the execution context.
140         * @param executor the executor instance being used.
141         * @param message
142         * @param isStart whether the error was generated while starting or ending an action.
143         * @param status the status to be set for the action.
144         * @throws CommandException thrown if unable to handle action error
145         */
146        protected void handleError(ActionExecutor.Context context, ActionExecutor executor, String message,
147                boolean isStart, WorkflowAction.Status status) throws CommandException {
148            LOG.warn("Setting Action Status to [{0}]", status);
149            ActionExecutorContext aContext = (ActionExecutorContext) context;
150            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
151    
152            if (!handleUserRetry(action)) {
153                incrActionErrorCounter(action.getType(), "error", 1);
154                action.setPending();
155                if (isStart) {
156                    action.setExecutionData(message, null);
157                    queue(new ActionEndXCommand(action.getId(), action.getType()));
158                }
159                else {
160                    action.setEndData(status, WorkflowAction.Status.ERROR.toString());
161                }
162            }
163        }
164    
165        /**
166         * Fail the job due to failed action
167         *
168         * @param context the execution context.
169         * @throws CommandException thrown if unable to fail job
170         */
171        public void failJob(ActionExecutor.Context context) throws CommandException {
172            ActionExecutorContext aContext = (ActionExecutorContext) context;
173            WorkflowActionBean action = (WorkflowActionBean) aContext.getAction();
174            WorkflowJobBean workflow = (WorkflowJobBean) context.getWorkflow();
175    
176            if (!handleUserRetry(action)) {
177                incrActionErrorCounter(action.getType(), "failed", 1);
178                LOG.warn("Failing Job due to failed action [{0}]", action.getName());
179                try {
180                    workflow.getWorkflowInstance().fail(action.getName());
181                    WorkflowInstance wfInstance = workflow.getWorkflowInstance();
182                    ((LiteWorkflowInstance) wfInstance).setStatus(WorkflowInstance.Status.FAILED);
183                    workflow.setWorkflowInstance(wfInstance);
184                    workflow.setStatus(WorkflowJob.Status.FAILED);
185                    action.setStatus(WorkflowAction.Status.FAILED);
186                    action.resetPending();
187                    queue(new NotificationXCommand(workflow, action));
188                    queue(new KillXCommand(workflow.getId()));
189                    InstrumentUtils.incrJobCounter(INSTR_FAILED_JOBS_COUNTER, 1, getInstrumentation());
190                }
191                catch (WorkflowException ex) {
192                    throw new CommandException(ex);
193                }
194            }
195        }
196    
197        /**
198         * Execute retry for action if this action is eligible for user-retry
199         *
200         * @param context the execution context.
201         * @return true if user-retry has to be handled for this action
202         * @throws CommandException thrown if unable to fail job
203         */
204        public boolean handleUserRetry(WorkflowActionBean action) throws CommandException {
205            String errorCode = action.getErrorCode();
206            Set<String> allowedRetryCode = LiteWorkflowStoreService.getUserRetryErrorCode();
207    
208            if (allowedRetryCode.contains(errorCode) && action.getUserRetryCount() < action.getUserRetryMax()) {
209                LOG.info("Preparing retry this action [{0}], errorCode [{1}], userRetryCount [{2}], "
210                        + "userRetryMax [{3}], userRetryInterval [{4}]", action.getId(), errorCode, action
211                        .getUserRetryCount(), action.getUserRetryMax(), action.getUserRetryInterval());
212                int interval = action.getUserRetryInterval() * 60 * 1000;
213                action.setStatus(WorkflowAction.Status.USER_RETRY);
214                action.incrmentUserRetryCount();
215                action.setPending();
216                queue(new ActionStartXCommand(action.getId(), action.getType()), interval);
217                return true;
218            }
219            return false;
220        }
221    
222        private void incrActionErrorCounter(String type, String error, int count) {
223            getInstrumentation().incr(INSTRUMENTATION_GROUP, type + "#ex." + error, count);
224        }
225    
226        protected void incrActionCounter(String type, int count) {
227            getInstrumentation().incr(INSTRUMENTATION_GROUP, type + "#" + getName(), count);
228        }
229    
230        protected void addActionCron(String type, Instrumentation.Cron cron) {
231            getInstrumentation().addCron(INSTRUMENTATION_GROUP, type + "#" + getName(), cron);
232        }

234        /**
235         * Workflow action executor context
236         *
237         */
238        public static class ActionExecutorContext implements ActionExecutor.Context {
239            private final WorkflowJobBean workflow;
240            private Configuration protoConf;
241            private final WorkflowActionBean action;
242            private final boolean isRetry;
243            private final boolean isUserRetry;
244            private boolean started;
245            private boolean ended;
246            private boolean executed;
247    
248            public ActionExecutorContext(WorkflowJobBean workflow, WorkflowActionBean action, boolean isRetry, boolean isUserRetry) {
249                this.workflow = workflow;
250                this.action = action;
251                this.isRetry = isRetry;
252                this.isUserRetry = isUserRetry;
253                try {
254                    protoConf = new XConfiguration(new StringReader(workflow.getProtoActionConf()));
255                }
256                catch (IOException ex) {
257                    throw new RuntimeException("It should not happen", ex);
258                }
259            }
260    
261            public String getCallbackUrl(String externalStatusVar) {
262                return Services.get().get(CallbackService.class).createCallBackUrl(action.getId(), externalStatusVar);
263            }
264    
265            public Configuration getProtoActionConf() {
266                return protoConf;
267            }
268    
269            public WorkflowJob getWorkflow() {
270                return workflow;
271            }
272    
273            public WorkflowAction getAction() {
274                return action;
275            }
276    
277            public ELEvaluator getELEvaluator() {
278                ELEvaluator evaluator = Services.get().get(ELService.class).createEvaluator("workflow");
279                DagELFunctions.configureEvaluator(evaluator, workflow, action);
280                return evaluator;
281            }
282    
283            public void setVar(String name, String value) {
284                name = action.getName() + WorkflowInstance.NODE_VAR_SEPARATOR + name;
285                WorkflowInstance wfInstance = workflow.getWorkflowInstance();
286                wfInstance.setVar(name, value);
287                workflow.setWorkflowInstance(wfInstance);
288            }
289    
290            public String getVar(String name) {
291                name = action.getName() + WorkflowInstance.NODE_VAR_SEPARATOR + name;
292                return workflow.getWorkflowInstance().getVar(name);
293            }
294    
295            public void setStartData(String externalId, String trackerUri, String consoleUrl) {
296                action.setStartData(externalId, trackerUri, consoleUrl);
297                started = true;
298            }
299    
300            public void setExecutionData(String externalStatus, Properties actionData) {
301                action.setExecutionData(externalStatus, actionData);
302                executed = true;
303            }
304    
305            public void setEndData(WorkflowAction.Status status, String signalValue) {
306                action.setEndData(status, signalValue);
307                ended = true;
308            }
309    
310            public boolean isRetry() {
311                return isRetry;
312            }
313    
314            public boolean isUserRetry() {
315                return isUserRetry;
316            }
317    
318            /**
319             * Returns whether setStartData has been called or not.
320             *
321             * @return true if start completion info has been set.
322             */
323            public boolean isStarted() {
324                return started;
325            }
326    
327            /**
328             * Returns whether setExecutionData has been called or not.
329             *
330             * @return true if execution completion info has been set, otherwise false.
331             */
332            public boolean isExecuted() {
333                return executed;
334            }
335    
336            /**
337             * Returns whether setEndData has been called or not.
338             *
339             * @return true if end completion info has been set.
340             */
341            public boolean isEnded() {
342                return ended;
343            }
344    
345            public void setExternalStatus(String externalStatus) {
346                action.setExternalStatus(externalStatus);
347            }
348    
349            @Override
350            public String getRecoveryId() {
351                return action.getId() + RECOVERY_ID_SEPARATOR + workflow.getRun();
352            }
353    
354            /* (non-Javadoc)
355             * @see org.apache.oozie.action.ActionExecutor.Context#getActionDir()
356             */
357            public Path getActionDir() throws HadoopAccessorException, IOException, URISyntaxException {
358                String name = getWorkflow().getId() + "/" + action.getName() + "--" + action.getType();
359                FileSystem fs = getAppFileSystem();
360                String actionDirPath = Services.get().getSystemId() + "/" + name;
361                Path fqActionDir = new Path(fs.getHomeDirectory(), actionDirPath);
362                return fqActionDir;
363            }
364    
365            /* (non-Javadoc)
366             * @see org.apache.oozie.action.ActionExecutor.Context#getAppFileSystem()
367             */
368            public FileSystem getAppFileSystem() throws HadoopAccessorException, IOException, URISyntaxException {
369                WorkflowJob workflow = getWorkflow();
370                XConfiguration jobConf = new XConfiguration(new StringReader(workflow.getConf()));
371                Configuration fsConf = new Configuration();
372                XConfiguration.copy(jobConf, fsConf);
373                return Services.get().get(HadoopAccessorService.class).createFileSystem(workflow.getUser(),
374                        workflow.getGroup(), new URI(getWorkflow().getAppPath()), fsConf);
375    
376            }
377    
378            /* (non-Javadoc)
379             * @see org.apache.oozie.action.ActionExecutor.Context#setErrorInfo(java.lang.String, java.lang.String)
380             */
381            @Override
382            public void setErrorInfo(String str, String exMsg) {
383                action.setErrorInfo(str, exMsg);
384            }
385        }
