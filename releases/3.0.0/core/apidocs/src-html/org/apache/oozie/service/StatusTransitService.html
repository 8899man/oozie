053    /**
054     * StateTransitService is scheduled to run at the configured interval.
055     * <p/>
056     * It is to update job's status according to its child actions' status. If all child actions' pending flag equals 0 (job
057     * done), we reset the job's pending flag to 0. If all child actions are succeeded, we set the job's status to
058     * SUCCEEDED.
059     */
060    public class StatusTransitService implements Service {
061        public static final String CONF_PREFIX = Service.CONF_PREFIX + "StatusTransitService.";
062        public static final String CONF_STATUSTRANSIT_INTERVAL = CONF_PREFIX + "statusTransit.interval";
063        private static int limit = -1;
064        private static Date lastInstanceStartTime = null;
065        private final static XLog LOG = XLog.getLog(StatusTransitRunnable.class);
066    
067        /**
068         * StateTransitRunnable is the runnable which is scheduled to run at the configured interval.
069         * <p/>
070         * It is to update job's status according to its child actions' status. If all child actions' pending flag equals 0
071         * (job done), we reset the job's pending flag to 0. If all child actions are succeeded, we set the job's status to
072         * SUCCEEDED.
073         */
074        static class StatusTransitRunnable implements Runnable {
075            private JPAService jpaService = null;
076            private MemoryLocks.LockToken lock;
077    
078            public StatusTransitRunnable() {
079                jpaService = Services.get().get(JPAService.class);
080                if (jpaService == null) {
081                    LOG.error("Missing JPAService");
082                }
083            }

085            public void run() {
086                try {
087                    Date curDate = new Date(); // records the start time of this service run;
088    
089                    // first check if there is some other instance running;
090                    lock = Services.get().get(MemoryLocksService.class).getWriteLock(StatusTransitService.class.getName(),
091                            lockTimeout);
092                    if (lock == null) {
093                        LOG.info("This StatusTransitService instance"
094                                + " will not run since there is already an instance running");
095                    }
096                    else {
097                        LOG.info("Acquired lock for [{0}]", StatusTransitService.class.getName());
098                        // running coord jobs transit service
099                        coordTransit();
100                        // running bundle jobs transit service
101                        bundleTransit();
102    
103                        lastInstanceStartTime = curDate;
104                    }
105                }
106                catch (Exception ex) {
107                    LOG.warn("Exception happened during bundle job status transition", ex);
108                }
109                finally {
110                    // release lock;
111                    if (lock != null) {
112                        lock.release();
113                        LOG.info("Released lock for [{0}]", StatusTransitService.class.getName());
114                    }
115                }
116            }

118            /**
119             * Aggregate bundle actions' status to bundle jobs
120             *
121             * @throws JPAExecutorException thrown if failed in db updates or retrievals
122             * @throws CommandException thrown if failed to run commands
123             */
124            private void bundleTransit() throws JPAExecutorException, CommandException {
125                List<BundleJobBean> pendingJobCheckList = null;
126                List<BundleJobBean> runningJobCheckList = null;
127                List<List<BundleJobBean>> bundleLists = new ArrayList<List<BundleJobBean>>();
128                if (lastInstanceStartTime == null) {
129                    LOG.info("Running bundle status service first instance");
130                    // this is the first instance, we need to check for all pending jobs;
131                    pendingJobCheckList = jpaService.execute(new BundleJobsGetPendingJPAExecutor(limit));
132                    runningJobCheckList = jpaService.execute(new BundleJobsGetRunningJPAExecutor(limit));
133                    bundleLists.add(pendingJobCheckList);
134                    bundleLists.add(runningJobCheckList);
135                }
136                else {
137                    LOG.info("Running bundle status service from last instance time =  "
138                            + DateUtils.convertDateToString(lastInstanceStartTime));
139                    // this is not the first instance, we should only check jobs that have actions been
140                    // updated >= start time of last service run;
141                    List<BundleActionBean> actionList = jpaService
142                            .execute(new BundleActionsGetByLastModifiedTimeJPAExecutor(lastInstanceStartTime));
143                    Set<String> bundleIds = new HashSet<String>();
144                    for (BundleActionBean action : actionList) {
145                        bundleIds.add(action.getBundleId());
146                    }
147                    pendingJobCheckList = new ArrayList<BundleJobBean>();
148                    for (String bundleId : bundleIds.toArray(new String[bundleIds.size()])) {
149                        BundleJobBean bundle = jpaService.execute(new BundleJobGetJPAExecutor(bundleId));
150                        // Running bundle job might have pending false
151                        if (bundle.isPending() || bundle.getStatus().equals(Job.Status.RUNNING)) {
152                            pendingJobCheckList.add(bundle);
153                        }
154                    }
155                    runningJobCheckList = pendingJobCheckList;
156                    bundleLists.add(pendingJobCheckList);
157                }
158                updateBundleJobStatus(bundleLists);
159            }

161            private void updateBundleJobStatus(List<List<BundleJobBean>> bundleLists) throws JPAExecutorException,
162                    CommandException {
163                if (bundleLists != null) {
164                    for (List<BundleJobBean> listBundleBean : bundleLists) {
165                        for (BundleJobBean bundleJob : listBundleBean) {
166                            String jobId = bundleJob.getId();
167                            Job.Status[] bundleStatus = new Job.Status[1];
168                            bundleStatus[0] = bundleJob.getStatus();
169                            List<BundleActionBean> bundleActions = jpaService
170                                    .execute(new BundleActionsGetJPAExecutor(jobId));
171                            HashMap<Job.Status, Integer> bundleActionStatus = new HashMap<Job.Status, Integer>();
172                            boolean foundPending = false;
173                            for (BundleActionBean bAction : bundleActions) {
174                                if (!bAction.isPending()) {
175                                    int counter = 0;
176                                    if (bundleActionStatus.containsKey(bAction.getStatus())) {
177                                        counter = bundleActionStatus.get(bAction.getStatus()) + 1;
178                                    }
179                                    else {
180                                        ++counter;
181                                    }
182                                    bundleActionStatus.put(bAction.getStatus(), counter);
183                                    if (bAction.getCoordId() == null
184                                            && (bAction.getStatus() == Job.Status.FAILED || bAction.getStatus() == Job.Status.KILLED)) {
185                                        (new BundleKillXCommand(jobId)).call();
186                                        LOG.info("Bundle job [" + jobId
187                                                + "] has been killed since one of its coordinator job failed submission.");
188                                    }
189                                }
190                                else {
191                                    foundPending = true;
192                                    break;
193                                }
194                            }
195    
196                            if(foundPending){
197                                continue;
198                            }
199    
200                            if (checkTerminalStatus(bundleActionStatus, bundleActions, bundleStatus)) {
201                                LOG.info("Bundle job [" + jobId + "] Status set to " + bundleStatus[0].toString());
202                                updateBundleJob(bundleActionStatus, bundleActions, bundleJob, bundleStatus[0]);
203                                continue;
204                            }
205                            else if (checkPrepStatus(bundleActionStatus, bundleActions, bundleStatus)) {
206                                LOG.info("Bundle job [" + jobId + "] Status set to " + bundleStatus[0].toString());
207                                updateBundleJob(bundleActionStatus, bundleActions, bundleJob, bundleStatus[0]);
208                                continue;
209                            }
210                            else if (checkPausedStatus(bundleActionStatus, bundleActions, bundleStatus)) {
211                                LOG.info("Bundle job [" + jobId + "] Status set to " + bundleStatus[0].toString());
212                                updateBundleJob(bundleActionStatus, bundleActions, bundleJob, bundleStatus[0]);
213                                continue;
214                            }
215                            else if (checkSuspendStatus(bundleActionStatus, bundleActions, bundleStatus)) {
216                                LOG.info("Bundle job [" + jobId + "] Status set to " + bundleStatus[0].toString());
217                                updateBundleJob(bundleActionStatus, bundleActions, bundleJob, bundleStatus[0]);
218                                continue;
219                            }
220                            else if (checkRunningStatus(bundleActionStatus, bundleActions, bundleStatus)) {
221                                LOG.info("Bundle job [" + jobId + "] Status set to " + bundleStatus[0].toString());
222                                updateBundleJob(bundleActionStatus, bundleActions, bundleJob, bundleStatus[0]);
223                                continue;
224                            }
225                        }
226                    }
227                }
228            }

230            private void updateCoordJobStatus(List<CoordinatorJobBean> CoordList) throws JPAExecutorException,
231                    CommandException {
232                if (CoordList != null) {
233    
234                    for (CoordinatorJobBean coordJob : CoordList) {
235                        String jobId = coordJob.getId();
236                        Job.Status[] coordStatus = new Job.Status[1];
237                        coordStatus[0] = coordJob.getStatus();
238                        List<CoordinatorActionBean> coordActions = jpaService.execute(new CoordJobGetActionsJPAExecutor(
239                                jobId));
240                        HashMap<CoordinatorAction.Status, Integer> coordActionStatus = new HashMap<CoordinatorAction.Status, Integer>();
241                        boolean foundPending = false;
242                        for (CoordinatorActionBean cAction : coordActions) {
243                            if (!cAction.isPending()) {
244                                int counter = 0;
245                                if (coordActionStatus.containsKey(cAction.getStatus())) {
246                                    counter = coordActionStatus.get(cAction.getStatus()) + 1;
247                                }
248                                else {
249                                    ++counter;
250                                }
251                                coordActionStatus.put(cAction.getStatus(), counter);
252                            }
253                            else {
254                                foundPending = true;
255                                break;
256                            }
257                        }
258    
259                        if (foundPending) {
260                            continue;
261                        }
262    
263                        if (checkCoordTerminalStatus(coordActionStatus, coordActions, coordStatus)) {
264                            LOG.info("Coord job [" + jobId + "] Status set to " + coordStatus[0].toString());
265                            updateCoordJob(coordActionStatus,coordActions,coordJob, coordStatus[0]);
266                            continue;
267                        }
268                        else if (checkCoordSuspendStatus(coordActionStatus, coordActions, coordStatus)) {
269                            LOG.info("Coord job [" + jobId + "] Status set to " + coordStatus[0].toString());
270                            updateCoordJob(coordActionStatus,coordActions,coordJob, coordStatus[0]);
271                            continue;
272                        }
273                    }
274    
275                }
276            }

278            private boolean checkTerminalStatus(HashMap<Job.Status, Integer> bundleActionStatus,
279                    List<BundleActionBean> bundleActions, Job.Status[] bundleStatus) {
280                boolean ret = false;
281                int totalValuesSucceed = 0;
282                if (bundleActionStatus.containsKey(Job.Status.SUCCEEDED)) {
283                    totalValuesSucceed = bundleActionStatus.get(Job.Status.SUCCEEDED);
284                }
285                int totalValuesFailed = 0;
286                if (bundleActionStatus.containsKey(Job.Status.FAILED)) {
287                    totalValuesFailed = bundleActionStatus.get(Job.Status.FAILED);
288                }
289                int totalValuesKilled = 0;
290                if (bundleActionStatus.containsKey(Job.Status.KILLED)) {
291                    totalValuesKilled = bundleActionStatus.get(Job.Status.KILLED);
292                }
293    
294                int totalValuesDoneWithError = 0;
295                if (bundleActionStatus.containsKey(Job.Status.DONEWITHERROR)) {
296                    totalValuesDoneWithError = bundleActionStatus.get(Job.Status.DONEWITHERROR);
297                }
298    
299                if (bundleActions.size() == (totalValuesSucceed + totalValuesFailed + totalValuesKilled + totalValuesDoneWithError)) {
300                    // If all the bundle actions are succeeded then bundle job should be succeeded.
301                    if (bundleActions.size() == totalValuesSucceed) {
302                        bundleStatus[0] = Job.Status.SUCCEEDED;
303                        ret = true;
304                    }
305                    else if (bundleActions.size() == totalValuesKilled) {
306                        // If all the bundle actions are KILLED then bundle job should be KILLED.
307                        bundleStatus[0] = Job.Status.KILLED;
308                        ret = true;
309                    }
310                    else if (bundleActions.size() == totalValuesFailed) {
311                        // If all the bundle actions are FAILED then bundle job should be FAILED.
312                        bundleStatus[0] = Job.Status.FAILED;
313                        ret = true;
314                    }
315                    else {
316                        bundleStatus[0] = Job.Status.DONEWITHERROR;
317                        ret = true;
318                    }
319                }
320                return ret;
321            }

323            private boolean checkCoordTerminalStatus(HashMap<CoordinatorAction.Status, Integer> coordActionStatus,
324                    List<CoordinatorActionBean> coordActions, Job.Status[] coordStatus) {
325                boolean ret = false;
326                int totalValuesSucceed = 0;
327                if (coordActionStatus.containsKey(CoordinatorAction.Status.SUCCEEDED)) {
328                    totalValuesSucceed = coordActionStatus.get(CoordinatorAction.Status.SUCCEEDED);
329                }
330                int totalValuesFailed = 0;
331                if (coordActionStatus.containsKey(CoordinatorAction.Status.FAILED)) {
332                    totalValuesFailed = coordActionStatus.get(CoordinatorAction.Status.FAILED);
333                }
334                int totalValuesKilled = 0;
335                if (coordActionStatus.containsKey(CoordinatorAction.Status.KILLED)) {
336                    totalValuesKilled = coordActionStatus.get(CoordinatorAction.Status.KILLED);
337                }
338    
339                int totalValuesTimeOut = 0;
340                if (coordActionStatus.containsKey(CoordinatorAction.Status.TIMEDOUT)) {
341                    totalValuesTimeOut = coordActionStatus.get(CoordinatorAction.Status.TIMEDOUT);
342                }
343    
344                if (coordActions.size() == (totalValuesSucceed + totalValuesFailed + totalValuesKilled + totalValuesTimeOut)) {
345                    // If all the bundle actions are succeeded then bundle job should be succeeded.
346                    if (coordActions.size() == totalValuesSucceed) {
347                        coordStatus[0] = Job.Status.SUCCEEDED;
348                        ret = true;
349                    }
350                    else if (coordActions.size() == totalValuesKilled) {
351                        // If all the bundle actions are KILLED then bundle job should be KILLED.
352                        coordStatus[0] = Job.Status.KILLED;
353                        ret = true;
354                    }
355                    else if (coordActions.size() == totalValuesFailed) {
356                        // If all the bundle actions are FAILED then bundle job should be FAILED.
357                        coordStatus[0] = Job.Status.FAILED;
358                        ret = true;
359                    }
360                    else if (coordActions.size() == totalValuesTimeOut) {
361                        // If all the bundle actions are TIMEOUT then bundle job should be FAILED.
362                        coordStatus[0] = Job.Status.FAILED;
363                        ret = true;
364                    }
365                    else {
366                        coordStatus[0] = Job.Status.DONEWITHERROR;
367                        ret = true;
368                    }
369                }
370                return ret;
371            }

373            private boolean checkPrepStatus(HashMap<Job.Status, Integer> bundleActionStatus,
374                    List<BundleActionBean> bundleActions, Job.Status[] bundleStatus) {
375                boolean ret = false;
376                if (bundleActionStatus.containsKey(Job.Status.PREP)) {
377                    // If all the bundle actions are PREP then bundle job should be RUNNING.
378                    if (bundleActions.size() > bundleActionStatus.get(Job.Status.PREP)) {
379                        bundleStatus[0] = Job.Status.RUNNING;
380                        ret = true;
381                    }
382                }
383                return ret;
384            }

386            private boolean checkPausedStatus(HashMap<Job.Status, Integer> bundleActionStatus,
387                    List<BundleActionBean> bundleActions, Job.Status[] bundleStatus) {
388                boolean ret = false;
389                if (bundleActionStatus.containsKey(Job.Status.PAUSED)) {
390                    if (bundleActions.size() == bundleActionStatus.get(Job.Status.PAUSED)) {
391                        bundleStatus[0] = Job.Status.PAUSED;
392                        ret = true;
393                    }
394                    else if (bundleActionStatus.containsKey(Job.Status.PAUSEDWITHERROR)
395                            && (bundleActions.size() == bundleActionStatus.get(Job.Status.PAUSED)
396                                    + bundleActionStatus.get(Job.Status.PAUSEDWITHERROR))) {
397                        // bundleStatus = Job.Status.PAUSEDWITHERROR;
398                        // We need to change this to PAUSEDWITHERROR in future when we add this to coordinator
399                        bundleStatus[0] = Job.Status.PAUSED;
400                        ret = true;
401                    }
402                }
403                return ret;
404            }

438            private boolean checkRunningStatus(HashMap<Job.Status, Integer> bundleActionStatus,
439                    List<BundleActionBean> bundleActions, Job.Status[] bundleStatus) {
440                boolean ret = false;
441                if (bundleActionStatus.containsKey(Job.Status.RUNNING)) {
442                    // If all the bundle actions are succeeded then bundle job should be succeeded.
443                    if (bundleActions.size() == bundleActionStatus.get(Job.Status.RUNNING)) {
444                        bundleStatus[0] = Job.Status.RUNNING;
445                        ret = true;
446                    }
447                    else if (bundleActionStatus.get(Job.Status.RUNNING) > 0) {
448                        if ((bundleActionStatus.containsKey(Job.Status.FAILED) && bundleActionStatus.get(Job.Status.FAILED) > 0)
449                                || (bundleActionStatus.containsKey(Job.Status.KILLED) && bundleActionStatus
450                                        .get(Job.Status.KILLED) > 0)
451                                || (bundleActionStatus.containsKey(Job.Status.DONEWITHERROR) && bundleActionStatus
452                                        .get(Job.Status.DONEWITHERROR) > 0)
453                                || (bundleActionStatus.containsKey(Job.Status.RUNNINGWITHERROR) && bundleActionStatus
454                                        .get(Job.Status.RUNNINGWITHERROR) > 0)) {
455                            // bundleStatus = Job.Status.RUNNINGWITHERROR;
456                            // We need to change this to RUNNINGWIHERROR in future when we add this to coordinator
457                            bundleStatus[0] = Job.Status.RUNNING;
458                            ret = true;
459                        }
460                    }
461                }
462                return ret;
463            }

465            private void updateBundleJob(HashMap<Job.Status, Integer> bundleActionStatus,
466                    List<BundleActionBean> bundleActions, BundleJobBean bundleJob, Job.Status bundleStatus)
467                    throws JPAExecutorException {
468                String jobId = bundleJob.getId();
469                boolean pendingBundleJob = bundleJob.isPending();
470                // Checking the bundle pending should be updated or not
471                int totalNonPendingActions = 0;
472                for (Job.Status js : bundleActionStatus.keySet()) {
473                    totalNonPendingActions += bundleActionStatus.get(js);
474                }
475    
476                if (totalNonPendingActions == bundleActions.size()) {
477                    pendingBundleJob = false;
478                }
479    
480                // Update the Bundle Job
481                bundleJob.setStatus(bundleStatus);
482                if (pendingBundleJob) {
483                    bundleJob.setPending();
484                    LOG.info("Bundle job [" + jobId + "] Pending set to TRUE");
485                }
486                else {
487                    bundleJob.resetPending();
488                    LOG.info("Bundle job [" + jobId + "] Pending set to FALSE");
489                }
490                jpaService.execute(new BundleJobUpdateJPAExecutor(bundleJob));
491            }

493            private void updateCoordJob(HashMap<CoordinatorAction.Status, Integer> coordActionStatus,
494                    List<CoordinatorActionBean> coordActions, CoordinatorJobBean coordJob, Job.Status coordStatus)
495                    throws JPAExecutorException, CommandException {
496                Job.Status prevStatus = coordJob.getStatus();
497                // Update the Coord Job
498                if (coordJob.getStatus() == Job.Status.SUCCEEDED || coordJob.getStatus() == Job.Status.FAILED
499                        || coordJob.getStatus() == Job.Status.KILLED || coordJob.getStatus() == Job.Status.DONEWITHERROR) {
500                    if (coordStatus == Job.Status.SUSPENDED) {
501                        LOG.info("Coord Job [" + coordJob.getId()
502                                + "] status can not be updated as its already in Terminal state");
503                        return;
504                    }
505                }
506    
507                String jobId = coordJob.getId();
508                boolean pendingCoordJob = coordJob.isPending();
509                // Checking the bundle pending should be updated or not
510                int totalNonPendingActions = 0;
511                for (CoordinatorAction.Status js : coordActionStatus.keySet()) {
512                    totalNonPendingActions += coordActionStatus.get(js);
513                }
514    
515                if (totalNonPendingActions == coordActions.size()) {
516                    pendingCoordJob = false;
517                }
518    
519                if (pendingCoordJob) {
520                    coordJob.setPending();
521                    LOG.info("Coord job [" + jobId + "] Pending set to TRUE");
522                }
523                else {
524                    coordJob.resetPending();
525                    LOG.info("Coord job [" + jobId + "] Pending set to FALSE");
526                }
527                coordJob.setStatus(coordStatus);
528                jpaService.execute(new CoordJobUpdateJPAExecutor(coordJob));
529                // update bundle action only when status changes in coord job
530                if (coordJob.getBundleId() != null) {
531                    if (!prevStatus.equals(coordJob.getStatus())) {
532                        BundleStatusUpdateXCommand bundleStatusUpdate = new BundleStatusUpdateXCommand(coordJob, prevStatus);
533                        bundleStatusUpdate.call();
534                    }
535                }
536            }

538            /**
539             * Aggregate coordinator actions' status to coordinator jobs
540             *
541             * @throws JPAExecutorException thrown if failed in db updates or retrievals
542             * @throws CommandException thrown if failed to run commands
543             */
544            private void coordTransit() throws JPAExecutorException, CommandException {
545                List<CoordinatorJobBean> pendingJobCheckList = null;
546                if (lastInstanceStartTime == null) {
547                    LOG.info("Running coordinator status service first instance");
548                    // this is the first instance, we need to check for all pending jobs;
549                    pendingJobCheckList = jpaService.execute(new CoordJobsGetPendingJPAExecutor(limit));
550                }
551                else {
552                    LOG.info("Running coordinator status service from last instance time =  "
553                            + DateUtils.convertDateToString(lastInstanceStartTime));
554                    // this is not the first instance, we should only check jobs that have actions been
555                    // updated >= start time of last service run;
556                    List<CoordinatorActionBean> actionList = jpaService
557                            .execute(new CoordActionsGetByLastModifiedTimeJPAExecutor(lastInstanceStartTime));
558                    Set<String> coordIds = new HashSet<String>();
559                    for (CoordinatorActionBean action : actionList) {
560                        coordIds.add(action.getJobId());
561                    }
562                    pendingJobCheckList = new ArrayList<CoordinatorJobBean>();
563                    for (String coordId : coordIds.toArray(new String[coordIds.size()])) {
564                        CoordinatorJobBean coordJob = jpaService.execute(new CoordJobGetJPAExecutor(coordId));
565                        if (coordJob.isPending()) {
566                            pendingJobCheckList.add(coordJob);
567                        }
568                    }
569                }
570                updateCoordJobStatus(pendingJobCheckList);
571            }
572        }

575         * Initializes the {@link StatusTransitService}.
