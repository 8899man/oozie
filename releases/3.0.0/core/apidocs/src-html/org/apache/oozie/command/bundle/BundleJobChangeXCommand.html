015    package org.apache.oozie.command.bundle;
016    
017    import java.util.Date;
018    import java.util.HashSet;
019    import java.util.List;
020    import java.util.Map;
021    import java.util.Set;
022    
023    import org.apache.oozie.BundleActionBean;
024    import org.apache.oozie.BundleJobBean;
025    import org.apache.oozie.ErrorCode;
026    import org.apache.oozie.XException;
027    import org.apache.oozie.client.Job;
028    import org.apache.oozie.client.OozieClient;
029    import org.apache.oozie.command.CommandException;
030    import org.apache.oozie.command.PreconditionException;
031    import org.apache.oozie.command.XCommand;
032    import org.apache.oozie.command.coord.CoordChangeXCommand;
033    import org.apache.oozie.executor.jpa.BundleActionUpdateJPAExecutor;
034    import org.apache.oozie.executor.jpa.BundleActionsGetJPAExecutor;
035    import org.apache.oozie.executor.jpa.BundleJobGetJPAExecutor;
036    import org.apache.oozie.executor.jpa.BundleJobUpdateJPAExecutor;
037    import org.apache.oozie.service.JPAService;
038    import org.apache.oozie.service.Services;
039    import org.apache.oozie.util.DateUtils;
040    import org.apache.oozie.util.JobUtils;
041    import org.apache.oozie.util.LogUtils;
042    import org.apache.oozie.util.ParamChecker;
043    
044    public class BundleJobChangeXCommand extends XCommand<Void> {
045        private String jobId;
046        private String changeValue;
047        private JPAService jpaService;
048        private List<BundleActionBean> bundleActions;
049        private BundleJobBean bundleJob;
050        private Date newPauseTime = null;
051        boolean isChangePauseTime = false;
052    
053        private static final Set<String> ALLOWED_CHANGE_OPTIONS = new HashSet<String>();
054        static {
055            ALLOWED_CHANGE_OPTIONS.add("pausetime");
056        }
057    
058        /**
059         * @param id bundle job id
060         * @param changeValue change value
061         *
062         * @throws CommandException thrown if failed to change bundle
063         */
064        public BundleJobChangeXCommand(String id, String changeValue) throws CommandException {
065            super("bundle_change", "bundle_change", 1);
066            this.jobId = ParamChecker.notEmpty(id, "id");
067            this.changeValue = ParamChecker.notEmpty(changeValue, "changeValue");
068        }
069    
070        /**
071         * Check if new pause time is future time.
072         *
073         * @param newPauseTime new pause time.
074         * @param newEndTime new end time, can be null meaning no change on end time.
075         * @throws CommandException thrown if new pause time is not valid.
076         */
077        private void checkPauseTime(Date newPauseTime) throws CommandException {
078            // New pauseTime has to be a non-past time.
079            Date d = new Date();
080            if (newPauseTime.before(d)) {
081                throw new CommandException(ErrorCode.E1317, newPauseTime, "must be a non-past time");
082            }
083        }
084    
085        /**
086         * validate if change value is valid.
087         *
088         * @param changeValue change value.
089         * @throws CommandException thrown if changeValue cannot be parsed properly.
090         */
091        private void validateChangeValue(String changeValue) throws CommandException {
092            Map<String, String> map = JobUtils.parseChangeValue(changeValue);
093    
094            if (map.size() > ALLOWED_CHANGE_OPTIONS.size() || !map.containsKey(OozieClient.CHANGE_VALUE_PAUSETIME)) {
095                throw new CommandException(ErrorCode.E1317, changeValue, "can only change pausetime");
096            }
097    
098            if (map.containsKey(OozieClient.CHANGE_VALUE_PAUSETIME)) {
099                isChangePauseTime = true;
100            }
101            else {
102                throw new CommandException(ErrorCode.E1317, changeValue, "should change pausetime");
103            }
104    
105            String value = map.get(OozieClient.CHANGE_VALUE_PAUSETIME);
106            if (!value.equals(""))   {
107                try {
108                    newPauseTime = DateUtils.parseDateUTC(value);
109                }
110                catch (Exception ex) {
111                    throw new CommandException(ErrorCode.E1317, value, "is not a valid date");
112                }
113    
114                checkPauseTime(newPauseTime);
115            }
116        }
117    
118        /* (non-Javadoc)
119         * @see org.apache.oozie.command.XCommand#execute()
120         */
121        @Override
122        protected Void execute() throws CommandException {
123            try {
124                if (isChangePauseTime) {
125                    bundleJob.setPauseTime(newPauseTime);
126    
127                    for (BundleActionBean action : this.bundleActions) {
128                        // queue coord change commands;
129                        if (action.getStatus() != Job.Status.KILLED && action.getCoordId() != null) {
130                            queue(new CoordChangeXCommand(action.getCoordId(), changeValue));
131                            LOG.info("Queuing CoordChangeXCommand coord job = " + action.getCoordId() + " to change "
132                                    + changeValue);
133                            action.setPending(action.getPending() + 1);
134                            jpaService.execute(new BundleActionUpdateJPAExecutor(action));
135                        }
136                    }
137                    jpaService.execute(new BundleJobUpdateJPAExecutor(bundleJob));
138                }
139                return null;
140            }
141            catch (XException ex) {
142                throw new CommandException(ex);
143            }
144        }
145    
146        /* (non-Javadoc)
147         * @see org.apache.oozie.command.XCommand#getEntityKey()
148         */
149        @Override
150        protected String getEntityKey() {
151            return this.jobId;
152        }
153    
154        /* (non-Javadoc)
155         * @see org.apache.oozie.command.XCommand#isLockRequired()
156         */
157        @Override
158        protected boolean isLockRequired() {
159            return true;
160        }
161    
162        /* (non-Javadoc)
163         * @see org.apache.oozie.command.XCommand#loadState()
164         */
165        @Override
166        protected void loadState() throws CommandException {
167            try{
168                this.bundleActions = jpaService.execute(new BundleActionsGetJPAExecutor(jobId));
169            }
170            catch(Exception Ex){
171                throw new CommandException(ErrorCode.E1311,this.jobId);
172            }
173        }
174    
175        /* (non-Javadoc)
176         * @see org.apache.oozie.command.XCommand#verifyPrecondition()
177         */
178        @Override
179        protected void verifyPrecondition() throws CommandException, PreconditionException {
180        }
181    
182        /* (non-Javadoc)
183         * @see org.apache.oozie.command.XCommand#eagerLoadState()
184         */
185        @Override
186        protected void eagerLoadState() throws CommandException {
187            try {
188                jpaService = Services.get().get(JPAService.class);
189    
190                if (jpaService != null) {
191                    this.bundleJob = jpaService.execute(new BundleJobGetJPAExecutor(jobId));
192                    LogUtils.setLogInfo(bundleJob, logInfo);
193                }
194                else {
195                    throw new CommandException(ErrorCode.E0610);
196                }
197            }
198            catch (XException ex) {
199                throw new CommandException(ex);
200            }
201        }
202    
203        /* (non-Javadoc)
204         * @see org.apache.oozie.command.XCommand#eagerVerifyPrecondition()
205         */
206        @Override
207        protected void eagerVerifyPrecondition() throws CommandException, PreconditionException {
208            validateChangeValue(changeValue);
209    
210            if (bundleJob == null) {
211                LOG.info("BundleChangeCommand not succeeded - " + "job " + jobId + " does not exist");
212                throw new PreconditionException(ErrorCode.E1314, jobId);
213            }
214            if (bundleJob.getStatus() == Job.Status.SUCCEEDED || bundleJob.getStatus() == Job.Status.FAILED
215                    || bundleJob.getStatus() == Job.Status.KILLED || bundleJob == null) {
216                LOG.info("BundleChangeCommand not succeeded - " + "job " + jobId + " finished, status is " + bundleJob.getStatusStr());
217                throw new PreconditionException(ErrorCode.E1312, jobId, bundleJob.getStatus().toString());
218            }
219        }
220    
221    }
