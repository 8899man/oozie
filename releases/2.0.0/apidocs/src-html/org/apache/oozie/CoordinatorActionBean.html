056    
057        @NamedQuery(name = "DELETE_COMPLETED_COORD_ACTIONS", query = "delete from CoordinatorActionBean a where a.id = :id and (a.status = 'SUCCEEDED' OR a.status = 'FAILED' OR a.status = 'KILLED')"),
058    
059        @NamedQuery(name = "GET_COORD_ACTIONS", query = "select OBJECT(w) from CoordinatorActionBean w"),

061        @NamedQuery(name = "GET_COMPLETED_ACTIONS_OLDER_THAN", query = "select OBJECT(a) from CoordinatorActionBean a where a.createdTimestamp < :createdTime and (a.status = 'SUCCEEDED' OR a.status = 'FAILED' OR a.status = 'KILLED')"),

088            })
089    public class CoordinatorActionBean extends JsonCoordinatorAction implements
090            Writable {
091    
092        @Basic
093        @Index
094        @Column(name = "job_id")
095        private String jobId;
096    
097        @Basic
098        @Index
099        @Column(name = "status")
100        private String status = null;
101    
102        @Basic
103        @Column(name = "nominal_time")
104        private java.sql.Timestamp nominalTimestamp = null;
105    
106        @Basic
107        @Index
108        @Column(name = "last_modified_time")
109        private java.sql.Timestamp lastModifiedTimestamp = null;
110    
111        @Basic
112        @Index
113        @Column(name = "created_time")
114        private java.sql.Timestamp createdTimestamp = null;
115    
116        @Basic
117        @Index
118        @Column(name = "external_id")
119        private String externalId;
120    
121        @Column(name = "sla_xml")
122        @Lob
123        private String slaXml = null;
124    
125        public CoordinatorActionBean() {
126        }
127    
128        /**
129         * Serialize the coordinator bean to a data output.
130         *
131         * @param dataOutput data output.
132         * @throws IOException thrown if the coordinator bean could not be serialized.
133         */
134        public void write(DataOutput dataOutput) throws IOException {
135            WritableUtils.writeStr(dataOutput, getJobId());
136            WritableUtils.writeStr(dataOutput, getType());
137            WritableUtils.writeStr(dataOutput, getId());
138            WritableUtils.writeStr(dataOutput, getCreatedConf());
139            WritableUtils.writeStr(dataOutput, getStatus().toString());
140            dataOutput.writeInt(getActionNumber());
141            WritableUtils.writeStr(dataOutput, getRunConf());
142            WritableUtils.writeStr(dataOutput, getExternalStatus());
143            WritableUtils.writeStr(dataOutput, getTrackerUri());
144            WritableUtils.writeStr(dataOutput, getErrorCode());
145            WritableUtils.writeStr(dataOutput, getErrorMessage());
146        }
147    
148        /**
149         * Deserialize a coordinator bean from a data input.
150         *
151         * @param dataInput data input.
152         * @throws IOException thrown if the workflow bean could not be deserialized.
153         */
154        public void readFields(DataInput dataInput) throws IOException {
155            setJobId(WritableUtils.readStr(dataInput));
156            setType(WritableUtils.readStr(dataInput));
157            setId(WritableUtils.readStr(dataInput));
158            setCreatedConf(WritableUtils.readStr(dataInput));
159            setStatus(CoordinatorAction.Status.valueOf(WritableUtils
160                    .readStr(dataInput)));
161            setRunConf(WritableUtils.readStr(dataInput));
162            setExternalStatus(WritableUtils.readStr(dataInput));
163            setTrackerUri(WritableUtils.readStr(dataInput));
164            setConsoleUrl(WritableUtils.readStr(dataInput));
165            long d = dataInput.readLong();
166            if (d != -1) {
167                setCreatedTime(new Date(d));
168            }
169            d = dataInput.readLong();
170            if (d != -1) {
171                setLastModifiedTime(new Date(d));
172            }
173            d = dataInput.readLong();
174            d = dataInput.readLong();
175        }
176

180        }
181

185            this.jobId = id;
186        }

191        }

194        public void setStatus(Status status) {
195            super.setStatus(status);
196            this.status = status.toString();
197        }
198

201            this.createdTimestamp = DateUtils.convertDateToTimestamp(createdTime);
202            super.setCreatedTime(createdTime);
203        }
204    
205        @Override
206        public void setNominalTime(Date nominalTime) {
207            this.nominalTimestamp = DateUtils.convertDateToTimestamp(nominalTime);
208            super.setNominalTime(nominalTime);
209        }
210    
211        @Override
212        public void setLastModifiedTime(Date lastModifiedTime) {
213            this.lastModifiedTimestamp = DateUtils.convertDateToTimestamp(lastModifiedTime);
214            super.setLastModifiedTime(lastModifiedTime);
215        }
216

219            return DateUtils.toDate(createdTimestamp);

223            return createdTimestamp;

228            return DateUtils.toDate(lastModifiedTimestamp);
