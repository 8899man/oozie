018    package org.apache.oozie.command.coord;
019    
020    import java.sql.Timestamp;
021    
022    import org.apache.oozie.CoordinatorJobBean;
023    import org.apache.oozie.client.CoordinatorJob;
024    import org.apache.oozie.command.CommandException;
025    import org.apache.oozie.store.CoordinatorStore;
026    import org.apache.oozie.store.StoreException;
027    import org.apache.oozie.util.DateUtils;
028    import org.apache.oozie.util.XLog;
029    
030    public class CoordJobMatLookupCommand extends CoordinatorCommand<Void> {
031        private final XLog log = XLog.getLog(getClass());
032        private int materializationWindow;
033        private String jobId;
034    
035        public CoordJobMatLookupCommand(String id, int materializationWindow) {
036            super("materialization_lookup", "materialization_lookup", -1, XLog.STD);
037            this.jobId = id;
038            this.materializationWindow = materializationWindow;
039        }
040    
041        @Override
042        protected Void call(CoordinatorStore store) throws StoreException, CommandException {
043            //CoordinatorJobBean coordJob = store.getCoordinatorJob(jobId, true);
044            CoordinatorJobBean coordJob = store.getEntityManager().find(CoordinatorJobBean.class, jobId);
045            setLogInfo(coordJob);
046    
047            if (!(coordJob.getStatus() == CoordinatorJobBean.Status.PREP || coordJob.getStatus() == CoordinatorJobBean.Status.RUNNING)) {
048                log.debug("CoordJobMatLookupCommand for jobId=" + jobId + " job is not in PREP or RUNNING but in "
049                        + coordJob.getStatus());
050                return null;
051            }
052    
053            if (coordJob.getNextMaterializedTimestamp() != null
054                    && coordJob.getNextMaterializedTimestamp().compareTo(coordJob.getEndTimestamp()) >= 0) {
055                log.debug("CoordJobMatLookupCommand for jobId=" + jobId + " job is already materialized");
056                return null;
057            }
058    
059            if (coordJob.getNextMaterializedTimestamp() != null
060                    && coordJob.getNextMaterializedTimestamp().compareTo(new Timestamp(System.currentTimeMillis())) >= 0) {
061                log.debug("CoordJobMatLookupCommand for jobId=" + jobId + " job is already materialized");
062                return null;
063            }
064    
065            Timestamp startTime = coordJob.getNextMaterializedTimestamp();
066            if (startTime == null) {
067                startTime = coordJob.getStartTimestamp();
068            }
069            // calculate end time by adding materializationWindow to start time.
070            // need to convert materializationWindow from secs to milliseconds
071            long startTimeMilli = startTime.getTime();
072            long endTimeMilli = startTimeMilli + (materializationWindow * 1000);
073            Timestamp endTime = new Timestamp(endTimeMilli);
074            // if MaterializationWindow end time is greater than endTime
075            // for job, then set it to endTime of job
076            Timestamp jobEndTime = coordJob.getEndTimestamp();
077            if (endTime.compareTo(jobEndTime) > 0) {
078                endTime = jobEndTime;
079            }
080            // update status of job from PREP or RUNNING to PREMATER in coordJob
081            coordJob.setStatus(CoordinatorJob.Status.PREMATER);
082            store.updateCoordinatorJobStatus(coordJob);
083    
084            log.debug("Materializing coord job id=" + jobId + ", start=" + DateUtils.toDate(startTime) + ", end=" + DateUtils.toDate(endTime)
085                    + ", window=" + materializationWindow + ", status=PREMATER");
086            queueCallable(new CoordActionMaterializeCommand(jobId, startTime, endTime), 100);
087            return null;
088        }
089    
090        @Override
091        protected Void execute(CoordinatorStore store) throws StoreException, CommandException {
092            log.info("STARTED CoordJobMatLookupCommand jobId=" + jobId + ", materializationWindow="
093                    + materializationWindow);
094            try {
095                if (lock(jobId)) {
096                    call(store);
097                }
098                else {
099                    queueCallable(new CoordJobMatLookupCommand(jobId, materializationWindow), LOCK_FAILURE_REQUEUE_INTERVAL);
100                    log.warn("CoordJobMatLookupCommand lock was not acquired - failed jobId=" + jobId
101                            + ". Requeing the same.");
102                }
103            }
104            catch (InterruptedException e) {
105                queueCallable(new CoordJobMatLookupCommand(jobId, materializationWindow), LOCK_FAILURE_REQUEUE_INTERVAL);
106                log.warn("CoordJobMatLookupCommand lock acquiring failed with exception " + e.getMessage() + " for jobId="
107                        + jobId + " Requeing the same.");
108            }
109            finally {
110                log.info("ENDED CoordJobMatLookupCommand jobId=" + jobId + ", materializationWindow="
111                        + materializationWindow);
112            }
113            return null;
114        }
115    }
