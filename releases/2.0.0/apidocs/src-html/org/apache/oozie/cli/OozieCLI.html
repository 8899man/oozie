243        /**
244         * Run a CLI programmatically. <p/> It does not exit the JVM. <p/> A CLI instance can be used only once.
245         *
246         * @param args options and arguments for the Oozie CLI.
247         * @return '0' (success), '-1' (failure).
248         */
249        public synchronized int run(String[] args) {
250            if (used) {
251                throw new IllegalStateException("CLI instance already used");
252            }
253            used = true;
254            CLIParser parser = new CLIParser(OOZIE_OPTION, getCLIHelp());
255            parser.addCommand(HELP_CMD, "", "display usage", new Options(), false);
256            parser.addCommand(VERSION_CMD, "", "show client version", new Options(), false);
257            parser.addCommand(JOB_CMD, "", "job operations", createJobOptions(), false);
258            parser.addCommand(JOBS_CMD, "", "jobs status", createJobsOptions(), false);
259            parser.addCommand(ADMIN_CMD, "", "admin operations", createAdminOptions(), false);
260            parser.addCommand(VALIDATE_CMD, "", "validate a workflow XML file", new Options(), true);
261            parser.addCommand(SLA_CMD, "", "sla operations (Supported in Oozie-2.0 or later)", createSlaOptions(), false);
262    
263            try {
264                CLIParser.Command command = parser.parse(args);
265                if (command.getName().equals(HELP_CMD)) {
266                    parser.showHelp();
267                }
268                else {
269                    if (command.getName().equals(JOB_CMD)) {
270                        jobCommand(command.getCommandLine());
271                    }
272                    else {
273                        if (command.getName().equals(JOBS_CMD)) {
274                            jobsCommand(command.getCommandLine());
275                        }
276                        else {
277                            if (command.getName().equals(ADMIN_CMD)) {
278                                adminCommand(command.getCommandLine());
279                            }
280                            else {
281                                if (command.getName().equals(VERSION_CMD)) {
282                                    versionCommand();
283                                }
284                                else {
285                                    if (command.getName().equals(VALIDATE_CMD)) {
286                                        validateCommand(command.getCommandLine());
287                                    }
288                                    else {
289                                        if (command.getName().equals(SLA_CMD)) {
290                                            slaCommand(command.getCommandLine());
291                                        }
292                                    }
293                                }
294                            }
295                        }
296                    }
297                }
298    
299                return 0;
300            }
301            catch (OozieCLIException ex) {
302                System.err.println("Error: " + ex.getMessage());
303                return -1;
304            }
305            catch (ParseException ex) {
306                System.err.println("Invalid sub-command: " + ex.getMessage());
307                System.err.println();
308                System.err.println(parser.shortHelp());
309                return -1;
310            }
311            catch (Exception ex) {
312                ex.printStackTrace();
313                System.err.println(ex.getMessage());
314                return -1;
315            }
316        }

318        private String getOozieUrl(CommandLine commandLine) {
319            String url = commandLine.getOptionValue(OOZIE_OPTION);
320            if (url == null) {
321                url = System.getenv(ENV_OOZIE_URL);
322                if (url == null) {
323                    throw new IllegalArgumentException(
324                            "Oozie URL is not available neither in command option or in the environment");
325                }
326            }
327            return url;

330        // Canibalized from Hadoop <code>Configuration.loadResource()</code>.
331        private Properties parse(InputStream is, Properties conf) throws IOException {
332            try {
333                DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
334                // ignore all comments inside the xml file
335                docBuilderFactory.setIgnoringComments(true);
336                DocumentBuilder builder = docBuilderFactory.newDocumentBuilder();
337                Document doc = builder.parse(is);
338                return parseDocument(doc, conf);
339            }
340            catch (SAXException e) {
341                throw new IOException(e);
342            }
343            catch (ParserConfigurationException e) {
344                throw new IOException(e);
345            }
346        }

348        // Canibalized from Hadoop <code>Configuration.loadResource()</code>.
349        private Properties parseDocument(Document doc, Properties conf) throws IOException {
350            try {
351                Element root = doc.getDocumentElement();
352                if (!"configuration".equals(root.getTagName())) {
353                    throw new RuntimeException("bad conf file: top-level element not <configuration>");
354                }
355                NodeList props = root.getChildNodes();
356                for (int i = 0; i < props.getLength(); i++) {
357                    Node propNode = props.item(i);
358                    if (!(propNode instanceof Element)) {
359                        continue;
360                    }
361                    Element prop = (Element) propNode;
362                    if (!"property".equals(prop.getTagName())) {
363                        throw new RuntimeException("bad conf file: element not <property>");
364                    }
365                    NodeList fields = prop.getChildNodes();
366                    String attr = null;
367                    String value = null;
368                    for (int j = 0; j < fields.getLength(); j++) {
369                        Node fieldNode = fields.item(j);
370                        if (!(fieldNode instanceof Element)) {
371                            continue;
372                        }
373                        Element field = (Element) fieldNode;
374                        if ("name".equals(field.getTagName()) && field.hasChildNodes()) {
375                            attr = ((Text) field.getFirstChild()).getData();
376                        }
377                        if ("value".equals(field.getTagName()) && field.hasChildNodes()) {
378                            value = ((Text) field.getFirstChild()).getData();
379                        }
380                    }
381    
382                    if (attr != null && value != null) {
383                        conf.setProperty(attr, value);
384                    }
385                }
386                return conf;
387            }
388            catch (DOMException e) {
389                throw new IOException(e);
390            }
391        }

393        private Properties getConfiguration(CommandLine commandLine) throws IOException {
394            Properties conf = new Properties();
395            conf.setProperty("user.name", System.getProperty("user.name"));
396            String configFile = commandLine.getOptionValue(CONFIG_OPTION);
397            if (configFile == null) {
398                throw new IOException("configuration file not specified");
399            }
400            else {
401                File file = new File(configFile);
402                if (!file.exists()) {
403                    throw new IOException("configuration file [" + configFile + "] not found");
404                }
405                if (configFile.endsWith(".properties")) {
406                    conf.load(new FileReader(file));
407                }
408                else {
409                    if (configFile.endsWith(".xml")) {
410                        parse(new FileInputStream(configFile), conf);
411                    }
412                    else {
413                        throw new IllegalArgumentException("configuration must be a '.properties' or a '.xml' file");
414                    }
415                }
416            }
417            return conf;
418        }

420        /**
421         * Create a OozieClient. <p/> It injects any '-Dheader:' as header to the the {@link
422         * org.apache.oozie.client.OozieClient}.
423         *
424         * @param commandLine the parsed command line options.
425         * @return a pre configured workflow client.
426         * @throws OozieCLIException thrown if the Oozie client could not be configured.
427         */
428        protected OozieClient createOozieClient(CommandLine commandLine) throws OozieCLIException {
429            OozieClient wc = new OozieClient(getOozieUrl(commandLine));
430            for (Map.Entry entry : System.getProperties().entrySet()) {
431                String key = (String) entry.getKey();
432                if (key.startsWith(WS_HEADER_PREFIX)) {
433                    String header = key.substring(WS_HEADER_PREFIX.length());
434                    wc.setHeader(header, (String) entry.getValue());
435                }
436            }
437            return wc;
438        }

442        private void jobCommand(CommandLine commandLine) throws IOException, OozieCLIException {
443            OozieClient wc = createOozieClient(commandLine);
444    
445            List<String> options = new ArrayList<String>();
446            for (Option option : commandLine.getOptions()) {
447                options.add(option.getOpt());
448            }
449    
450            try {
451                if (options.contains(SUBMIT_OPTION)) {
452                    System.out.println(JOB_ID_PREFIX + wc.submit(getConfiguration(commandLine)));
453                }
454                else {
455                    if (options.contains(START_OPTION)) {
456                        wc.start(commandLine.getOptionValue(START_OPTION));
457                    }
458                    else {
459                        if (options.contains(DRYRUN_OPTION)) {
460                            String[] dryrunStr = wc.dryrun(getConfiguration(commandLine)).split("action for new instance");
461                            int arraysize = dryrunStr.length;
462                            System.out.println("***coordJob after parsing: ***");
463                            System.out.println(dryrunStr[0]);
464                            int aLen = dryrunStr.length - 1;
465                            if (aLen < 0) {
466                                aLen = 0;
467                            }
468                            System.out.println("***total coord actions is " + aLen + " ***");
469                            for (int i = 1; i <= arraysize - 1; i++) {
470                                System.out.println(RULER);
471                                System.out.println("coordAction instance: " + i + ":");
472                                System.out.println(dryrunStr[i]);
473                            }
474                        }
475                        else {
476                            if (options.contains(SUSPEND_OPTION)) {
477                                wc.suspend(commandLine.getOptionValue(SUSPEND_OPTION));
478                            }
479                            else {
480                                if (options.contains(RESUME_OPTION)) {
481                                    wc.resume(commandLine.getOptionValue(RESUME_OPTION));
482                                }
483                                else {
484                                    if (options.contains(KILL_OPTION)) {
485                                        wc.kill(commandLine.getOptionValue(KILL_OPTION));
486                                    }
487                                    else {
488                                        if (options.contains(RUN_OPTION)) {
489                                            System.out.println(JOB_ID_PREFIX + wc.run(getConfiguration(commandLine)));
490                                        }
491                                        else {
492                                            if (options.contains(RERUN_OPTION)) {
493                                                wc.reRun(commandLine.getOptionValue(RERUN_OPTION), getConfiguration(commandLine));
494                                            }
495                                            else {
496                                                if (options.contains(INFO_OPTION)) {
497                                                    if (commandLine.getOptionValue(INFO_OPTION).endsWith("-C")) {
498                                                        String s = commandLine.getOptionValue(OFFSET_OPTION);
499                                                        int start = Integer.parseInt((s != null) ? s : "0");
500                                                        s = commandLine.getOptionValue(LEN_OPTION);
501                                                        int len = Integer.parseInt((s != null) ? s : "0");
502                                                        printCoordJob(wc.getCoordJobInfo(commandLine.getOptionValue(INFO_OPTION), start, len), options
503                                                                .contains(LOCAL_TIME_OPTION), options.contains(VERBOSE_OPTION));
504                                                    }
505                                                    else {
506                                                        if (commandLine.getOptionValue(INFO_OPTION).contains("-C@")) {
507                                                            printCoordAction(wc.getCoordActionInfo(commandLine.getOptionValue(INFO_OPTION)), options
508                                                                    .contains(LOCAL_TIME_OPTION));
509                                                        }
510                                                        else {
511                                                            if (commandLine.getOptionValue(INFO_OPTION).contains("-W@")) {
512                                                                printWorkflowAction(wc.getWorkflowActionInfo(commandLine.getOptionValue(INFO_OPTION)), options
513                                                                        .contains(LOCAL_TIME_OPTION));
514                                                            }
515                                                            else {
516                                                                String s = commandLine.getOptionValue(OFFSET_OPTION);
517                                                                int start = Integer.parseInt((s != null) ? s : "0");
518                                                                s = commandLine.getOptionValue(LEN_OPTION);
519                                                                String jobtype = commandLine.getOptionValue(JOBTYPE_OPTION);
520                                                                jobtype = (jobtype != null) ? jobtype : "wf";
521                                                                int len = Integer.parseInt((s != null) ? s : "0");
522                                                                printJob(wc.getJobInfo(commandLine.getOptionValue(INFO_OPTION), start, len), options
523                                                                        .contains(LOCAL_TIME_OPTION), options.contains(VERBOSE_OPTION));
524                                                            }
525                                                        }
526                                                    }
527                                                }
528                                                else {
529                                                    if (options.contains(LOG_OPTION)) {
530                                                        System.out.println(wc.getJobLog(commandLine.getOptionValue(LOG_OPTION)));
531                                                    }
532                                                    else {
533                                                        if (options.contains(DEFINITION_OPTION)) {
534                                                            System.out.println(wc.getJobDefinition(commandLine.getOptionValue(DEFINITION_OPTION)));
535                                                        }
536                                                    }
537                                                }
538                                            }
539                                        }
540                                    }
541                                }
542                            }
543                        }
544                    }
545                }
546    
547            }
548            catch (OozieClientException ex) {
549                throw new OozieCLIException(ex.toString(), ex);
550            }
551        }

554            System.out.println("Job ID : " + coordJob.getId());
555    
556            System.out.println(RULER);
557    
558            List<CoordinatorAction> actions = coordJob.getActions();
559            System.out.println("Job Name : " + maskIfNull(coordJob.getAppName()));
560            System.out.println("App Path : " + maskIfNull(coordJob.getAppPath()));
561            System.out.println("Status   : " + coordJob.getStatus());
562            System.out.println(RULER);
563    
564            if (verbose) {
565                System.out.println("ID" + VERBOSE_DELIMITER + "Action Number" + VERBOSE_DELIMITER + "Console URL"
566                        + VERBOSE_DELIMITER + "Error Code" + VERBOSE_DELIMITER + "Error Message" + VERBOSE_DELIMITER
567                        + "External ID" + VERBOSE_DELIMITER + "External Status" + VERBOSE_DELIMITER + "Job ID"
568                        + VERBOSE_DELIMITER + "Tracker URI" + VERBOSE_DELIMITER + "Created" + VERBOSE_DELIMITER + "Status"
569                        + VERBOSE_DELIMITER + "Last Modified" + VERBOSE_DELIMITER + "Missing Dependencies");
570                System.out.println(RULER);
571    
572                for (CoordinatorAction action : actions) {
573                    System.out.println(maskIfNull(action.getId()) + VERBOSE_DELIMITER + action.getActionNumber()
574                            + VERBOSE_DELIMITER + maskIfNull(action.getConsoleUrl()) + VERBOSE_DELIMITER
575                            + maskIfNull(action.getErrorCode()) + VERBOSE_DELIMITER + maskIfNull(action.getErrorMessage())
576                            + VERBOSE_DELIMITER + maskIfNull(action.getExternalId()) + VERBOSE_DELIMITER
577                            + maskIfNull(action.getExternalStatus()) + VERBOSE_DELIMITER + maskIfNull(action.getJobId())
578                            + VERBOSE_DELIMITER + maskIfNull(action.getTrackerUri()) + VERBOSE_DELIMITER

803        private void adminCommand(CommandLine commandLine) throws OozieCLIException {
804            OozieClient wc = createOozieClient(commandLine);
805    
806            List<String> options = new ArrayList<String>();
807            for (Option option : commandLine.getOptions()) {
808                options.add(option.getOpt());
809            }
810    
811            try {
812                SYSTEM_MODE status = SYSTEM_MODE.NORMAL;
813                if (options.contains(VERSION_OPTION)) {
814                    System.out.println("Oozie server build version: " + wc.getServerBuildVersion());
815                }
816                else {
817                    if (options.contains(SYSTEM_MODE_OPTION)) {
818                        String systemModeOption = commandLine.getOptionValue(SYSTEM_MODE_OPTION).toUpperCase();
819                        try {
820                            // status = safeModeOption.equalsIgnoreCase("ON");
821                            status = SYSTEM_MODE.valueOf(systemModeOption);
822                        }
823                        catch (Exception e) {
824                            throw new OozieCLIException("Invalid input provided for option: " + SYSTEM_MODE_OPTION
825                                    + " value given :" + systemModeOption
826                                    + " Expected values are: NORMAL/NOWEBSERVICE/SAFEMODE ");
827                        }
828                        wc.setSystemMode(status);
829    
830                    }
831                    else {
832                        if (options.contains(STATUS_OPTION)) {
833                            // status = wc.isInSafeMode();
834                            status = wc.getSystemMode();
835                        }
836                    }
837                    // System.out.println("Safemode: " + (status ? "ON" : "OFF"));
838                    System.out.println("System mode: " + status);
839                }
840            }
841            catch (OozieClientException ex) {
842                throw new OozieCLIException(ex.toString(), ex);
843            }
844        }

900        private String maskDate(Date date, boolean isLocalTimeZone) {
901            if (date == null) {
902                return "-";
903            }
904    
905            // SimpleDateFormat dateFormater = new SimpleDateFormat("yyyy-MM-dd
906            // HH:mm Z", Locale.US);
907            SimpleDateFormat dateFormater = new SimpleDateFormat("yyyy-MM-dd HH:mm", Locale.US);
908            if (!isLocalTimeZone) {
909                dateFormater.setTimeZone(TimeZone.getTimeZone("GMT"));
910            }
911            return dateFormater.format(date);
912        }

914        private void validateCommand(CommandLine commandLine) throws OozieCLIException {
915            String[] args = commandLine.getArgs();
916            if (args.length != 1) {
917                throw new OozieCLIException("One file must be specified");
918            }
919            File file = new File(args[0]);
920            if (file.exists()) {
921                try {
922                    List<StreamSource> sources = new ArrayList<StreamSource>();
923                    sources.add(new StreamSource(Thread.currentThread().getContextClassLoader().getResourceAsStream(
924                            "oozie-workflow-0.1.xsd")));
925                    SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
926                    Schema schema = factory.newSchema(sources.toArray(new StreamSource[sources.size()]));
927                    Validator validator = schema.newValidator();
928                    validator.validate(new StreamSource(new FileReader(file)));
929                    System.out.println("Valid worflow-app");
930                }
931                catch (Exception ex) {
932                    throw new OozieCLIException("Invalid workflow-app, " + ex.toString(), ex);
933                }
934            }
935            else {
936                throw new OozieCLIException("File does not exists");
937            }
938        }
