018    package org.apache.oozie.service;
019    
020    import org.apache.commons.logging.LogFactory;
021    import org.apache.hadoop.conf.Configuration;
022    import org.apache.hadoop.util.ReflectionUtils;
023    import org.apache.oozie.client.OozieClient.SYSTEM_MODE;
024    import org.apache.oozie.util.XLog;
025    import org.apache.oozie.util.Instrumentable;
026    import org.apache.oozie.util.IOUtils;
027    import org.apache.oozie.ErrorCode;
028    
029    import java.util.ArrayList;
030    import java.util.Collections;
031    import java.util.LinkedHashMap;
032    import java.util.List;
033    import java.util.Map;
034    import java.io.IOException;
035    import java.io.File;
036    
037    /**
038     * Services is a singleton that manages the lifecycle of all registered {@link Services}. <p/> It has 2 built in
039     * services: {@link XLogService} and {@link ConfigurationService}. <p/> The rest of the services are loaded from the
040     * {@link #CONF_SERVICE_CLASSES} configuration property. The services class names must be separated by commas (spaces
041     * and enters are allowed). <p/> The {@link #CONF_SYSTEM_MODE} configuration property is any of
042     * NORMAL/SAFEMODE/NOWEBSERVICE. <p/> Services are loaded and initialized in the order they are defined in the in
043     * configuration property. <p/> After all services are initialized, if the Instrumentation service is present, all
044     * services that implement the {@link Instrumentable} are instrumented. <p/> Services are destroyed in reverse order.
045     * <p/> If services initialization fail, initialized services are immediatly destroyed.
046     */
047    public class Services {
048        private static final int MAX_SYSTEM_ID_LEN = 10;
049    
050        public static final String CONF_SYSTEM_ID = "oozie.system.id";
051    
052        public static final String CONF_SERVICE_CLASSES = "oozie.services";
053    
054        public static final String CONF_SYSTEM_MODE = "oozie.systemmode";
055    
056        public static final String CONF_DELETE_RUNTIME_DIR = "oozie.delete.runtime.dir.on.shutdown";
057    
058        private static Services SERVICES;
059    
060        private SYSTEM_MODE systemMode;
061        private String runtimeDir;
062        private Configuration conf;
063        private Map<Class<? extends Service>, Service> services = new LinkedHashMap<Class<? extends Service>, Service>();
064        private String systemId;
065    
066        /**
067         * Create a services. <p/> The built in services are initialized.
068         *
069         * @throws ServiceException thrown if any of the built in services could not initialize.
070         */
071        public Services() throws ServiceException {
072            if (SERVICES != null) {
073                XLog log = XLog.getLog(getClass());
074                log.warn(XLog.OPS, "Previous services singleton active, destroying it");
075                SERVICES.destroy();
076                SERVICES = null;
077            }
078            setServiceInternal(XLogService.class, false);
079            setServiceInternal(ConfigurationService.class, true);
080            conf = get(ConfigurationService.class).getConf();
081            systemId = conf.get(CONF_SYSTEM_ID, ("oozie-" + System.getProperty("user.name")));
082            if (systemId.length() > MAX_SYSTEM_ID_LEN) {
083                systemId = systemId.substring(0, MAX_SYSTEM_ID_LEN);
084                XLog.getLog(getClass()).warn("System ID [{0}] exceeds maximun lenght [{1}], trimming", systemId,
085                                             MAX_SYSTEM_ID_LEN);
086            }
087            setSystemMode(SYSTEM_MODE.valueOf(conf.get(CONF_SYSTEM_MODE, SYSTEM_MODE.NORMAL.toString())));
088            runtimeDir = createRuntimeDir();
089        }
090    
091        private String createRuntimeDir() throws ServiceException {
092            try {
093                File file = File.createTempFile(getSystemId(), ".dir");
094                file.delete();
095                if (!file.mkdir()) {
096                    ServiceException ex = new ServiceException(ErrorCode.E0001, file.getAbsolutePath());
097                    XLog.getLog(getClass()).fatal(ex);
098                    throw ex;
099                }
100                XLog.getLog(getClass()).info("Initialized runtime directory [{0}]", file.getAbsolutePath());
101                return file.getAbsolutePath();
102            }
103            catch (IOException ex) {
104                ServiceException sex = new ServiceException(ErrorCode.E0001, ex);
105                XLog.getLog(getClass()).fatal(ex);
106                throw sex;
107            }
108        }
109    
110        /**
111         * Return active system mode. <p/> .
112         *
113         * @return
114         */
115    
116        public SYSTEM_MODE getSystemMode() {
117            return systemMode;
118        }
119    
120        /**
121         * Return the runtime directory of the Oozie instance. <p/> The directory is created under TMP and it is always a
122         * new directory per Services initialization.
123         *
124         * @return the runtime directory of the Oozie instance.
125         */
126        public String getRuntimeDir() {
127            return runtimeDir;
128        }
129    
130        /**
131         * Return the system ID, the value defined in the {@link #CONF_SYSTEM_ID} configuration property.
132         *
133         * @return the system ID, the value defined in the {@link #CONF_SYSTEM_ID} configuration property.
134         */
135        public String getSystemId() {
136            return systemId;
137        }
138    
139        /**
140         * Set and set system mode.
141         *
142         * @param .
143         */
144    
145        public synchronized void setSystemMode(SYSTEM_MODE sysMode) {
146            if (this.systemMode != sysMode) {
147                XLog log = XLog.getLog(getClass());
148                log.info(XLog.OPS, "Exiting " + this.systemMode + " Entering " + sysMode);
149            }
150            this.systemMode = sysMode;
151        }
152    
153        /**
154         * Return the services configuration.
155         *
156         * @return services configuraiton.
157         */
158        public Configuration getConf() {
159            return conf;
160        }
161    
162        /**
163         * Initialize all services define in the {@link #CONF_SERVICE_CLASSES} configuration property.
164         *
165         * @throws ServiceException thrown if any of the services could not initialize.
166         */
167        @SuppressWarnings("unchecked")
168        public void init() throws ServiceException {
169            XLog log = new XLog(LogFactory.getLog(getClass()));
170            log.trace("Initializing");
171            SERVICES = this;
172            try {
173                Class<? extends Service>[] serviceClasses = (Class<? extends Service>[]) conf.getClasses(
174                        CONF_SERVICE_CLASSES);
175                if (serviceClasses != null) {
176                    for (Class<? extends Service> serviceClass : serviceClasses) {
177                        setService(serviceClass);
178                    }
179                }
180            }
181            catch (RuntimeException ex) {
182                XLog.getLog(getClass()).fatal(ex.getMessage(), ex);
183                throw ex;
184            }
185            catch (ServiceException ex) {
186                SERVICES = null;
187                throw ex;
188            }
189            InstrumentationService instrService = get(InstrumentationService.class);
190            if (instrService != null) {
191                for (Service service : services.values()) {
192                    if (service instanceof Instrumentable) {
193                        ((Instrumentable) service).instrument(instrService.get());
194                    }
195                }
196            }
197            log.info("Initialized");
198            log.info("Oozie System ID [{0}] started!", getSystemId());
199        }
200    
201        /**
202         * Destroy all services.
203         */
204        public void destroy() {
205            XLog log = new XLog(LogFactory.getLog(getClass()));
206            log.trace("Shutting down");
207            boolean deleteRuntimeDir = false;
208            if (conf != null) {
209                deleteRuntimeDir = conf.getBoolean(CONF_DELETE_RUNTIME_DIR, false);
210            }
211            if (services != null) {
212                List<Service> list = new ArrayList<Service>(services.values());
213                Collections.reverse(list);
214                for (Service service : list) {
215                    try {
216                        log.trace("Destroying service[{0}]", service.getInterface());
217                        if (service.getInterface() == XLogService.class) {
218                            log.info("Shutdown");
219                        }
220                        service.destroy();
221                    }
222                    catch (Throwable ex) {
223                        log.error("Error destroying service[{0}], {1}", service.getInterface(), ex.getMessage(), ex);
224                    }
225                }
226            }
227            if (deleteRuntimeDir) {
228                try {
229                    IOUtils.delete(new File(runtimeDir));
230                }
231                catch (IOException ex) {
232                    log.error("Error deleting runtime directory [{0}], {1}", runtimeDir, ex.getMessage(), ex);
233                }
234            }
235            services = null;
236            conf = null;
237            SERVICES = null;
238        }
239    
240        /**
241         * Return a service by its public interface.
242         *
243         * @param serviceKlass service public interface.
244         * @return the associated service, or <code>null</code> if not define.
245         */
246        @SuppressWarnings("unchecked")
247        public <T extends Service> T get(Class<T> serviceKlass) {
248            return (T) services.get(serviceKlass);
249        }
250    
251        /**
252         * Set a service programmatically. <p/> The service will be initialized by the services. <p/> If a service is
253         * already defined with the same public interface it will be destroyed.
254         *
255         * @param klass service klass
256         * @throws ServiceException if the service could not be initialized, at this point all services have been
257         * destroyed.
258         */
259        public void setService(Class<? extends Service> klass) throws ServiceException {
260            setServiceInternal(klass, true);
261        }
262    
263        private void setServiceInternal(Class<? extends Service> klass, boolean logging) throws ServiceException {
264            try {
265                Service newService = (Service) ReflectionUtils.newInstance(klass, null);
266                Service oldService = services.get(newService.getInterface());
267                if (oldService != null) {
268                    oldService.destroy();
269                }
270                if (logging) {
271                    XLog log = new XLog(LogFactory.getLog(getClass()));
272                    log.trace("Initializing service[{0}] class[{1}]", newService.getInterface(), newService.getClass());
273                }
274                newService.init(this);
275                services.put(newService.getInterface(), newService);
276            }
277            catch (ServiceException ex) {
278                XLog.getLog(getClass()).fatal(ex.getMessage(), ex);
279                destroy();
280                throw ex;
281            }
282        }
283    
284        /**
285         * Return the services singleton.
286         *
287         * @return services singleton, <code>null</code> if not initialized.
288         */
289        public static Services get() {
290            return SERVICES;
291        }
