018    package org.apache.oozie.service;
019    
020    import org.apache.hadoop.conf.Configuration;
021    import org.apache.oozie.command.coord.CoordPurgeCommand;
022    import org.apache.oozie.command.wf.PurgeCommand;
023    import org.apache.oozie.service.CallableQueueService;
024    import org.apache.oozie.service.SchedulerService;
025    import org.apache.oozie.service.Service;
026    import org.apache.oozie.service.Services;
027    
028    /**
029     * The PurgeService schedules purging of completed jobs and associated action older than a specified age.
030     */
031    public class PurgeService implements Service {
032    
033        public static final String CONF_PREFIX = Service.CONF_PREFIX + "PurgeService.";
034        /**
035         * Age of completed jobs to be deleted, in days.
036         */
037        public static final String CONF_OLDER_THAN = CONF_PREFIX + "older.than";
038        public static final String COORD_CONF_OLDER_THAN = CONF_PREFIX + "coord.older.than";
039        /**
040         * Time interval, in seconds, at which the purge jobs service will be scheduled to run.
041         */
042        public static final String CONF_PURGE_INTERVAL = CONF_PREFIX + "purge.interval";
043        private static final String COORD_PURGE_LIMIT = CONF_PREFIX + "coord.purge.limit";
044    
045        /**
046         * PurgeRunnable is the runnable which is scheduled to run at the configured interval. PurgeCommand is queued to
047         * remove completed jobs and associated actions older than the configured age.
048         */
049        static class PurgeRunnable implements Runnable {
050            private int olderThan;
051            private int coordOlderThan;
052            private int limit;
053    
054            public PurgeRunnable(int olderThan, int coordOlderThan, int limit) {
055                this.olderThan = olderThan;
056                this.coordOlderThan = coordOlderThan;
057                this.limit = limit;
058            }
059    
060            public void run() {
061                Services.get().get(CallableQueueService.class).queue(new PurgeCommand(olderThan));
062                Services.get().get(CallableQueueService.class).queue(new CoordPurgeCommand(coordOlderThan, limit));
063            }
064    
065        }
066    
067        /**
068         * Initializes the {@link PurgeService}.
069         *
070         * @param services services instance.
071         */
072        @Override
073        public void init(Services services) {
074            Configuration conf = services.getConf();
075            Runnable purgeJobsRunnable = new PurgeRunnable(conf.getInt(
076                    CONF_OLDER_THAN, 30), conf.getInt(COORD_CONF_OLDER_THAN, 7),
077                                          conf.getInt(COORD_PURGE_LIMIT, 100));
078            services.get(SchedulerService.class).schedule(purgeJobsRunnable, 10, conf.getInt(CONF_PURGE_INTERVAL, 3600),
079                                                          SchedulerService.Unit.SEC);
080        }
081    
082        /**
083         * Destroy the Purge Jobs Service.
084         */
085        @Override
086        public void destroy() {
087        }
088    
089        /**
090         * Return the public interface for the purge jobs service.
091         *
092         * @return {@link PurgeService}.
093         */
094        @Override
095        public Class<? extends Service> getInterface() {
096            return PurgeService.class;
097        }
098    }
