060        public static final String CONF_PREFIX = Service.CONF_PREFIX + "StoreService.";
061        private static final String INTSRUMENTATION_GROUP = "jdbc";
062        public static final String CONF_URL = CONF_PREFIX + "jdbc.url";
063        public static final String CONF_DRIVER = CONF_PREFIX + "jdbc.driver";
064        ;
065        public static final String CONF_USERNAME = CONF_PREFIX + "jdbc.username";
066        public static final String CONF_PASSWORD = CONF_PREFIX + "jdbc.password";

068    
069        @SuppressWarnings("unchecked")
070        private static EntityManagerFactory factory = Persistence.createEntityManagerFactory("oozie",
071                                                                                             new java.util.HashMap());
072    
073        /**
074         * Return instance of store.
075         *
076         * @return {@link Store}.
077         */
078        @SuppressWarnings("unchecked")
079        public <S extends Store> S getStore(Class<S> klass) throws StoreException {
080            if (WorkflowStore.class.equals(klass)) {
081                return (S) Services.get().get(WorkflowStoreService.class).create();
082            }
083            else {
084                if (CoordinatorStore.class.equals(klass)) {
085                    return (S) Services.get().get(CoordinatorStoreService.class).create();
086                }
087                else {
088                    if (SLAStore.class.equals(klass)) {
089                        return (S) Services.get().get(SLAStoreService.class).create();
090                    }
091                }
092            }
093            // to do add checks for other stores - coordinator and SLA stores
094            throw new StoreException(ErrorCode.E0607, " can not get store StoreService.getStore(Class)");
095        }
096    
097        /**
098         * Return instance of store with an EntityManager pointing to an existing Store.
099         *
100         * @return {@link Store}.
101         */
102        @SuppressWarnings("unchecked")
103        public <S extends Store, T extends Store> S getStore(Class<S> klass, T store) throws StoreException {
104            if (WorkflowStore.class.equals(klass)) {
105                return (S) Services.get().get(WorkflowStoreService.class).create(store);
106            }
107            else {
108                if (CoordinatorStore.class.equals(klass)) {
109                    return (S) Services.get().get(CoordinatorStoreService.class).create(store);
110                }
111                else {
112                    if (SLAStore.class.equals(klass)) {
113                        return (S) Services.get().get(SLAStoreService.class).create(store);
114                    }
115                }
116            }
117            throw new StoreException(ErrorCode.E0607, " StoreService.getStore(Class, store)");
118        }

127        }

129        /**
130         * Initializes the {@link StoreService}.
131         *
132         * @param services services instance.
133         */
134        public void init(Services services) throws ServiceException {
135            Configuration conf = services.getConf();
136            Properties props = new Properties();
137            props.put("url", conf.get(CONF_URL, "jdbc:hsqldb:mem:testdb"));
138            props.put("driverClassName", conf.get(CONF_DRIVER, "org.hsqldb.jdbcDriver"));
139            props.put("url", conf.get(CONF_URL, "jdbc:hsqldb:mem:testdb"));
140            props.put("username", conf.get(CONF_USERNAME, "sa"));
141            props.put("password", conf.get(CONF_PASSWORD, "").trim());
142            props.put("maxActive", conf.get(CONF_MAX_ACTIVE_CONN, "10"));
143            EntityManager entityManager = getEntityManager();
144            entityManager.find(WorkflowActionBean.class, 1);
145            entityManager.find(WorkflowJobBean.class, 1);
146            entityManager.find(CoordinatorActionBean.class, 1);
147            entityManager.find(CoordinatorJobBean.class, 1);
148            entityManager.find(JsonWorkflowAction.class, 1);
149            entityManager.find(JsonWorkflowJob.class, 1);
150            entityManager.find(JsonCoordinatorAction.class, 1);
151            entityManager.find(JsonCoordinatorJob.class, 1);
152            entityManager.find(SLAEventBean.class, 1);
153            entityManager.find(JsonSLAEvent.class, 1);
154            XLog.getLog(getClass()).info(XLog.STD, "*** StoreService *** " + "Initialized all entities!");
155            // need to use a pseudo no-op transaction so all entities, datasource
156            // and connection pool are initialized
157            // one time only
158            entityManager.getTransaction().begin();
159            OpenJPAEntityManagerFactorySPI spi = (OpenJPAEntityManagerFactorySPI) factory;
160            XLog.getLog(getClass()).warn("StoreService initialized *** {0}",
161                                         spi.getConfiguration().getConnectionProperties());
162            entityManager.getTransaction().commit();
163            entityManager.close();
164        }

190        /**
191         * for unit test only Return a raw direct JDBC connection. <p/> The connection is not from the connection pool. <p/>
192         * The conection is not instrumented.
193         *
194         * @return a raw Direct JDBC connection.
195         * @throws SQLException thrown if the connection could not be obtained.
196         */
197        public Connection getRawConnection() throws SQLException {
198            String driver = Services.get().getConf().get(CONF_DRIVER, "org.hsqldb.jdbcDriver");
199            String url = Services.get().getConf().get(CONF_URL, "jdbc:hsqldb:mem:testdb");
200            String user = Services.get().getConf().get(CONF_USERNAME, "sa");
201            String password = Services.get().getConf().get(CONF_PASSWORD, "").trim();
202            try {
203                Class.forName(driver);
204            }
205            catch (ClassNotFoundException ex) {
206                throw new RuntimeException(ex);
207            }
208            return DriverManager.getConnection(url, user, password);
209        }
