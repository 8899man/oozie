049    /**
050     * The Recovery Service checks for pending actions and premater coordinator jobs older than a configured age and then
051     * queues them for execution.
052     */
053    public class RecoveryService implements Service {
054    
055        public static final String CONF_PREFIX = Service.CONF_PREFIX + "RecoveryService.";
056        public static final String CONF_PREFIX_WF_ACTIONS = Service.CONF_PREFIX + "wf.actions.";
057        public static final String CONF_PREFIX_COORD = Service.CONF_PREFIX + "coord.";
058        /**
059         * Time interval, in seconds, at which the recovery service will be scheduled to run.
060         */
061        public static final String CONF_SERVICE_INTERVAL = CONF_PREFIX + "interval";
062        /**
063         * The number of callables to be queued in a batch.
064         */
065        public static final String CONF_CALLABLE_BATCH_SIZE = CONF_PREFIX + "callable.batch.size";
066        /**
067         * Age of actions to queue, in seconds.
068         */
069        public static final String CONF_WF_ACTIONS_OLDER_THAN = CONF_PREFIX_WF_ACTIONS + "older.than";
070        /**
071         * Age of coordinator jobs to recover, in seconds.
072         */
073        public static final String CONF_COORD_OLDER_THAN = CONF_PREFIX_COORD + "older.than";
074    
075        private static final String INSTRUMENTATION_GROUP = "recovery";
076        private static final String INSTR_RECOVERED_ACTIONS_COUNTER = "actions";
077        private static final String INSTR_RECOVERED_COORD_JOBS_COUNTER = "coord_jobs";
078        private static final String INSTR_RECOVERED_COORD_ACTIONS_COUNTER = "coord_actions";
079    
080        /**
081         * RecoveryRunnable is the Runnable which is scheduled to run with the configured interval, and takes care of the
082         * queuing of commands.
083         */
084        static class RecoveryRunnable<S extends Store> implements Runnable {
085            private long olderThan;
086            private long coordOlderThan;
087            private long delay = 0;
088            private List<XCallable<Void>> callables;
089            private List<XCallable<Void>> delayedCallables;
090            private StringBuilder msg = null;
091    
092            public RecoveryRunnable(long olderThan, long coordOlderThan) {
093                this.olderThan = olderThan;
094                this.coordOlderThan = coordOlderThan;
095            }
096    
097            public void run() {
098                XLog.Info.get().clear();
099                XLog log = XLog.getLog(getClass());
100                msg = new StringBuilder();
101                runWFRecovery();
102                runCoordJobRecovery();
103                runCoordActionRecovery();
104                runCoordActionRecoveryForReady();
105                log.debug("QUEUING [{0}] for potential recovery", msg.toString());
106                boolean ret = false;
107                if (null != callables) {
108                    ret = Services.get().get(CallableQueueService.class).queueSerial(callables);
109                    if (ret == false) {
110                        log.warn("Unable to queue the callables commands for RecoveryService. "
111                                + "Most possibly command queue is full. Queue size is :"
112                                + Services.get().get(CallableQueueService.class).queueSize());
113                    }
114                    callables = null;
115                }
116                if (null != delayedCallables) {
117                    ret = Services.get().get(CallableQueueService.class).queueSerial(delayedCallables, this.delay);
118                    if (ret == false) {
119                        log.warn("Unable to queue the delayedCallables commands for RecoveryService. "
120                                + "Most possibly delayedQueue is full. DelayedQueue size is :"
121                                + Services.get().get(CallableQueueService.class).delayedQueueSize());
122                    }
123                    delayedCallables = null;
124                    this.delay = 0;
125                }
126            }
127    
128            /**
129             * Recover coordinator jobs that are running and have lastModifiedTimestamp older than the specified interval
130             */
131            private void runCoordJobRecovery() {
132                XLog.Info.get().clear();
133                XLog log = XLog.getLog(getClass());
134    
135                CoordinatorStore store = null;
136                try {
137                    store = Services.get().get(StoreService.class).getStore(CoordinatorStore.class);
138                    store.beginTrx();
139    
140                    // get list of all jobs that have lastModifiedTimestamp older
141                    // than the specified interval
142                    List<CoordinatorJobBean> jobs = store.getCoordinatorJobsOlderThanStatus(coordOlderThan,
143                                                                                            CoordinatorJob.Status.PREMATER.toString(), 50, false);
144                    //log.debug("QUEUING[{0}] PREMATER coord jobs for potential recovery", jobs.size());
145                    msg.append(", COORD_JOBS : " + jobs.size());
146                    for (CoordinatorJobBean coordJob : jobs) {
147                        Services.get().get(InstrumentationService.class).get().incr(INSTRUMENTATION_GROUP,
148                                                                                    INSTR_RECOVERED_COORD_JOBS_COUNTER, 1);
149                        queueCallable(new CoordRecoveryCommand(coordJob.getId()));
150                    }
151    
152                    store.commitTrx();
153                }
154                catch (StoreException ex) {
155                    if (store != null) {
156                        store.rollbackTrx();
157                    }
158                    log.warn("Exception while accessing the store", ex);
159                }
160                finally {
161                    try {
162                        if (store != null) {
163                            store.closeTrx();
164                        }
165                    }
166                    catch (RuntimeException rex) {
167                        log.warn("Exception while attempting to close store", rex);
168                    }
169                }
170            }
171    
172            /**
173             * Recover coordinator actions that are staying in WAITING or SUBMITTED too long
174             */
175            private void runCoordActionRecovery() {
176                XLog.Info.get().clear();
177                XLog log = XLog.getLog(getClass());
178    
179                CoordinatorStore store = null;
180                try {
181                    store = Services.get().get(StoreService.class).getStore(CoordinatorStore.class);
182                    store.beginTrx();
183    
184                    List<CoordinatorActionBean> cactions = store.getRecoveryActionsOlderThan(coordOlderThan, false);
185                    //log.debug("QUEUING[{0}] WAITING and SUBMITTED coord actions for potential recovery", cactions.size());
186                    msg.append(", COORD_ACTIONS : " + cactions.size());
187                    for (CoordinatorActionBean caction : cactions) {
188                        Services.get().get(InstrumentationService.class).get().incr(INSTRUMENTATION_GROUP,
189                                                                                    INSTR_RECOVERED_COORD_ACTIONS_COUNTER, 1);
190                        if (caction.getStatus() == CoordinatorActionBean.Status.WAITING) {
191                            queueCallable(new CoordActionInputCheckCommand(caction.getId()));
192                            log.info("Recover a WAITTING coord action :" + caction.getId());
193                        }
194                        else {
195                            if (caction.getStatus() == CoordinatorActionBean.Status.SUBMITTED) {
196                                CoordinatorJobBean coordJob = store.getCoordinatorJob(caction.getJobId(), false);
197                                queueCallable(new CoordActionStartCommand(caction.getId(), coordJob.getUser(), coordJob
198                                        .getAuthToken()));
199                                log.info("Recover a SUBMITTED coord action :" + caction.getId());
200                            }
201                        }
202                    }
203                    store.commitTrx();
204                }
205                catch (StoreException ex) {
206                    if (store != null) {
207                        store.rollbackTrx();
208                    }
209                    log.warn("Exception while accessing the store", ex);
210                }
211                finally {
212                    try {
213                        if (store != null) {
214                            store.closeTrx();
215                        }
216                    }
217                    catch (RuntimeException rex) {
218                        log.warn("Exception while attempting to close store", rex);
219                    }
220                }
221            }
222    
223            /**
224             * Recover coordinator actions that are staying in READY too long
225             */
226            private void runCoordActionRecoveryForReady() {
227                XLog.Info.get().clear();
228                XLog log = XLog.getLog(getClass());
229    
230                CoordinatorStore store = null;
231                try {
232                    store = Services.get().get(StoreService.class).getStore(CoordinatorStore.class);
233                    store.beginTrx();
234                    List<String> jobids = store.getRecoveryActionsGroupByJobId(coordOlderThan);
235                    //log.debug("QUEUING[{0}] READY coord jobs for potential recovery", jobids.size());
236                    msg.append(", COORD_READY_JOBS : " + jobids.size());
237                    for (String jobid : jobids) {
238                        queueCallable(new CoordActionReadyCommand(jobid));
239                        log.info("Recover READY coord actions for jobid :" + jobid);
240                    }
241                    store.commitTrx();
242                }
243                catch (StoreException ex) {
244                    if (store != null) {
245                        store.rollbackTrx();
246                    }
247                    log.warn("Exception while accessing the store", ex);
248                }
249                finally {
250                    try {
251                        if (store != null) {
252                            store.closeTrx();
253                        }
254                    }
255                    catch (RuntimeException rex) {
256                        log.warn("Exception while attempting to close store", rex);
257                    }
258                }
259            }

261            /**
262             * Recover wf actions
263             */
264            private void runWFRecovery() {
265                XLog.Info.get().clear();
266                XLog log = XLog.getLog(getClass());
267                // queue command for action recovery
268                WorkflowStore store = null;
269                try {
270                    store = Services.get().get(StoreService.class).getStore(WorkflowStore.class);
271                    store.beginTrx();
272                    List<WorkflowActionBean> actions = null;
273                    try {
274                        actions = store.getPendingActions(olderThan);
275                    }
276                    catch (StoreException ex) {
277                        log.warn("Exception while reading pending actions from storage", ex);
278                    }
279                    //log.debug("QUEUING[{0}] pending wf actions for potential recovery", actions.size());
280                    msg.append(" WF_ACTIONS " + actions.size());
281    
282                    for (WorkflowActionBean action : actions) {
283                        Services.get().get(InstrumentationService.class).get().incr(INSTRUMENTATION_GROUP,
284                                                                                    INSTR_RECOVERED_ACTIONS_COUNTER, 1);
285                        if (action.getStatus() == WorkflowActionBean.Status.PREP
286                                || action.getStatus() == WorkflowActionBean.Status.START_MANUAL) {
287                            queueCallable(new ActionStartCommand(action.getId(), action.getType()));
288                        }
289                        else {
290                            if (action.getStatus() == WorkflowActionBean.Status.START_RETRY) {
291                                Date nextRunTime = action.getPendingAge();
292                                queueCallable(new ActionStartCommand(action.getId(), action.getType()), nextRunTime.getTime()
293                                        - System.currentTimeMillis());
294                            }
295                            else {
296                                if (action.getStatus() == WorkflowActionBean.Status.DONE
297                                        || action.getStatus() == WorkflowActionBean.Status.END_MANUAL) {
298                                    queueCallable(new ActionEndCommand(action.getId(), action.getType()));
299                                }
300                                else {
301                                    if (action.getStatus() == WorkflowActionBean.Status.END_RETRY) {
302                                        Date nextRunTime = action.getPendingAge();
303                                        queueCallable(new ActionEndCommand(action.getId(), action.getType()), nextRunTime.getTime()
304                                                - System.currentTimeMillis());
305                                    }
306                                    else {
307                                        if (action.getStatus() == WorkflowActionBean.Status.OK
308                                                || action.getStatus() == WorkflowActionBean.Status.ERROR) {
309                                            queueCallable(new SignalCommand(action.getJobId(), action.getId()));
310                                        }
311                                    }
312                                }
313                            }
314                        }
315                    }
316                    store.commitTrx();
317                }
318                catch (StoreException ex) {
319                    if (store != null) {
320                        store.rollbackTrx();
321                    }
322                    log.warn("Exception while getting store to get pending actions", ex);
323                }
324                finally {
325                    try {
326                        if (store != null) {
327                            store.closeTrx();
328                        }
329                    }
330                    catch (RuntimeException rex) {
331                        log.warn("Exception while attempting to close store", rex);
332                    }
333                }
334            }
335    
336            /**
337             * Adds callables to a list. If the number of callables in the list reaches {@link
338             * RecoveryService#CONF_CALLABLE_BATCH_SIZE}, the entire batch is queued and the callables list is reset.
339             *
340             * @param callable the callable to queue.
341             */
342            private void queueCallable(XCallable<Void> callable) {
343                if (callables == null) {
344                    callables = new ArrayList<XCallable<Void>>();
345                }
346                callables.add(callable);
347                if (callables.size() == Services.get().getConf().getInt(CONF_CALLABLE_BATCH_SIZE, 10)) {
348                    boolean ret = Services.get().get(CallableQueueService.class).queueSerial(callables);
349                    if (ret == false) {
350                        XLog.getLog(getClass()).warn(
351                                "Unable to queue the callables commands for RecoveryService. "
352                                        + "Most possibly command queue is full. Queue size is :"
353                                        + Services.get().get(CallableQueueService.class).queueSize());
354                    }
355                    callables = new ArrayList<XCallable<Void>>();
356                }
357            }
358    
359            /**
360             * Adds callables to a list. If the number of callables in the list reaches {@link
361             * RecoveryService#CONF_CALLABLE_BATCH_SIZE}, the entire batch is queued with the delay set to the maximum delay
362             * of the callables in the list. The callables list and the delay is reset.
363             *
364             * @param callable the callable to queue.
365             * @param delay the delay for the callable.
366             */
367            private void queueCallable(XCallable<Void> callable, long delay) {
368                if (delayedCallables == null) {
369                    delayedCallables = new ArrayList<XCallable<Void>>();
370                }
371                this.delay = Math.max(this.delay, delay);
372                delayedCallables.add(callable);
373                if (delayedCallables.size() == Services.get().getConf().getInt(CONF_CALLABLE_BATCH_SIZE, 10)) {
374                    boolean ret = Services.get().get(CallableQueueService.class).queueSerial(delayedCallables, this.delay);
375                    if (ret == false) {
376                        XLog.getLog(getClass()).warn(
377                                "Unable to queue the delayedCallables commands for RecoveryService. "
378                                        + "Most possibly delayedQueue is full. DelayedQueue size is :"
379                                        + Services.get().get(CallableQueueService.class).delayedQueueSize());
380                    }
381                    delayedCallables = new ArrayList<XCallable<Void>>();
382                    this.delay = 0;
383                }
384            }
385        }

387        /**
388         * Initializes the RecoveryService.
389         *
390         * @param services services instance.
391         */
392        @Override
393        public void init(Services services) {
394            Configuration conf = services.getConf();
395            Runnable recoveryRunnable = new RecoveryRunnable(conf.getInt(CONF_WF_ACTIONS_OLDER_THAN, 120), conf.getInt(
396                    CONF_COORD_OLDER_THAN, 600));
397            services.get(SchedulerService.class).schedule(recoveryRunnable, 10, conf.getInt(CONF_SERVICE_INTERVAL, 600),
398                                                          SchedulerService.Unit.SEC);
399        }
