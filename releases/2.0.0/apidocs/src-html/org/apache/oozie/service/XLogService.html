044    /**
045     * Built in service that initializes and manages the log4j. <p/> The log4j configuration file to use is read from the
046     * system property {@link #LOG4J_FILE}, if the system propery is not set the default value is {@link
047     * #DEFAULT_LOG4J_PROPERTIES}. The log4j configuration file can be a Java Properties file (.properties) or an XML file
048     * (.xml). <p/> If the system property {@link ConfigurationService#CONFIG_PATH} is set, the log4j configuration file is
049     * read from that directory. Otherwise is read from the classpath root. <p/> The reload interval of the log4j
050     * configuration is set by the system property {@link #RELOAD_INTERVAL}, the value is in seconds, the default value is
051     * {@link #DEFAULT_RELOAD_INTERVAL}. <p/> The log4j configuration is reloaded only when read from a configuration
052     * directory, when read from the classpath reloading is not in effect.
053     */
054    public class XLogService implements Service, Instrumentable {
055        private static final String INSTRUMENTATION_GROUP = "logging";
056    
057        /**
058         * System property that indicates the log4j configuration file to load.
059         */
060        public static final String LOG4J_FILE = "oozie.log4j.file";
061    
062        /**
063         * System property that indicates the reload interval of the configuration file.
064         */
065        public static final String RELOAD_INTERVAL = "oozie.log4j.reload";
066    
067        /**
068         * Default value for the log4j configuration file if {@link #LOG4J_FILE} is not set.
069         */
070        public static final String DEFAULT_LOG4J_PROPERTIES = "oozie-log4j.properties";
071    
072        /**
073         * Default value for the reload interval if {@link #RELOAD_INTERVAL} is not set.
074         */
075        public static final String DEFAULT_RELOAD_INTERVAL = "10";
076    
077        // for testing purposes
078        static boolean testingDefaultFile;
079    
080        private long interval;
081        private boolean fromClasspath;
082        private String configFile;
083    
084        private static final String STARTUP_MESSAGE = "{E}"
085                + " ******************************************************************************* {E}"
086                + "  STARTUP MSG: Oozie BUILD_VERSION [{0}] compiled by [{1}] on [{2}]{E}"
087                + "  STARTUP MSG:       revision [{3}]@[{4}]{E}"
088                + "*******************************************************************************";
089    
090        private String oozieLogPath;
091        private String oozieLogName;
092        private int oozieLogRotation = -1;
093    
094        public XLogService() {
095            testingDefaultFile = false;
096        }
097    
098        /**
099         * Initialize the log service.
100         *
101         * @param services services instance.
102         * @throws ServiceException thrown if the log service could not be initialized.
103         */
104        public void init(Services services) throws ServiceException {
105            try {
106                LogManager.resetConfiguration();
107                String defaultConfig = (testingDefaultFile) ? "test-" + DEFAULT_LOG4J_PROPERTIES : DEFAULT_LOG4J_PROPERTIES;
108                String log4jPathOri = System.getProperty(LOG4J_FILE, defaultConfig);
109    
110                String configPath = System.getProperty(ConfigurationService.CONFIG_PATH);
111    
112                fromClasspath = configureLog4J(configPath, log4jPathOri);
113    
114                XLog log = new XLog(LogFactory.getLog(getClass()));
115    
116                String from = (fromClasspath) ? "classpath" : "path";
117                long interval = Long.parseLong(System.getProperty(RELOAD_INTERVAL, DEFAULT_RELOAD_INTERVAL));
118                String reload = (fromClasspath) ? "disabled" : Long.toString(interval) + " sec";
119                log.info(XLog.OPS, STARTUP_MESSAGE, BuildInfo.getBuildInfo().getProperty(BuildInfo.BUILD_VERSION),
120                         BuildInfo.getBuildInfo().getProperty(BuildInfo.BUILD_USER_NAME), BuildInfo.getBuildInfo()
121                        .getProperty(BuildInfo.BUILD_TIME), BuildInfo.getBuildInfo().getProperty(
122                        BuildInfo.BUILD_SVN_REVISION), BuildInfo.getBuildInfo()
123                        .getProperty(BuildInfo.BUILD_SVN_URL));
124    
125                log.info("Log4j configuration loaded from [{0}]", from);
126                log.info("Log4j configuration file [{0}]", configFile);
127                log.info("Log4j reload interval [{0}]", reload);
128                if (log.isDebugEnabled()) {
129                    ClassLoader cl = Thread.currentThread().getContextClassLoader();
130                    InputStream is = (fromClasspath) ? cl.getResourceAsStream(configFile) : new FileInputStream(configFile);
131                    BufferedReader br = new BufferedReader(new InputStreamReader(is));
132                    StringBuffer sb = new StringBuffer();
133                    String line;
134                    while ((line = br.readLine()) != null) {
135                        sb.append("   ").append(line).append('\n');
136                    }
137                    br.close();
138                    log.trace("Log4j configuration:{E}----{E}{0}----{E}", sb);
139                }
140    
141                // Getting configuration for oozie log via WS
142                ClassLoader cl = Thread.currentThread().getContextClassLoader();
143                InputStream is = (fromClasspath) ? cl.getResourceAsStream(configFile) : new FileInputStream(configFile);
144                Properties props = new Properties();
145                props.load(is);
146                Configuration conf = new XConfiguration();
147                for (Map.Entry entry : props.entrySet()) {
148                    conf.set((String) entry.getKey(), (String) entry.getValue());
149                }
150                String logFile = conf.get("log4j.appender.oozie.File");
151                if (logFile == null) {
152                    log.warn("Oozie log via WS not configured properly, missing property [{0}]",
153                             "log4j.appender.oozie.File");
154                }
155                else {
156                    logFile = logFile.trim();
157                    int i = logFile.lastIndexOf("/");
158                    if (i == -1) {
159                        log.warn("Oozie apps log via WS not configured properly, log file is not an absolute path [{0}]",
160                                 logFile);
161                    }
162                    else {
163                        String pattern = conf.get("log4j.appender.oozie.DatePattern");
164                        if (pattern == null) {
165                            log.warn("Oozie apps log via WS not configured properly, missing property [{0}]",
166                                     "log4j.appender.oozie.DatePattern");
167                        }
168                        else {
169                            pattern = pattern.trim();
170                            if (pattern.endsWith("HH")) {
171                                oozieLogRotation = 60 * 60;
172                            }
173                            else {
174                                if (pattern.endsWith("dd")) {
175                                    oozieLogRotation = 60 * 60 * 24;
176                                }
177                                else {
178                                    log.warn("Oozie apps log via WS not configured properly, invalid DatePatter [{0}], "
179                                            + "it should end with 'HH' or 'dd'", pattern);
180                                }
181                            }
182                            if (oozieLogRotation > 0) {
183                                oozieLogPath = logFile.substring(0, i);
184                                oozieLogName = logFile.substring(i + 1);
185                            }
186                        }
187                    }
188                }
189            }
190            catch (IOException ex) {
191                throw new ServiceException(ErrorCode.E0010, ex.getMessage(), ex);
192            }
193            XLog.Info.reset();
194            XLog.Info.defineParameter(USER);
195            XLog.Info.defineParameter(GROUP);
196            XLogStreamer.Filter.reset();
197            XLogStreamer.Filter.defineParameter(USER);
198            XLogStreamer.Filter.defineParameter(GROUP);
199        }
200    
201        /**
202         * Destroy the log service.
203         */
204        public void destroy() {
205            LogManager.shutdown();
206            XLog.Info.reset();
207            XLogStreamer.Filter.reset();
208        }

229        private boolean configureLog4J(String configPath, String log4jFile) throws IOException {
230            boolean fromClasspath;
231            if (configPath != null) {
232                interval = Long.parseLong(System.getProperty(RELOAD_INTERVAL, DEFAULT_RELOAD_INTERVAL));
233                File file = new File(configPath, log4jFile);
234                if (!file.exists()) {
235                    throw new RuntimeException(XLog.format("Log4j configuration [{0}] not found in path [{1}]", log4jFile,
236                                                           configPath));
237                }
238                log4jFile = file.getAbsolutePath();
239                if (log4jFile.endsWith(".properties")) {
240                    PropertyConfigurator.configureAndWatch(log4jFile, interval * 1000);
241                }
242                else {
243                    throw new RuntimeException(XLog.format("Log4j configuration [{0}] must be a '.properties' file",
244                                                           log4jFile));
245                }
246                configFile = log4jFile;
247                fromClasspath = false;
248            }
249            else {
250                ClassLoader cl = Thread.currentThread().getContextClassLoader();
251                URL log4jUrl = cl.getResource(log4jFile);
252                if (log4jUrl != null) {
253                    if (log4jFile.endsWith(".properties")) {
254                        PropertyConfigurator.configure(log4jUrl);
255                    }
256                    else {
257                        throw new RuntimeException(XLog.format("Log4j configuration [{0}] must be a '.properties' file",
258                                                               log4jFile));
259                    }
260                }
261                else {
262                    throw new RuntimeException(XLog.format("Log4j configuration [{0}] not found in classpath", log4jFile));
263                }
264                configFile = log4jFile;
265                fromClasspath = true;
266            }
267            return fromClasspath;
268        }

270        /**
271         * Instruments the log service. <p/> It sets instrumentation variables indicating the config file, reload interval
272         * and if loaded from the classpath.
273         *
274         * @param instr instrumentation to use.
275         */
276        public void instrument(Instrumentation instr) {
277            instr.addVariable("oozie", "version", new Instrumentation.Variable<String>() {
278                public String getValue() {
279                    return BuildInfo.getBuildInfo().getProperty(BuildInfo.BUILD_VERSION);
280                }
281            });
282            instr.addVariable(INSTRUMENTATION_GROUP, "config.file", new Instrumentation.Variable<String>() {
283                public String getValue() {
284                    return configFile;
285                }
286            });
287            instr.addVariable(INSTRUMENTATION_GROUP, "reload.interval", new Instrumentation.Variable<Long>() {
288                public Long getValue() {
289                    return interval;
290                }
291            });
292            instr.addVariable(INSTRUMENTATION_GROUP, "from.classpath", new Instrumentation.Variable<Boolean>() {
293                public Boolean getValue() {
294                    return fromClasspath;
295                }
296            });
297        }

299        /**
300         * Stream the log of a job.
301         *
302         * @param filter log streamer filter.
303         * @param startTime start time for log events to filter.
304         * @param endTime end time for log events to filter.
305         * @param writer writer to stream the log to.
306         * @throws IOException thrown if the log cannot be streamed.
307         */
308        public void streamLog(XLogStreamer.Filter filter, Date startTime, Date endTime, Writer writer) throws IOException {
309            if (oozieLogPath != null) {
310                new XLogStreamer(filter, writer, oozieLogPath, oozieLogName, oozieLogRotation)
311                        .streamLog(startTime, endTime);
312            }
313            else {
314                writer.write("Log streaming disabled!!");
315            }
316    
317        }
