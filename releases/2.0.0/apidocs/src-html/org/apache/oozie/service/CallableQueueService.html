039    /**
040     * The callable queue service queues {@link XCallable}s for asynchronous execution. <p/> Callables can be queued for
041     * immediate execution or for delayed execution (some time in the future). <p/> Callables are consumed from the queue
042     * for execution based on their priority. <p/> When the queues (for immediate execution and for delayed execution) are
043     * full, teh callable queue service stops queuing callables. <p/> A threadpool is used to execute the callables
044     * asynchronously. <p/> The following configuration parameters control the callable queue service: <p/> {@link
045     * #CONF_QUEUE_SIZE} size of the immmediate execution queue. Defaulf value is 1000. <p/> {@link
046     * #CONF_DELAYED_QUEUE_SIZE} size of the delayed execution queue. Defaulf value is 1000. <p/> {@link #CONF_THREADS}
047     * number of threads in the threadpool used for asynchronous command execution. When this number of threads is reached,
048     * commands remain the queue until threads become available.
049     *
050     * Sets up a priority queue for the execution of Commands via a ThreadPool. Sets up a Delyaed Queue to handle actions
051     * which will be ready for execution sometime in the future.
052     */
053    public class CallableQueueService implements Service, Instrumentable {
054        private static final String INSTRUMENTATION_GROUP = "callablequeue";
055        private static final String INSTR_IN_QUEUE_TIME_TIMER = "time.in.queue";
056        private static final String INSTR_EXECUTED_COUNTER = "executed";
057        private static final String INSTR_FAILED_COUNTER = "failed";
058        private static final String INSTR_QUEUED_COUNTER = "queued";
059        private static final String INSTR_DELAYD_QUEUED_COUNTER = "delayed.queued";
060        private static final String INSTR_QUEUE_SIZE_SAMPLER = "queue.size";
061        private static final String INSTR_DELAYED_QUEUE_SIZE_SAMPLER = "delayed.queue.size";
062        private static final String INSTR_THREADS_ACTIVE_SAMPLER = "threads.active";
063    
064        public static final String CONF_PREFIX = Service.CONF_PREFIX + "CallableQueueService.";
065    
066        public static final String CONF_QUEUE_SIZE = CONF_PREFIX + "queue.size";
067        public static final String CONF_DELAYED_QUEUE_SIZE = CONF_PREFIX + "delayed.queue.size";
068        public static final String CONF_THREADS = CONF_PREFIX + "threads";
069        public static final String CONF_CALLABLE_CONCURRENCY = CONF_PREFIX + "callable.concurrency";

071        public static final int CONCURRENCY_DELAY = 500;
072    
073        public static final int SAFE_MODE_DELAY = 60000;

075        private Map<String, AtomicInteger> activeCallables;
076        private int maxCallableConcurrency;
077    
078        private synchronized boolean callableBegin(XCallable callable) {
079            AtomicInteger counter = activeCallables.get(callable.getType());
080            if (counter == null) {
081                counter = new AtomicInteger(1);
082                activeCallables.put(callable.getType(), counter);
083                return true;
084            }
085            else {
086                return counter.incrementAndGet() <= maxCallableConcurrency;
087            }
088        }
089    
090        private synchronized void callableEnd(XCallable callable) {
091            AtomicInteger counter = activeCallables.get(callable.getType());
092            if (counter == null) {
093                throw new IllegalStateException("It should not happen");
094            }
095            else {
096                counter.decrementAndGet();
097            }
098        }

100        // Callables are wrapped with the this wrapper for execution, for logging and instrumentation.
101        // The wrapper implements Runnable and Comparable to be able to work with an executor and a priority queue.
102        class CallableWrapper implements Runnable, Comparable<CallableWrapper> {
103            private XCallable<Void> callable;
104            private Instrumentation.Cron cron;
105    
106            public CallableWrapper(XCallable<Void> callable) {
107                this.callable = callable;
108                cron = new Instrumentation.Cron();
109                cron.start();
110            }
111    
112            public void run() {
113                if (Services.get().getSystemMode() == SYSTEM_MODE.SAFEMODE) {
114                    log.info("CallableWrapper[run] System is in SAFEMODE. Hence no callable run. But requeueing in delayQueue " + queue.size());
115                    delayedQueue.put(new DelayedCallableWrapper(callable, SAFE_MODE_DELAY));
116                    return;
117                }
118                try {
119                    if (callableBegin(callable)) {
120                        cron.stop();
121                        addInQueueCron(cron);
122                        XLog.Info.get().clear();
123                        XLog log = XLog.getLog(getClass());
124                        log.trace("executing callable [{0}]", callable.getName());
125                        try {
126                            callable.call();
127                            incrCounter(INSTR_EXECUTED_COUNTER, 1);
128                            log.trace("executed callable [{0}]", callable.getName());
129                        }
130                        catch (Exception ex) {
131                            incrCounter(INSTR_FAILED_COUNTER, 1);
132                            log.warn("exception callable [{0}], {1}", callable.getName(), ex.getMessage(), ex);
133                        }
134                        finally {
135                            XLog.Info.get().clear();
136                            callableEnd(callable);
137                        }
138                    }
139                    else {
140                        log.warn("max concurrency for callable type[{0}] exceeded, requeueing with [{1}]ms delay",
141                                 callable.getType(), CONCURRENCY_DELAY);
142                        queue(callable, CONCURRENCY_DELAY);
143                        incrCounter(callable.getType() + "#exceeded.concurrency", 1);
144                    }
145                }
146                finally {
147                    callableEnd(callable);
148                }
149            }
150    
151            public int compareTo(CallableWrapper callableWrapper) {
152                //priority is descending order
153                int diff = callableWrapper.callable.getPriority() - callable.getPriority();
154                if (diff == 0) {
155                    //createdTime is ascending order
156                    Long lDiff = callable.getCreatedTime() - callableWrapper.callable.getCreatedTime();
157                    diff = (lDiff < Integer.MIN_VALUE) ? Integer.MIN_VALUE : ((lDiff > Integer.MAX_VALUE) ? Integer.MAX_VALUE : lDiff.intValue());
158                }
159                return diff;
160            }
161        }

163        class CompositeCallable<Void> implements XCallable<Void> {
164            private List<XCallable<Void>> callables;
165            private String name;
166            private int priority;
167            private long createdTime;
168    
169            public CompositeCallable(List<XCallable<Void>> callables) {
170                this.callables = new ArrayList<XCallable<Void>>(callables);
171                priority = Integer.MIN_VALUE;
172                createdTime = Long.MAX_VALUE;
173                StringBuilder sb = new StringBuilder();
174                String separator = "[";
175                for (XCallable callable : callables) {
176                    priority = Math.max(priority, callable.getPriority());
177                    createdTime = Math.min(createdTime, callable.getCreatedTime());
178                    sb.append(separator).append(callable.getName());
179                    separator = ",";
180                }
181                sb.append("]");
182                name = sb.toString();
183            }
184    
185            @Override
186            public String getName() {
187                return name;
188            }
189    
190            @Override
191            public String getType() {
192                return "#composite#";
193            }
194    
195            @Override
196            public int getPriority() {
197                return priority;
198            }
199    
200            @Override
201            public long getCreatedTime() {
202                return createdTime;
203            }
204    
205            public Void call() throws Exception {
206                XLog log = XLog.getLog(getClass());
207    
208                for (XCallable callable : callables) {
209                    log.trace("executing callable [{0}]", callable.getName());
210                    try {
211                        callable.call();
212                        incrCounter(INSTR_EXECUTED_COUNTER, 1);
213                        log.trace("executed callable [{0}]", callable.getName());
214                    }
215                    catch (Exception ex) {
216                        incrCounter(INSTR_FAILED_COUNTER, 1);
217                        log.warn("exception callable [{0}], {1}", callable.getName(), ex.getMessage(), ex);
218                    }
219                }
220    
221                // ticking -1 not to count the call to the composite callable
222                incrCounter(INSTR_EXECUTED_COUNTER, -1);
223                return null;
224            }
225    
226        }

228        // delayed callables are wrapped with the this wrapper for delayed queueing.
229        // The wrapper implements Comparable to be able to work with a priority queue.
230        private static class DelayedCallableWrapper implements Comparable<DelayedCallableWrapper> {
231            private Long executionTime;
232            private XCallable<Void> callable;
233    
234            DelayedCallableWrapper(XCallable<Void> callable, long delay) {
235                this.callable = callable;
236                this.executionTime = System.currentTimeMillis() + delay;
237            }
238    
239            public int compareTo(DelayedCallableWrapper dc) {
240                return (int) (getExecutionTime() - dc.getExecutionTime());
241            }
242    
243            public long getExecutionTime() {
244                return executionTime;
245            }
246    
247            public XCallable<Void> getCallable() {
248                return callable;
249            }

255        private int queueSize;
256        private PriorityBlockingQueue<CallableWrapper> queue;
257        private int delayedQueueSize;
258        private PriorityBlockingQueue<DelayedCallableWrapper> delayedQueue;
259        private AtomicLong delayQueueExecCounter = new AtomicLong(0);
260        private ThreadPoolExecutor executor;
261        private Instrumentation instrumentation;

263        /**
264         * Convenience method for instrumentation counters.
265         *
266         * @param name counter name.
267         * @param count count to increment the counter.
268         */
269        private void incrCounter(String name, int count) {
270            if (instrumentation != null) {
271                instrumentation.incr(INSTRUMENTATION_GROUP, name, count);
272            }
273        }
274    
275        private void addInQueueCron(Instrumentation.Cron cron) {
276            if (instrumentation != null) {
277                instrumentation.addCron(INSTRUMENTATION_GROUP, INSTR_IN_QUEUE_TIME_TIMER, cron);
278            }
279        }

281        /**
282         * Initialize the command queue service.
283         *
284         * @param services services instance.
285         */
286        @Override
287        @SuppressWarnings("unchecked")
288        public void init(Services services) {
289            Configuration conf = services.getConf();
290    
291            queueSize = conf.getInt(CONF_QUEUE_SIZE, 1000);
292            int threads = conf.getInt(CONF_THREADS, 10);
293            delayedQueueSize = conf.getInt(CONF_DELAYED_QUEUE_SIZE, 1000);
294    
295            queue = new PriorityBlockingQueue<CallableWrapper>(queueSize);
296            delayedQueue = new PriorityBlockingQueue<DelayedCallableWrapper>(delayedQueueSize);
297            executor = new ThreadPoolExecutor(threads, threads, 10, TimeUnit.SECONDS, (PriorityBlockingQueue) queue);
298    
299            activeCallables = new HashMap<String, AtomicInteger>();
300            maxCallableConcurrency = conf.getInt(CONF_CALLABLE_CONCURRENCY, 3);
301    
302            // every 100 milliseconds this runnable polls the delayed queue for commands ready for execution.
303            // while there are commands ready for execution they are removed from the delayed queue and added to the
304            // execution queue.
305            // if the execution queue is full, the delayed command is added back to the delay queue and the loop
306            // is interruted until the next polling time.
307            Runnable delayedQueuePoller = new Runnable() {
308                public void run() {
309                    int queued = 0;
310                    if (!delayedQueue.isEmpty()) {
311                        while (!delayedQueue.isEmpty() && delayedQueue.peek().getExecutionTime() < System.currentTimeMillis()) {
312                            DelayedCallableWrapper delayed = delayedQueue.poll();
313                            if (!queue(delayed.getCallable())) {
314                                delayedQueue.add(delayed);
315                                break;
316                            }
317                            queued++;
318                        }
319                    }
320                    if (delayQueueExecCounter.get() % 3000 == 0) {
321                        XLog.getLog(getClass()).debug(
322                                "Total Instances of delayedQueuePoller  " + delayQueueExecCounter + " has queued " + queued
323                                        + " of commands from dealy queue to regular queue");
324                    }
325                    delayQueueExecCounter.getAndIncrement();
326                }
327            };
328            services.get(SchedulerService.class).schedule(delayedQueuePoller, 0, 100, SchedulerService.Unit.MILLISEC);
329        }

331        /**
332         * Destroy the command queue service.
333         */
334        @Override
335        public void destroy() {
336            try {
337                long limit = System.currentTimeMillis() + 30 * 1000;// 30 seconds
338                executor.shutdownNow();
339                while (!executor.awaitTermination(1000, TimeUnit.MILLISECONDS)) {
340                    log.info("Waiting for executor to shutdown");
341                    if (System.currentTimeMillis() > limit) {
342                        log.warn("Gave up, continuing without waiting for executor to shutdown");
343                        break;
344                    }
345                }
346            }
347            catch (InterruptedException ex) {
348                log.warn(ex);
349            }
350        }

362        /**
363         * Queue a callable for asynchronous execution.
364         *
365         * @param callable callable to queue.
366         * @return <code>true</code> if the callable was queued, <code>false</code> if the queue is full and the callable
367         *         was not queued.
368         */
369        public synchronized boolean queue(XCallable<Void> callable) {
370            if (Services.get().getSystemMode() == SYSTEM_MODE.SAFEMODE) {
371                log.info("[queue] System is in SAFEMODE. Hence no callable is queued. current queue size " + queue.size());
372                return false;
373            }
374    
375            if (queue.size() < queueSize) {
376                incrCounter(INSTR_QUEUED_COUNTER, 1);
377                try {
378                    executor.execute(new CallableWrapper(callable));
379                }
380                catch (Exception e) {
381                    log.warn("Didnot able to submit to executor:", e);
382                }
383                return true;
384            }
385            return false;
386        }

388        /**
389         * Queue a list of callables for serial execution. <p/> Useful to serialize callables that may compete with each
390         * other for resources. <p/> All callables will be processed with the priority of the highest priority of all
391         * callables.
392         *
393         * @param callables callables to be executed by the composite callable.
394         * @return <code>true</code> if the callables were queued, <code>false</code> if the queue is full and the callables
395         *         were not queued.
396         */
397        @SuppressWarnings("unchecked")
398        public synchronized boolean queueSerial(List<? extends XCallable<Void>> callables) {
399            if (Services.get().getSystemMode() == SYSTEM_MODE.SAFEMODE) {
400                log.info("[queueSerial] System is in SAFEMODE. Hence no callable is queued current queue size " + queue.size());
401                return false;
402            }
403            ParamChecker.notNullElements(callables, "callables");
404            if (callables.size() == 0) {
405                return true;
406            }
407            if (queue.size() < queueSize) {
408                incrCounter(INSTR_QUEUED_COUNTER, callables.size());
409                executor.execute(new CallableWrapper(new CompositeCallable(callables)));
410                return true;
411            }
412            return false;
413        }

429        /**
430         * Queue a callable for asynchronous execution sometime in the future.
431         *
432         * @param callable callable to queue for delayed execution
433         * @param delay time, in milliseconds, that the callable should be delayed.
434         * @return <code>true</code> if the callable was queued, <code>false</code> if the queue is full and the callable
435         *         was not queued.
436         */
437        public synchronized boolean queue(XCallable<Void> callable, long delay) {
438            if (Services.get().getSystemMode() == SYSTEM_MODE.SAFEMODE) {
439                log.info("[queue(delay)] System is in SAFEMODE. Hence no callable is queued. queue size " + queue.size());
440                return false;
441            }
442    
443            if (delayedQueue.size() < delayedQueueSize) {
444                incrCounter(INSTR_DELAYD_QUEUED_COUNTER, 1);
445                delayedQueue.put(new DelayedCallableWrapper(callable, delay));
446                return true;
447            }
448            return false;
449        }

451        /**
452         * Queue a list of callables for serial execution sometime in the future. <p/> Useful to serialize callables that
453         * may compete with each other for resources. <p/> All callables will be processed with the priority of the highest
454         * priority of all callables.
455         *
456         * @param callables callables to be executed by the composite callable.
457         * @param delay time, in milliseconds, that the callable should be delayed.
458         * @return <code>true</code> if the callables were queued, <code>false</code> if the queue is full and the callables
459         *         were not queued.
460         */
461        @SuppressWarnings("unchecked")
462        public synchronized boolean queueSerial(List<? extends XCallable<Void>> callables, long delay) {
463            if (Services.get().getSystemMode() == SYSTEM_MODE.SAFEMODE) {
464                log.info("[queue(delay)] System is in SAFEMODE. Hence no callable is queued. queue size " + queue.size());
465                return false;
466            }
467            ParamChecker.notNullElements(callables, "callables");
468            if (callables.size() == 0) {
469                return true;
470            }
471            if (queue.size() < queueSize) {
472                incrCounter(INSTR_QUEUED_COUNTER, callables.size());
473                boolean ret = queue(new CompositeCallable(callables), delay);
474                return ret;
475            }
476            return false;
477        }

479        /**
480         * Instruments the callable queue service.
481         *
482         * @param instr instance to instrument the callable queue service to.
483         */
484        public void instrument(Instrumentation instr) {
485            instrumentation = instr;
486            instr.addSampler(INSTRUMENTATION_GROUP, INSTR_QUEUE_SIZE_SAMPLER, 60, 1, new Instrumentation.Variable<Long>() {
487                public Long getValue() {
488                    return (long) queue.size();
489                }
490            });
491            instr.addSampler(INSTRUMENTATION_GROUP, INSTR_DELAYED_QUEUE_SIZE_SAMPLER, 60, 1, new Instrumentation.Variable<Long>() {
492                public Long getValue() {
493                    return (long) delayedQueue.size();
494                }
495            });
496            instr.addSampler(INSTRUMENTATION_GROUP, INSTR_THREADS_ACTIVE_SAMPLER, 60, 1, new Instrumentation.Variable<Long>() {
497                public Long getValue() {
498                    return (long) executor.getActiveCount();
499                }
500            });
