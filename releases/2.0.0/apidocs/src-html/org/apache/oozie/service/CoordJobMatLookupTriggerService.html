018    package org.apache.oozie.service;
019    
020    import java.util.ArrayList;
021    import java.util.Date;
022    import java.util.List;
023    
024    import org.apache.hadoop.conf.Configuration;
025    import org.apache.oozie.CoordinatorJobBean;
026    import org.apache.oozie.command.coord.CoordJobMatLookupCommand;
027    import org.apache.oozie.store.CoordinatorStore;
028    import org.apache.oozie.store.StoreException;
029    import org.apache.oozie.util.XCallable;
030    import org.apache.oozie.util.XLog;
031    
032    /**
033     * The coordinator Materialization Lookup trigger service schedule lookup trigger command for every interval (default is
034     * 5 minutes ). This interval could be configured through oozie configuration defined is either oozie-default.xml or
035     * oozie-site.xml using the property name oozie.service.CoordJobMatLookupTriggerService.lookup.interval
036     */
037    public class CoordJobMatLookupTriggerService implements Service {
038        public static final String CONF_PREFIX = Service.CONF_PREFIX + "CoordJobMatLookupTriggerService.";
039        /**
040         * Time interval, in seconds, at which the Job materialization service will be scheduled to run.
041         */
042        public static final String CONF_LOOKUP_INTERVAL = CONF_PREFIX + "lookup.interval";
043        /**
044         * This configuration defined the duration for which job should be materialized in future
045         */
046        public static final String CONF_MATERIALIZATION_WINDOW = CONF_PREFIX + "materialization.window";
047        /**
048         * The number of callables to be queued in a batch.
049         */
050        public static final String CONF_CALLABLE_BATCH_SIZE = CONF_PREFIX + "callable.batch.size";
051    
052        private static final String INSTRUMENTATION_GROUP = "coord_job_mat_lookup";
053        private static final String INSTR_MAT_JOBS_COUNTER = "jobs";
054        private static final int CONF_LOOKUP_INTERVAL_DEFAULT = 300;
055        private static final int CONF_MATERIALIZATION_WINDOW_DEFAULT = 3600;
056    
057        /**
058         * This runnable class will run in every "interval" to queue CoordJobMatLookupTriggerCommand.
059         */
060        static class CoordJobMatLookupTriggerRunnable implements Runnable {
061            private int materializationWindow;
062            private long delay = 0;
063            private List<XCallable<Void>> callables;
064            private List<XCallable<Void>> delayedCallables;
065    
066            public CoordJobMatLookupTriggerRunnable(int materializationWindow) {
067                this.materializationWindow = materializationWindow;
068            }
069    
070            @Override
071            public void run() {
072                runCoordJobMatLookup();
073    
074                if (null != callables) {
075                    boolean ret = Services.get().get(CallableQueueService.class).queueSerial(callables);
076                    if (ret == false) {
077                        XLog.getLog(getClass()).warn(
078                                "Unable to queue the callables commands for CoordJobMatLookupTriggerRunnable. "
079                                        + "Most possibly command queue is full. Queue size is :"
080                                        + Services.get().get(CallableQueueService.class).queueSize());
081                    }
082                    callables = null;
083                }
084                if (null != delayedCallables) {
085                    boolean ret = Services.get().get(CallableQueueService.class).queueSerial(delayedCallables, this.delay);
086                    if (ret == false) {
087                        XLog.getLog(getClass()).warn(
088                                "Unable to queue the delayedCallables commands for CoordJobMatLookupTriggerRunnable. "
089                                        + "Most possibly delayedQueue is full. DelayedQueue size is :"
090                                        + Services.get().get(CallableQueueService.class).delayedQueueSize());
091                    }
092                    delayedCallables = null;
093                    this.delay = 0;
094                }
095            }
096    
097            /**
098             * Recover coordinator jobs that should be materialized
099             */
100            private void runCoordJobMatLookup() {
101                XLog.Info.get().clear();
102                XLog log = XLog.getLog(getClass());
103    
104                CoordinatorStore store = null;
105                try {
106                    store = Services.get().get(StoreService.class).getStore(CoordinatorStore.class);
107                    store.beginTrx();
108    
109                    // get current date
110                    Date currDate = new Date(new Date().getTime() - CONF_LOOKUP_INTERVAL_DEFAULT * 100);
111                    // get list of all jobs that have actions that should be
112                    // materialized.
113                    List<CoordinatorJobBean> materializeJobs = store.getCoordinatorJobsToBeMaterialized(currDate, 50);
114                    log.debug("CoordJobMatLookupTriggerService - Curr Date= " + currDate + ", Num jobs to materialize = "
115                            + materializeJobs.size());
116                    for (CoordinatorJobBean coordJob : materializeJobs) {
117                        Services.get().get(InstrumentationService.class).get().incr(INSTRUMENTATION_GROUP,
118                                                                                    INSTR_MAT_JOBS_COUNTER, 1);
119                        queueCallable(new CoordJobMatLookupCommand(coordJob.getId(), materializationWindow));
120                    }
121    
122                    store.commitTrx();
123                }
124                catch (StoreException ex) {
125                    if (store != null) {
126                        store.rollbackTrx();
127                    }
128                    log.warn("Exception while accessing the store", ex);
129                }
130                finally {
131                    try {
132                        if (store != null) {
133                            store.closeTrx();
134                        }
135                    }
136                    catch (RuntimeException rex) {
137                        log.warn("Exception while attempting to close store", rex);
138                    }
139                }
140            }
141    
142            /**
143             * Adds callables to a list. If the number of callables in the list reaches {@link
144             * CoordJobMatLookupTriggerService#CONF_CALLABLE_BATCH_SIZE}, the entire batch is queued and the callables list
145             * is reset.
146             *
147             * @param callable the callable to queue.
148             */
149            private void queueCallable(XCallable<Void> callable) {
150                if (callables == null) {
151                    callables = new ArrayList<XCallable<Void>>();
152                }
153                callables.add(callable);
154                if (callables.size() == Services.get().getConf().getInt(CONF_CALLABLE_BATCH_SIZE, 10)) {
155                    boolean ret = Services.get().get(CallableQueueService.class).queueSerial(callables);
156                    if (ret == false) {
157                        XLog.getLog(getClass()).warn(
158                                "Unable to queue the callables commands for CoordJobMatLookupTriggerRunnable. "
159                                        + "Most possibly command queue is full. Queue size is :"
160                                        + Services.get().get(CallableQueueService.class).queueSize());
161                    }
162                    callables = new ArrayList<XCallable<Void>>();
163                }
164            }
165    
166            /**
167             * Adds callables to a list. If the number of callables in the list reaches {@link
168             * CoordJobMatLookupTriggerService#CONF_CALLABLE_BATCH_SIZE}, the entire batch is queued with the delay set to
169             * the maximum delay of the callables in the list. The callables list and the delay is reset.
170             *
171             * @param callable the callable to queue.
172             * @param delay the delay for the callable.
173             */
174            private void queueCallable(XCallable<Void> callable, long delay) {
175                if (delayedCallables == null) {
176                    delayedCallables = new ArrayList<XCallable<Void>>();
177                }
178                this.delay = Math.max(this.delay, delay);
179                delayedCallables.add(callable);
180                if (delayedCallables.size() == Services.get().getConf().getInt(CONF_CALLABLE_BATCH_SIZE, 10)) {
181                    boolean ret = Services.get().get(CallableQueueService.class).queueSerial(delayedCallables, this.delay);
182                    if (ret == false) {
183                        XLog.getLog(getClass()).warn(
184                                "Unable to queue the delayedCallables commands for CoordJobMatLookupTriggerRunnable. "
185                                        + "Most possibly delayedQueue is full. DelayedQueue size is :"
186                                        + Services.get().get(CallableQueueService.class).delayedQueueSize());
187                    }
188                    delayedCallables = new ArrayList<XCallable<Void>>();
189                    this.delay = 0;
190                }
191            }
192    
193        }
194    
195        @Override
196        public void init(Services services) throws ServiceException {
197            Configuration conf = services.getConf();
198            Runnable lookupTriggerJobsRunnable = new CoordJobMatLookupTriggerRunnable(conf.getInt(
199                    CONF_MATERIALIZATION_WINDOW, CONF_MATERIALIZATION_WINDOW_DEFAULT));// Default is 1 hour
200            services.get(SchedulerService.class).schedule(lookupTriggerJobsRunnable, 10,
201                                                          conf.getInt(CONF_LOOKUP_INTERVAL, CONF_LOOKUP_INTERVAL_DEFAULT),// Default is 5 minutes
202                                                          SchedulerService.Unit.SEC);
203            return;
204        }
205    
206        @Override
207        public void destroy() {
208            // TODO Auto-generated method stub
209    
210        }
211    
212        @Override
213        public Class<? extends Service> getInterface() {
214            return CoordJobMatLookupTriggerService.class;
215        }
