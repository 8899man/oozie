018    package org.apache.oozie.servlet;
019    
020    import org.apache.oozie.client.rest.RestConstants;
021    import org.apache.oozie.service.Services;
022    import org.apache.oozie.util.IOUtils;
023    import org.apache.oozie.util.PropertiesUtils;
024    import org.apache.oozie.util.XLog;
025    import org.apache.oozie.DagEngine;
026    import org.apache.oozie.DagEngineException;
027    import org.apache.oozie.ErrorCode;
028    import org.apache.oozie.service.DagEngineService;
029    import org.apache.oozie.service.CallbackService;
030    
031    import javax.servlet.ServletException;
032    import javax.servlet.http.HttpServletRequest;
033    import javax.servlet.http.HttpServletResponse;
034    import java.io.IOException;
035    import java.util.Arrays;
036    import java.util.Collections;
037    import java.util.Properties;
038    
039    public class CallbackServlet extends JsonRestServlet {
040        private static final String INSTRUMENTATION_NAME = "callback";
041    
042        private static final ResourceInfo RESOURCE_INFO =
043                new ResourceInfo("", Arrays.asList("POST", "GET"), Collections.EMPTY_LIST);
044    
045        public final static String CONF_MAX_DATA_LEN = "oozie.servlet.CallbackServlet.max.data.len";
046    
047        private static int maxDataLen;
048    
049        public CallbackServlet() {
050            super(INSTRUMENTATION_NAME, RESOURCE_INFO);
051        }
052    
053        public void init() {
054            maxDataLen = Services.get().getConf().getInt(CONF_MAX_DATA_LEN, 2 * 1024);
055        }
056    
057        /**
058         * GET callback
059         */
060        protected void doGet(HttpServletRequest request, HttpServletResponse response)
061                throws ServletException, IOException {
062            String queryString = request.getQueryString();
063            XLog.getLog(getClass()).debug("Received a CallbackServlet.doGet() with query string " + queryString);
064            CallbackService callbackService = Services.get().get(CallbackService.class);
065            if (!callbackService.isValid(queryString)) {
066                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0402, queryString);
067            }
068            DagEngine dagEngine = Services.get().get(DagEngineService.class).getSystemDagEngine();
069            try {
070                XLog.getLog(getClass()).info(XLog.STD, "callback for action [{0}]",
071                                             callbackService.getActionId(queryString));
072                dagEngine.processCallback(callbackService.getActionId(queryString),
073                                          callbackService.getExternalStatus(queryString), null);
074            }
075            catch (DagEngineException ex) {
076                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ex);
077            }
078        }
079    
080        /**
081         * POST callback
082         */
083        protected void doPost(HttpServletRequest request, HttpServletResponse response)
084                throws ServletException, IOException {
085            String queryString = request.getQueryString();
086            XLog.getLog(getClass()).debug("Received a CallbackServlet.doPost() with query string " + queryString);
087            CallbackService callbackService = Services.get().get(CallbackService.class);
088            if (!callbackService.isValid(queryString)) {
089                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0402, queryString);
090            }
091            validateContentType(request, RestConstants.TEXT_CONTENT_TYPE);
092            try {
093                XLog.getLog(getClass()).info(XLog.STD, "callback for action [{0}]",
094                                             callbackService.getActionId(queryString));
095                String data = IOUtils.getReaderAsString(request.getReader(), maxDataLen);
096                Properties props = PropertiesUtils.stringToProperties(data);
097                DagEngine dagEngine = Services.get().get(DagEngineService.class).getSystemDagEngine();
098                dagEngine.processCallback(callbackService.getActionId(queryString),
099                                          callbackService.getExternalStatus(queryString), props);
100            }
101            catch (IOException ex) {
102                if (ex.getMessage().startsWith("stream exceeds limit")) {
103                    //TODO, WE MUST SET THE ACTION TO ERROR
104                    throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ErrorCode.E0403, "data too long");
105                }
106                else {
107                    throw ex;
108                }
109            }
110            catch (DagEngineException ex) {
111                throw new XServletException(HttpServletResponse.SC_BAD_REQUEST, ex);
112            }
113        }
114    }
