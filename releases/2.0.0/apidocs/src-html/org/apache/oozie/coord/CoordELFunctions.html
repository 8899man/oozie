136        /**
137         * Calculate the difference of timezone offset in minutes between dataset and coordinator job. <p/> Depends on: <p/>
138         * 1. Timezone of both dataset and job <p/> 2. Action creation Time
139         *
140         * @return difference in minutes (DataSet TZ Offset - Application TZ offset)
141         */
142        public static int ph2_coord_tzOffset() {
143            Date actionCreationTime = getActionCreationtime();
144            TimeZone dsTZ = ParamChecker.notNull(getDatasetTZ(), "DatasetTZ");
145            TimeZone jobTZ = ParamChecker.notNull(getJobTZ(), "JobTZ");
146            // Apply the TZ into Calendar object
147            Calendar dsTime = Calendar.getInstance(dsTZ);
148            dsTime.setTime(actionCreationTime);
149            Calendar jobTime = Calendar.getInstance(jobTZ);
150            jobTime.setTime(actionCreationTime);
151            return (dsTime.get(Calendar.ZONE_OFFSET) - jobTime.get(Calendar.ZONE_OFFSET)) / (1000 * 60);
152        }

223        /**
224         * Used to specify a list of URI's that are used as input dir to the workflow job. <p/> Look for two evaluator-level
225         * variables <p/> A) .datain.<DATAIN_NAME> B) .datain.<DATAIN_NAME>.unresolved <p/> A defines the current list of
226         * URI. <p/> B defines whether there are any unresolved EL-function (i.e latest) <p/> If there are something
227         * unresolved, this function will echo back the original function <p/> otherwise it sends the uris.
228         *
229         * @param dataInName : Datain name
230         * @return the list of URI's separated by INSTANCE_SEPARATOR <p/> if there are unresolved EL function (i.e. latest)
231         *         , echo back <p/> the function without resolving the function.
232         */
233        public static String ph3_coord_dataIn(String dataInName) {
234            String uris = "";
235            ELEvaluator eval = ELEvaluator.getCurrent();
236            uris = (String) eval.getVariable(".datain." + dataInName);
237            Boolean unresolved = (Boolean) eval.getVariable(".datain." + dataInName + ".unresolved");
238            if (unresolved != null && unresolved.booleanValue() == true) {
239                return "${coord:dataIn('" + dataInName + "')}";
240            }
241            return uris;
242        }

278         * Determine how many hours is on the date of n-th dataset instance. <p/> It depends on: <p/> 1. Data set frequency
279         * <p/> 2. Data set Time unit (day, month, minute) <p/> 3. Data set Time zone/DST <p/> 4. End Day/Month flag <p/> 5.
280         * Data set initial instance <p/> 6. Action Creation Time
281         *
282         * @param n instance count <p/> domain: n is integer
283         * @return number of hours on that day <p/> returns -1 means n-th instance is earlier than Initial-Instance of DS
284         * @throws Exception
285         */
286        public static int ph2_coord_hoursInDay(int n) throws Exception {
287            int datasetFrequency = (int) getDSFrequency();
288            // /Calendar nominalInstanceCal =
289            // getCurrentInstance(getActionCreationtime());
290            Calendar nominalInstanceCal = getEffectiveNominalTime();
291            if (nominalInstanceCal == null) {
292                return -1;
293            }
294            nominalInstanceCal.add(getDSTimeUnit().getCalendarUnit(), datasetFrequency * n);
295            /*
296             * if (nominalInstanceCal.getTime().compareTo(getInitialInstance()) < 0)
297             * { return -1; }
298             */
299            nominalInstanceCal.setTimeZone(getDatasetTZ());// Use Dataset TZ
300            // DateUtils.moveToEnd(nominalInstanceCal, getDSEndOfFlag());
301            return DateUtils.hoursInDay(nominalInstanceCal);
302        }

374        /**
375         * Helper method to wrap around with "${..}". <p/>
376         *
377         * @param eval :EL evaluator
378         * @param expr : expression to evaluate
379         * @return Resolved expression or echo back the same expression
380         * @throws Exception
381         */
382        public static String evalAndWrap(ELEvaluator eval, String expr) throws Exception {
383            eval.setVariable(".wrap", null);
384            String result = eval.evaluate(expr, String.class);
385            if (eval.getVariable(".wrap") != null) {
386                return "${" + result + "}";
387            }
388            else {
389                return result;
390            }
391        }

514        /**
515         * @param n
516         * @return n-th instance Date-Time from current instance for data-set <p/> return empty string ("") if the
517         *         Action_Creation_time or the n-th instance <p/> is earlier than the Initial_Instance of dataset.
518         * @throws Exception
519         */
520        private static String coord_current_sync(int n) throws Exception {
521            int datasetFrequency = getDSFrequency();// in minutes
522            TimeUnit dsTimeUnit = getDSTimeUnit();
523            int[] instCount = new int[1];// used as pass by ref
524            Calendar nominalInstanceCal = getCurrentInstance(getActionCreationtime(), instCount);
525            if (nominalInstanceCal == null) {
526                return "";
527            }
528            nominalInstanceCal = getInitialInstanceCal();
529            int absInstanceCount = instCount[0] + n;
530            nominalInstanceCal.add(dsTimeUnit.getCalendarUnit(), datasetFrequency * absInstanceCount);
531    
532            if (nominalInstanceCal.getTime().compareTo(getInitialInstance()) < 0) {
533                return "";
534            }
535            String str = DateUtils.formatDateUTC(nominalInstanceCal);
536            return str;
537        }

539        /**
540         * @param offset
541         * @return n-th available latest instance Date-Time for SYNC data-set
542         * @throws Exception
543         */
544        private static String coord_latest_sync(int offset) throws Exception {
545            if (offset > 0) {
546                throw new RuntimeException("For latest there is no meaning " + "of positive instance. n should be <=0"
547                        + offset);
548            }
549            ELEvaluator eval = ELEvaluator.getCurrent();
550            String retVal = "";
551            int datasetFrequency = (int) getDSFrequency();// in minutes
552            TimeUnit dsTimeUnit = getDSTimeUnit();
553            int[] instCount = new int[1];
554            Calendar nominalInstanceCal = getCurrentInstance(getActualTime(), instCount);
555            if (nominalInstanceCal != null) {
556                Calendar initInstance = getInitialInstanceCal();
557                SyncCoordDataset ds = (SyncCoordDataset) eval.getVariable(DATASET);
558                if (ds == null) {
559                    throw new RuntimeException("Associated Dataset should be defined with key " + DATASET);
560                }
561                String uriTemplate = ds.getUriTemplate();
562                Configuration conf = (Configuration) eval.getVariable(CONFIGURATION);
563                if (conf == null) {
564                    throw new RuntimeException("Associated Configuration should be defined with key " + CONFIGURATION);
565                }
566                int available = 0;
567                boolean resolved = false;
568                String user = ParamChecker
569                        .notEmpty((String) eval.getVariable(OozieClient.USER_NAME), OozieClient.USER_NAME);
570                String group = ParamChecker.notEmpty((String) eval.getVariable(OozieClient.GROUP_NAME),
571                                                     OozieClient.GROUP_NAME);
572                String doneFlag = ds.getDoneFlag();
573                while (nominalInstanceCal.compareTo(initInstance) >= 0) {
574                    ELEvaluator uriEval = getUriEvaluator(nominalInstanceCal);
575                    String uriPath = uriEval.evaluate(uriTemplate, String.class);
576                    String pathWithDoneFlag = uriPath;
577                    if (doneFlag.length() > 0) {
578                        pathWithDoneFlag += "/" + doneFlag;
579                    }
580                    if (isPathAvailable(pathWithDoneFlag, user, group, conf)) {
581                        XLog.getLog(CoordELFunctions.class).debug("Found latest(" + available + "): " + pathWithDoneFlag);
582                        if (available == offset) {
583                            XLog.getLog(CoordELFunctions.class).debug("Found Latest File: " + pathWithDoneFlag);
584                            resolved = true;
585                            retVal = DateUtils.formatDateUTC(nominalInstanceCal);
586                            eval.setVariable("resolved_path", uriPath);
587                            break;
588                        }
589    
590                        available--;
591                    }
592                    // nominalInstanceCal.add(dsTimeUnit.getCalendarUnit(),
593                    // -datasetFrequency);
594                    nominalInstanceCal = (Calendar) initInstance.clone();
595                    instCount[0]--;
596                    nominalInstanceCal.add(dsTimeUnit.getCalendarUnit(), instCount[0] * datasetFrequency);
597                    // DateUtils.moveToEnd(nominalInstanceCal, getDSEndOfFlag());
598                }
599                if (!resolved) {
600                    // return unchanged latest function with variable 'is_resolved'
601                    // to 'false'
602                    eval.setVariable("is_resolved", Boolean.FALSE);
603                    retVal = "${coord:latest(" + offset + ")}";
604                }
605                else {
606                    eval.setVariable("is_resolved", Boolean.TRUE);
607                }
608            }
609            else {// No feasible nominal time
610                eval.setVariable("is_resolved", Boolean.FALSE);
611            }
612            return retVal;
613        }

635         * @param tm
636         * @return a new Evaluator to be used for URI-template evaluation
637         */
638        private static ELEvaluator getUriEvaluator(Calendar tm) {
639            ELEvaluator retEval = new ELEvaluator();

641            retEval.setVariable("MONTH", (tm.get(Calendar.MONTH) + 1) < 10 ? "0" + (tm.get(Calendar.MONTH) + 1) : (tm

751        /**
752         * Find the current instance based on effectiveTime (i.e Action_Creation_Time or Action_Start_Time)
753         *
754         * @return current instance i.e. current(0) returns null if effectiveTime is earlier than Initial Instance time of
755         *         the dataset.
756         */
757        private static Calendar getCurrentInstance(Date effectiveTime, int instanceCount[]) {
758            Date datasetInitialInstance = getInitialInstance();
759            TimeUnit dsTimeUnit = getDSTimeUnit();
760            TimeZone dsTZ = getDatasetTZ();
761            // Convert Date to Calendar for corresponding TZ
762            Calendar current = Calendar.getInstance();
763            current.setTime(datasetInitialInstance);
764            current.setTimeZone(dsTZ);
765    
766            Calendar calEffectiveTime = Calendar.getInstance();
767            calEffectiveTime.setTime(effectiveTime);
768            calEffectiveTime.setTimeZone(dsTZ);
769            instanceCount[0] = 0;
770            if (current.compareTo(calEffectiveTime) > 0) {
771                // Nominal Time < initial Instance
772                // TODO: getClass() call doesn't work from static method.
773                // XLog.getLog("CoordELFunction.class").warn("ACTION CREATED BEFORE INITIAL INSTACE "+
774                // current.getTime());
775                return null;
776            }
777            Calendar origCurrent = (Calendar) current.clone();
778            while (current.compareTo(calEffectiveTime) <= 0) {
779                current = (Calendar) origCurrent.clone();
780                instanceCount[0]++;
781                current.add(dsTimeUnit.getCalendarUnit(), instanceCount[0] * getDSFrequency());
782            }
783            instanceCount[0]--;
784    
785            current = (Calendar) origCurrent.clone();
786            current.add(dsTimeUnit.getCalendarUnit(), instanceCount[0] * getDSFrequency());
787            return current;
788        }

809        }
