119                List<Class> classes = getLauncherClasses();

143            for (String prop : DISALLOWED_PROPERTIES) {

152            Configuration conf = new XConfiguration();
153            conf.set(HADOOP_USER, context.getProtoActionConf().get(WorkflowAppService.HADOOP_USER));

156            Namespace ns = actionXml.getNamespace();
157            String jobTracker = actionXml.getChild("job-tracker", ns).getTextTrim();
158            String nameNode = actionXml.getChild("name-node", ns).getTextTrim();
159            conf.set(HADOOP_JOB_TRACKER, jobTracker);

161            conf.set("mapreduce.fileoutputcommitter.marksuccessfuljobs", "true");
162            return conf;
163        }

165        Configuration setupLauncherConf(Configuration conf, Element actionXml, Path appPath) throws ActionExecutorException {
166            try {
167                Namespace ns = actionXml.getNamespace();
168                Element e = actionXml.getChild("configuration", ns);
169                if (e != null) {
170                    String strConf = XmlUtils.prettyPrint(e).toString();
171                    XConfiguration inlineConf = new XConfiguration(new StringReader(strConf));
172    
173                    XConfiguration launcherConf = new XConfiguration();
174                    for (Map.Entry<String, String> entry : inlineConf) {
175                        if (entry.getKey().startsWith("oozie.launcher.")) {
176                            String name = entry.getKey().substring("oozie.launcher.".length());
177                            String value = entry.getValue();
178                            // setting original KEY
179                            launcherConf.set(entry.getKey(), value);
180                            // setting un-prefixed key (to allow Hadoop job config
181                            // for the launcher job
182                            launcherConf.set(name, value);
183                        }
184                    }
185                    checkForDisallowedProps(launcherConf, "inline launcher configuration");
186                    XConfiguration.copy(launcherConf, conf);
187                }
188                return conf;
189            }
190            catch (IOException ex) {
191                throw convertException(ex);
192            }
193        }

214        Configuration setupActionConf(Configuration actionConf, Context context, Element actionXml, Path appPath)
215                throws ActionExecutorException {
216            try {
217                Namespace ns = actionXml.getNamespace();
218                Element e = actionXml.getChild("job-xml", ns);
219                if (e != null) {
220                    String jobXml = e.getTextTrim();
221                    Path path = new Path(appPath, jobXml);
222                    FileSystem fs = getActionFileSystem(context, actionXml);
223                    Configuration jobXmlConf = new XConfiguration(fs.open(path));
224                    checkForDisallowedProps(jobXmlConf, "job-xml");
225                    XConfiguration.copy(jobXmlConf, actionConf);
226                }
227                e = actionXml.getChild("configuration", ns);
228                if (e != null) {
229                    String strConf = XmlUtils.prettyPrint(e).toString();
230                    XConfiguration inlineConf = new XConfiguration(new StringReader(strConf));
231                    checkForDisallowedProps(inlineConf, "inline configuration");
232                    XConfiguration.copy(inlineConf, actionConf);
233                }
234                return actionConf;
235            }
236            catch (IOException ex) {
237                throw convertException(ex);
238            }
239        }

241        Configuration addToCache(Configuration conf, Path appPath, String filePath, boolean archive)
242                throws ActionExecutorException {
243            Path path = null;
244            try {
245                if (filePath.startsWith("/")) {
246                    path = new Path(filePath);
247                }
248                else {
249                    path = new Path(appPath, filePath);
250                }
251                URI uri = new URI(path.toUri().getPath());
252                if (archive) {
253                    DistributedCache.addCacheArchive(uri, conf);
254                }
255                else {
256                    String fileName = filePath.substring(filePath.lastIndexOf("/") + 1);
257                    if (fileName.endsWith(".so") || fileName.contains(".so.")) {
258                        if (!fileName.endsWith(".so")) {
259                            int extAt = fileName.indexOf(".so.");
260                            fileName = fileName.substring(0, extAt + 3);
261                        }
262                        uri = new Path(path.toString() + "#" + fileName).toUri();
263                        uri = new URI(uri.getPath());
264                    }
265                    else {
266                        if (!fileName.contains("#")) {
267                            path = new Path(uri.toString());
268    
269                            String user = conf.get("user.name");
270                            String group = conf.get("group.name");
271                            Services.get().get(HadoopAccessorService.class).addFileToClassPath(user, group, path, conf);
272                        }
273                    }
274                    DistributedCache.addCacheFile(uri, conf);
275                }
276                DistributedCache.createSymlink(conf);
277                return conf;
278            }
279            catch (Exception ex) {
280                XLog.getLog(getClass()).debug(
281                        "Errors when add to DistributedCache. Path=" + path + ", archive=" + archive + ", conf="
282                                + XmlUtils.prettyPrint(conf).toString());
283                throw convertException(ex);
284            }
285        }

296        void prepareActionDir(FileSystem actionFs, Context context) throws ActionExecutorException {
297            try {
298                Path actionDir = context.getActionDir();
299                Path tempActionDir = new Path(actionDir.getParent(), actionDir.getName() + ".tmp");
300                if (!actionFs.exists(actionDir)) {
301                    try {
302                        actionFs.copyFromLocalFile(new Path(getOozieRuntimeDir(), getLauncherJarName()), new Path(
303                                tempActionDir, getLauncherJarName()));
304                        actionFs.rename(tempActionDir, actionDir);
305                    }
306                    catch (IOException ex) {
307                        actionFs.delete(tempActionDir, true);
308                        actionFs.delete(actionDir, true);
309                        throw ex;
310                    }
311                }
312            }
313            catch (Exception ex) {
314                throw convertException(ex);
315            }
316        }

331        @SuppressWarnings("unchecked")
332        void setLibFilesArchives(Context context, Element actionXml, Path appPath, Configuration conf)
333                throws ActionExecutorException {
334            Configuration proto = context.getProtoActionConf();
335    
336            addToCache(conf, appPath, getOozieLauncherJar(context), false);
337    
338            String[] paths = proto.getStrings(WorkflowAppService.APP_LIB_JAR_PATH_LIST);
339            if (paths != null) {
340                for (String jarPath : paths) {
341                    addToCache(conf, appPath, jarPath, false);
342                }
343            }
344    
345            paths = proto.getStrings(WorkflowAppService.APP_LIB_SO_PATH_LIST);
346            if (paths != null) {
347                for (String soPath : paths) {
348                    addToCache(conf, appPath, soPath, false);
349                }
350            }
351    
352            for (Element eProp : (List<Element>) actionXml.getChildren()) {
353                if (eProp.getName().equals("file")) {
354                    String path = eProp.getTextTrim();
355                    addToCache(conf, appPath, path, false);
356                }
357                else {
358                    if (eProp.getName().equals("archive")) {
359                        String path = eProp.getTextTrim();
360                        addToCache(conf, appPath, path, true);
361                    }
362                }
363            }
364        }

380        @SuppressWarnings("unchecked")
381        JobConf createLauncherConf(Context context, WorkflowAction action, Element actionXml, Configuration actionConf)
382                throws ActionExecutorException {
383            try {
384                Path appPath = new Path(context.getWorkflow().getAppPath());
385    
386                // launcher job configuration
387                Configuration launcherConf = createBaseHadoopConf(context, actionXml);
388                setupLauncherConf(launcherConf, actionXml, appPath);
389    
390                // we are doing init+copy because if not we are getting 'hdfs'
391                // scheme not known
392                // its seems that new JobConf(Conf) does not load defaults, it
393                // assumes parameter Conf does.
394                JobConf launcherJobConf = new JobConf();
395                XConfiguration.copy(launcherConf, launcherJobConf);
396                setLibFilesArchives(context, actionXml, appPath, launcherJobConf);
397                String jobName = XLog.format("oozie:launcher:T={0}:W={1}:A={2}:ID={3}", getType(), context.getWorkflow()
398                        .getAppName(), action.getName(), context.getWorkflow().getId());
399                launcherJobConf.setJobName(jobName);
400    
401                String jobId = context.getWorkflow().getId();
402                String actionId = action.getId();
403                Path actionDir = context.getActionDir();
404                String recoveryId = context.getRecoveryId();
405    
406                LauncherMapper.setupLauncherInfo(launcherJobConf, jobId, actionId, actionDir, recoveryId, actionConf);
407    
408                LauncherMapper.setupMainClass(launcherJobConf, getLauncherMain(launcherConf, actionXml));
409    
410                LauncherMapper.setupMaxOutputData(launcherJobConf, maxActionOutputLen);
411    
412                Namespace ns = actionXml.getNamespace();
413                List<Element> list = (List<Element>) actionXml.getChildren("arg", ns);
414                String[] args = new String[list.size()];
415                for (int i = 0; i < list.size(); i++) {
416                    args[i] = list.get(i).getTextTrim();
417                }
418                LauncherMapper.setupMainArguments(launcherJobConf, args);
419    
420                Element opt = actionXml.getChild("java-opts", ns);
421                if (opt != null) {
422                    String opts = launcherConf.get("mapred.child.java.opts", "");
423                    opts = opts + " " + opt.getTextTrim();
424                    opts = opts.trim();
425                    launcherJobConf.set("mapred.child.java.opts", opts);
426                }
427    
428                // properties from action that are needed by the launcher (QUEUE
429                // NAME)
430                // maybe we should add queue to the WF schema, below job-tracker
431                for (String name : SPECIAL_PROPERTIES) {
432                    String value = actionConf.get(name);
433                    if (value != null) {
434                        launcherJobConf.set(name, value);
435                    }
436                }
437    
438                // to disable cancelation of delegation token on launcher job end
439                launcherJobConf.setBoolean("mapreduce.job.complete.cancel.delegation.tokens", false);
440    
441                // setting the group owning the Oozie job to allow anybody in that
442                // group to kill the jobs.
443                launcherJobConf.set("mapreduce.job.acl-modify-job", context.getWorkflow().getGroup());
444    
445                return launcherJobConf;
446            }
447            catch (Exception ex) {
448                throw convertException(ex);
449            }
450        }

468        void submitLauncher(Context context, WorkflowAction action) throws ActionExecutorException {
469            JobClient jobClient = null;
470            boolean exception = false;
471            try {
472                Path appPath = new Path(context.getWorkflow().getAppPath());
473                Element actionXml = XmlUtils.parseXml(action.getConf());
474    
475                // action job configuration
476                Configuration actionConf = createBaseHadoopConf(context, actionXml);
477                setupActionConf(actionConf, context, actionXml, appPath);
478                XLog.getLog(getClass()).debug("Setting LibFilesArchives ");
479                setLibFilesArchives(context, actionXml, appPath, actionConf);
480                String jobName = XLog.format("oozie:action:T={0}:W={1}:A={2}:ID={3}", getType(), context.getWorkflow()
481                        .getAppName(), action.getName(), context.getWorkflow().getId());
482                actionConf.set("mapred.job.name", jobName);
483                injectActionCallback(context, actionConf);
484    
485                // setting the group owning the Oozie job to allow anybody in that
486                // group to kill the jobs.
487                actionConf.set("mapreduce.job.acl-modify-job", context.getWorkflow().getGroup());
488    
489                JobConf launcherJobConf = createLauncherConf(context, action, actionXml, actionConf);
490                injectLauncherCallback(context, launcherJobConf);
491                XLog.getLog(getClass()).debug("Creating Job Client for action " + action.getId());
492                jobClient = createJobClient(context, launcherJobConf);
493                String launcherId = LauncherMapper.getRecoveryId(launcherJobConf, context.getActionDir(), context
494                        .getRecoveryId());
495                boolean alreadyRunning = launcherId != null;
496                RunningJob runningJob;
497    
498                if (alreadyRunning) {
499                    runningJob = jobClient.getJob(JobID.forName(launcherId));
500                    if (runningJob == null) {
501                        String jobTracker = launcherJobConf.get("mapred.job.tracker");
502                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "JA017",
503                                                          "unknown job [{0}@{1}], cannot recover", launcherId, jobTracker);
504                    }
505                }
506                else {
507                    prepare(context, actionXml);
508                    XLog.getLog(getClass()).debug("Submitting the job through Job Client for action " + action.getId());
509    
510                    // setting up propagation of the delegation token.
511                    AuthHelper.get().set(jobClient, launcherJobConf);
512    
513                    runningJob = jobClient.submitJob(launcherJobConf);
514                    if (runningJob == null) {
515                        throw new ActionExecutorException(ActionExecutorException.ErrorType.ERROR, "JA017",
516                                                          "Error submitting launcher for action [{0}]", action.getId());
517                    }
518                    launcherId = runningJob.getID().toString();
519                    XLog.getLog(getClass()).debug("After submission get the launcherId " + launcherId);
520                }
521    
522                String jobTracker = launcherJobConf.get(HADOOP_JOB_TRACKER);
523                String consoleUrl = runningJob.getTrackingURL();
524                context.setStartData(launcherId, jobTracker, consoleUrl);
525            }
526            catch (Exception ex) {
527                exception = true;
528                throw convertException(ex);
529            }
530            finally {
531                if (jobClient != null) {
532                    try {
533                        jobClient.close();
534                    }
535                    catch (Exception e) {
536                        if (exception) {
537                            log.error("JobClient error: ", e);
538                        }
539                        else {
540                            throw convertException(e);
541                        }
542                    }
543                }
544            }
545        }

558        @Override
559        public void start(Context context, WorkflowAction action) throws ActionExecutorException {
560            try {
561                XLog.getLog(getClass()).debug("Starting action " + action.getId() + " getting Action File System");
562                FileSystem actionFs = getActionFileSystem(context, action);
563                XLog.getLog(getClass()).debug("Preparing action Dir through copying " + context.getActionDir());
564                prepareActionDir(actionFs, context);
565                XLog.getLog(getClass()).debug("Action Dir is ready. Submitting the action ");
566                submitLauncher(context, action);
567                XLog.getLog(getClass()).debug("Action submit completed. Performing check ");
568                check(context, action);
569                XLog.getLog(getClass()).debug("Action check is done after submission");
570            }
571            catch (Exception ex) {
572                throw convertException(ex);
573            }
574        }

576        @Override
577        public void end(Context context, WorkflowAction action) throws ActionExecutorException {
578            try {
579                String externalStatus = action.getExternalStatus();
580                WorkflowAction.Status status = externalStatus.equals(SUCCEEDED) ? WorkflowAction.Status.OK
581                                               : WorkflowAction.Status.ERROR;
582                context.setEndData(status, getActionSignal(status));
583            }
584            catch (Exception ex) {
585                throw convertException(ex);
586            }
587            finally {
588                try {
589                    FileSystem actionFs = getActionFileSystem(context, action);
590                    cleanUpActionDir(actionFs, context);
591                }
592                catch (Exception ex) {
593                    throw convertException(ex);
594                }
595            }
596        }

604        @Override
605        public void check(Context context, WorkflowAction action) throws ActionExecutorException {
606            JobClient jobClient = null;
607            boolean exception = false;
608            try {
609                Element actionXml = XmlUtils.parseXml(action.getConf());
610                FileSystem actionFs = getActionFileSystem(context, actionXml);
611                Configuration conf = createBaseHadoopConf(context, actionXml);
612                JobConf jobConf = new JobConf();
613                XConfiguration.copy(conf, jobConf);
614                jobClient = createJobClient(context, jobConf);
615                RunningJob runningJob = jobClient.getJob(JobID.forName(action.getExternalId()));
616                if (runningJob == null) {
617                    context.setExternalStatus(FAILED);
618                    context.setExecutionData(FAILED, null);
619                    throw new ActionExecutorException(ActionExecutorException.ErrorType.FAILED, "JA017",
620                                                      "Unknown hadoop job [{0}] associated with action [{1}].  Failing this action!", action
621                            .getExternalId(), action.getId());
622                }
623                if (runningJob.isComplete()) {
624                    Path actionDir = context.getActionDir();
625    
626                    String user = context.getWorkflow().getUser();
627                    String group = context.getWorkflow().getGroup();
628                    if (LauncherMapper.hasIdSwap(runningJob, user, group, actionDir)) {
629                        String launcherId = action.getExternalId();
630                        Path idSwapPath = LauncherMapper.getIdSwapPath(context.getActionDir());
631                        InputStream is = actionFs.open(idSwapPath);
632                        BufferedReader reader = new BufferedReader(new InputStreamReader(is));
633                        Properties props = PropertiesUtils.readProperties(reader, maxActionOutputLen);
634                        reader.close();
635                        String newId = props.getProperty("id");
636                        runningJob = jobClient.getJob(JobID.forName(newId));
637                        if (runningJob == null) {
638                            context.setExternalStatus(FAILED);
639                            throw new ActionExecutorException(ActionExecutorException.ErrorType.FAILED, "JA017",
640                                                              "Unknown hadoop job [{0}] associated with action [{1}].  Failing this action!", newId,
641                                                              action.getId());
642                        }
643    
644                        context.setStartData(newId, action.getTrackerUri(), runningJob.getTrackingURL());
645                        XLog.getLog(getClass()).info(XLog.STD, "External ID swap, old ID [{0}] new ID [{1}]", launcherId,
646                                                     newId);
647                    }
648                    if (runningJob.isComplete()) {
649                        XLog.getLog(getClass()).info(XLog.STD, "action completed, external ID [{0}]",
650                                                     action.getExternalId());
651                        if (runningJob.isSuccessful() && LauncherMapper.isMainSuccessful(runningJob)) {
652                            Properties props = null;
653                            if (getCaptureOutput(action)) {
654                                props = new Properties();
655                                if (LauncherMapper.hasOutputData(runningJob)) {
656                                    Path actionOutput = LauncherMapper.getOutputDataPath(context.getActionDir());
657                                    InputStream is = actionFs.open(actionOutput);
658                                    BufferedReader reader = new BufferedReader(new InputStreamReader(is));
659                                    props = PropertiesUtils.readProperties(reader, maxActionOutputLen);
660                                    reader.close();
661                                }
662                            }
663                            context.setExecutionData(SUCCEEDED, props);
664                            XLog.getLog(getClass()).info(XLog.STD, "action produced output");
665                        }
666                        else {
667                            XLog log = XLog.getLog(getClass());
668                            String errorReason;
669                            Path actionError = LauncherMapper.getErrorPath(context.getActionDir());
670                            if (actionFs.exists(actionError)) {
671                                InputStream is = actionFs.open(actionError);
672                                BufferedReader reader = new BufferedReader(new InputStreamReader(is));
673                                Properties props = PropertiesUtils.readProperties(reader, -1);
674                                reader.close();
675                                errorReason = props.getProperty("error.reason");
676                                log.warn("Launcher ERROR, reason: {0}", errorReason);
677                                String exMsg = props.getProperty("exception.message");
678                                context.setErrorInfo("JA018", exMsg);
679                                String exStackTrace = props.getProperty("exception.stacktrace");
680                                if (exMsg != null) {
681                                    log.warn("Launcher exception: {0}{E}{1}", exMsg, exStackTrace);
682                                }
683                            }
684                            else {
685                                errorReason = XLog.format("LauncherMapper died, check Hadoop log for job [{0}:{1}]", action
686                                        .getTrackerUri(), action.getExternalId());
687                                log.warn(errorReason);
688                            }
689                            context.setExecutionData(FAILED_KILLED, null);
690                        }
691                    }
692                    else {
693                        context.setExternalStatus(RUNNING);
694                        XLog.getLog(getClass()).info(XLog.STD, "checking action, external ID [{0}] status [{1}]",
695                                                     action.getExternalId(), action.getExternalStatus());
696                    }
697                }
698                else {
699                    context.setExternalStatus(RUNNING);
700                    XLog.getLog(getClass()).info(XLog.STD, "checking action, external ID [{0}] status [{1}]",
701                                                 action.getExternalId(), action.getExternalStatus());
702                }
703            }
704            catch (Exception ex) {
705                XLog.getLog(getClass()).warn("Exception in check(). Message[{0}]", ex.getMessage(), ex);
706                exception = true;
707                throw convertException(ex);
708            }
709            finally {
710                if (jobClient != null) {
711                    try {
712                        jobClient.close();
713                    }
714                    catch (Exception e) {
715                        if (exception) {
716                            log.error("JobClient error: ", e);
717                        }
718                        else {
719                            throw convertException(e);
720                        }
721                    }
722                }
723            }
724        }

726        protected boolean getCaptureOutput(WorkflowAction action) throws JDOMException {
727            Element eConf = XmlUtils.parseXml(action.getConf());
728            Namespace ns = eConf.getNamespace();
729            Element captureOutput = eConf.getChild("capture-output", ns);
730            return captureOutput != null;
731        }

733        @Override
734        public void kill(Context context, WorkflowAction action) throws ActionExecutorException {
735            JobClient jobClient = null;
736            boolean exception = false;
737            try {
738                Element actionXml = XmlUtils.parseXml(action.getConf());
739                Configuration conf = createBaseHadoopConf(context, actionXml);
740                JobConf jobConf = new JobConf();
741                XConfiguration.copy(conf, jobConf);
742                jobClient = createJobClient(context, jobConf);
743                RunningJob runningJob = jobClient.getJob(JobID.forName(action.getExternalId()));
744                if (runningJob != null) {
745                    runningJob.killJob();
746                }
747                context.setExternalStatus(KILLED);
748                context.setExecutionData(KILLED, null);
749            }
750            catch (Exception ex) {
751                exception = true;
752                throw convertException(ex);
753            }
754            finally {
755                try {
756                    FileSystem actionFs = getActionFileSystem(context, action);
757                    cleanUpActionDir(actionFs, context);
758                    if (jobClient != null) {
759                        jobClient.close();
760                    }
761                }
762                catch (Exception ex) {
763                    if (exception) {
764                        log.error("Error: ", ex);
765                    }
766                    else {
767                        throw convertException(ex);
768                    }
769                }
770            }
771        }
