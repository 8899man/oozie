266         * Return CoordinatorActions for a jobID. Action should be in READY state. Number of returned actions should be <=
267         * concurrency number. Sort returned actions based on execution order (FIFO, LIFO, LAST_ONLY)
268         *
269         * @param id job ID
270         * @param numResults number of results to return
271         * @param executionOrder execution for this job - FIFO, LIFO, LAST_ONLY
272         * @return List of CoordinatorActionBean
273         * @throws StoreException
274         */
275        public List<CoordinatorActionBean> getCoordinatorActionsForJob(final String id, final int numResults,
276                                                                       final String executionOrder) throws StoreException {
277            ParamChecker.notEmpty(id, "jobID");
278            List<CoordinatorActionBean> caBeans = doOperation("getCoordinatorActionsForJob",
279                                                              new Callable<List<CoordinatorActionBean>>() {
280                                                                  public List<CoordinatorActionBean> call() throws StoreException {
281    
282                                                                      List<CoordinatorActionBean> caBeans;
283                                                                      Query q;
284                                                                      // check if executionOrder is FIFO, LIFO, or LAST_ONLY
285                                                                      if (executionOrder.equalsIgnoreCase("FIFO")) {
286                                                                          q = entityManager.createNamedQuery("GET_COORD_ACTIONS_FOR_JOB_FIFO");
287                                                                      }
288                                                                      else {
289                                                                          q = entityManager.createNamedQuery("GET_COORD_ACTIONS_FOR_JOB_LIFO");
290                                                                      }
291                                                                      q.setParameter("jobId", id);
292                                                                      // if executionOrder is LAST_ONLY, only retrieve first
293                                                                      // record in LIFO,
294                                                                      // otherwise, use numResults if it is positive.
295                                                                      if (executionOrder.equalsIgnoreCase("LAST_ONLY")) {
296                                                                          q.setMaxResults(1);
297                                                                      }
298                                                                      else {
299                                                                          if (numResults > 0) {
300                                                                              q.setMaxResults(numResults);
301                                                                          }
302                                                                      }
303                                                                      caBeans = q.getResultList();
304                                                                      return caBeans;
305                                                                  }
306                                                              });
307            return caBeans;
308        }

401        private <V> V doOperation(String name, Callable<V> command) throws StoreException {
402            try {
403                Instrumentation.Cron cron = new Instrumentation.Cron();
404                cron.start();
405                V retVal;
406                try {
407                    retVal = command.call();
408                }
409                finally {
410                    cron.stop();
411                }
412                Services.get().get(InstrumentationService.class).get().addCron(INSTR_GROUP, name, cron);
413                return retVal;
414            }
415            catch (StoreException ex) {
416                throw ex;
417            }
418            catch (SQLException ex) {
419                throw new StoreException(ErrorCode.E0603, name, ex.getMessage(), ex);
420            }
421            catch (Exception e) {
422                throw new StoreException(ErrorCode.E0607, name, e.getMessage(), e);
423            }
424        }

502        public int purgeJobs(final long olderThanDays, final long limit) throws StoreException {
503    
504            Integer count = doOperation("coord-purge-jobs", new Callable<Integer>() {
505                public Integer call() throws SQLException, StoreException, WorkflowException {
506    
507                    Timestamp lastModTm = new Timestamp(System.currentTimeMillis() - (olderThanDays * DAY_IN_MS));
508    
509                    Query jobQ = entityManager.createNamedQuery("GET_COMPLETED_COORD_JOBS_OLDER_THAN_STATUS");
510                    jobQ.setParameter("lastModTime", lastModTm);
511                    jobQ.setMaxResults((int) limit);
512                    List<CoordinatorJobBean> coordJobs = jobQ.getResultList();
513                    int deleted = 0;
514                    for (CoordinatorJobBean a : coordJobs) {
515                        String jobId = a.getId();
516    
517                        Query actionQ = entityManager.createNamedQuery("GET_COORD_ACTIONS_COUNT_BY_JOBID");
518                        actionQ.setParameter("jobId", jobId);
519                        Long count = (Long) actionQ.getSingleResult();
520    
521                        if (count.intValue() == 0) {
522                            // remove surely removes - but expensive - to be
523                            // compared?
524                            entityManager.remove(a);
525                            deleted++;
526                        }
527                    }
528    
529                    return deleted;
530                }
531            });
532            return Integer.valueOf(count);
533        }

546        public CoordinatorJobInfo getCoordinatorInfo(final Map<String, List<String>> filter, final int start, final int len)
547                throws StoreException {
548    
549            CoordinatorJobInfo coordJobInfo = doOperation("getCoordinatorJobInfo", new Callable<CoordinatorJobInfo>() {
550                public CoordinatorJobInfo call() throws SQLException, StoreException {
551                    List<String> orArray = new ArrayList<String>();
552                    List<String> colArray = new ArrayList<String>();
553                    List<String> valArray = new ArrayList<String>();
554                    StringBuilder sb = new StringBuilder("");
555    
556                    StoreStatusFilter.filter(filter, orArray, colArray, valArray, sb, StoreStatusFilter.coordSeletStr,
557                                             StoreStatusFilter.coordCountStr);
558    
559                    int realLen = 0;
560    
561                    Query q = null;
562                    Query qTotal = null;
563                    if (orArray.size() == 0) {
564                        q = entityManager.createNamedQuery("GET_COORD_JOBS_COLUMNS");
565                        q.setFirstResult(start - 1);
566                        q.setMaxResults(len);
567                        qTotal = entityManager.createNamedQuery("GET_COORD_JOBS_COUNT");
568                    }
569                    else {
570                        StringBuilder sbTotal = new StringBuilder(sb);
571                        sb.append(" order by w.createdtime desc ");
572                        XLog.getLog(getClass()).debug("Created String is **** " + sb.toString());
573                        q = entityManager.createQuery(sb.toString());
574                        q.setFirstResult(start - 1);
575                        q.setMaxResults(len);
576                        qTotal = entityManager.createQuery(sbTotal.toString().replace(StoreStatusFilter.coordSeletStr,
577                                                                                      StoreStatusFilter.coordCountStr));
578                    }
579    
580                    for (int i = 0; i < orArray.size(); i++) {
581                        q.setParameter(colArray.get(i), valArray.get(i));
582                        qTotal.setParameter(colArray.get(i), valArray.get(i));
583                    }
584    
585                    OpenJPAQuery kq = OpenJPAPersistence.cast(q);
586                    JDBCFetchPlan fetch = (JDBCFetchPlan) kq.getFetchPlan();
587                    fetch.setFetchBatchSize(20);
588                    fetch.setResultSetType(ResultSetType.SCROLL_INSENSITIVE);
589                    fetch.setFetchDirection(FetchDirection.FORWARD);
590                    fetch.setLRSSizeAlgorithm(LRSSizeAlgorithm.LAST);
591                    List<?> resultList = q.getResultList();
592                    List<Object[]> objectArrList = (List<Object[]>) resultList;
593                    List<CoordinatorJobBean> coordBeansList = new ArrayList<CoordinatorJobBean>();
594    
595                    for (Object[] arr : objectArrList) {
596                        CoordinatorJobBean ww = getBeanForCoordinatorJobFromArray(arr);
597                        coordBeansList.add(ww);
598                    }
599    
600                    realLen = ((Long) qTotal.getSingleResult()).intValue();
601    
602                    return new CoordinatorJobInfo(coordBeansList, start, len, realLen);
603                }
604            });
605            return coordJobInfo;
606        }

608        private CoordinatorJobBean getBeanForCoordinatorJobFromArray(Object[] arr) {
